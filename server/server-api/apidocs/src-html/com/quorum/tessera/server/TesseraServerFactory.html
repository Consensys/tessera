001package com.quorum.tessera.server;
002
003import com.quorum.tessera.config.CommunicationType;
004import com.quorum.tessera.config.ServerConfig;
005
006import java.util.ArrayList;
007import java.util.List;
008import java.util.ServiceLoader;
009import java.util.Set;
010import org.slf4j.Logger;
011import org.slf4j.LoggerFactory;
012
013public interface TesseraServerFactory<T> {
014
015    TesseraServer createServer(ServerConfig config, Set<T> services);
016
017    Logger LOGGER = LoggerFactory.getLogger(TesseraServerFactory.class);
018    
019    static TesseraServerFactory create(CommunicationType communicationType) {
020
021        if(communicationType == CommunicationType.WEB_SOCKET) {
022            throw new UnsupportedOperationException("Websockets are not supported yet");
023        }
024
025        List<TesseraServerFactory> all = new ArrayList<>();
026        ServiceLoader.load(TesseraServerFactory.class).forEach(all::add);
027
028        return all.stream()
029                .filter(f -> f.communicationType() == communicationType)
030                .findFirst()
031                .orElseThrow(() -> new IllegalStateException("No server factory found for "+ communicationType));
032    }
033
034    CommunicationType communicationType();
035    
036}
