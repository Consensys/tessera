001package com.quorum.tessera.server.monitoring;
002
003public class MBeanResourceMetric implements MBeanMetric {
004    private String resourceMethod;
005    private String name;
006    private String value;
007
008    public MBeanResourceMetric(String resourceMethod, String name, String value) {
009        this.resourceMethod = resourceMethod;
010        this.name = name;
011        this.value = value;
012    }
013
014    public String getResourceMethod() {
015        return resourceMethod;
016    }
017
018    public String getName() {
019        return name;
020    }
021
022    public String getValue() {
023        return value;
024    }
025}
