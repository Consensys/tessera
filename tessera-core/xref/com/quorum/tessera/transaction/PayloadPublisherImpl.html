1   package com.quorum.tessera.transaction;
2   
3   import com.quorum.tessera.client.P2pClient;
4   import com.quorum.tessera.enclave.Enclave;
5   import com.quorum.tessera.enclave.EncodedPayload;
6   import com.quorum.tessera.enclave.PayloadEncoder;
7   import com.quorum.tessera.encryption.PublicKey;
8   import com.quorum.tessera.node.PartyInfoService;
9   import com.quorum.tessera.transaction.exception.PublishPayloadException;
10  import org.slf4j.Logger;
11  import org.slf4j.LoggerFactory;
12  
13  import java.util.Objects;
14  
15  public class PayloadPublisherImpl implements PayloadPublisher {
16  
17      private static final Logger LOGGER = LoggerFactory.getLogger(PayloadPublisherImpl.class);
18  
19      private final PayloadEncoder payloadEncoder;
20  
21      private final PartyInfoService partyInfoService;
22  
23      private final P2pClient p2pClient;
24  
25      private final Enclave enclave;
26  
27      public PayloadPublisherImpl(final PayloadEncoder payloadEncoder,
28                                  final PartyInfoService partyInfoService,
29                                  final P2pClient p2pClient,
30                                  final Enclave enclave) {
31          this.payloadEncoder = Objects.requireNonNull(payloadEncoder);
32          this.partyInfoService = Objects.requireNonNull(partyInfoService);
33          this.p2pClient = Objects.requireNonNull(p2pClient);
34          this.enclave = Objects.requireNonNull(enclave);
35      }
36  
37      @Override
38      public void publishPayload(final EncodedPayload payload, final PublicKey recipientKey) {
39  
40          if (enclave.getPublicKeys().contains(recipientKey)) {
41              //we are trying to send something to ourselves - don't do it
42              LOGGER.debug("Trying to send message to ourselves with key {}, not publishing", recipientKey.encodeToBase64());
43              return;
44          }
45  
46          final String targetUrl = partyInfoService.getURLFromRecipientKey(recipientKey);
47  
48          LOGGER.info("Publishing message to {}", targetUrl);
49  
50          final byte[] encoded = payloadEncoder.encode(payload);
51  
52          byte[] pushResponse = p2pClient.push(targetUrl, encoded);
53  
54          if(pushResponse == null) {
55              throw new PublishPayloadException("Unable to push payload to recipient " + recipientKey.encodeToBase64());
56          }
57  
58          LOGGER.info("Published to {}", targetUrl);
59      }
60  
61  }
