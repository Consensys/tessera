56      public TransactionManagerImpl(
57              Base64Decoder base64Decoder,
58              PayloadEncoder payloadEncoder,
59              EncryptedTransactionDAO encryptedTransactionDAO,
60              PayloadPublisher payloadPublisher,

74      @Override
75      @Transactional
76      public SendResponse send(SendRequest sendRequest) {
77  
78          final String sender = sendRequest.getFrom();
79  
80          final PublicKey senderPublicKey = Optional.ofNullable(sender)
81                  .map(base64Decoder::decode)
82                  .map(PublicKey::from)
83                  .orElseGet(enclave::defaultPublicKey);
84  
85          final byte[][] recipients = Stream.of(sendRequest)
86                  .filter(sr -> Objects.nonNull(sr.getTo()))
87                  .flatMap(s -> Stream.of(s.getTo()))
88                  .map(base64Decoder::decode)
89                  .toArray(byte[][]::new);
90  
91          final List<PublicKey> recipientList = Stream
92                  .of(recipients)
93                  .map(PublicKey::from)
94                  .collect(Collectors.toList());
95  
96          recipientList.add(senderPublicKey);
97  
98          recipientList.addAll(enclave.getForwardingKeys());
99  
100         final byte[] raw = sendRequest.getPayload();
101 
102         final EncodedPayload payload = enclave.encryptPayload(raw, senderPublicKey, recipientList);
103 
104         final MessageHash transactionHash = Optional.of(payload)
105                 .map(EncodedPayload::getCipherText)
106                 .map(messageHashFactory::createFromCipherText).get();
107 
108         final EncryptedTransaction newTransaction
109             = new EncryptedTransaction(transactionHash, this.payloadEncoder.encode(payload));
110 
111         this.encryptedTransactionDAO.save(newTransaction);
112 
113         recipientList.forEach(recipient -> {
114             final EncodedPayload outgoing = payloadEncoder.forRecipient(payload, recipient);
115             payloadPublisher.publishPayload(outgoing, recipient);
116         });
117 
118         final byte[] key = transactionHash.getHashBytes();
119 
120         final String encodedKey = base64Decoder.encodeToString(key);
121 
122         return new SendResponse(encodedKey);
123     }

125     @Override
126     @Transactional
127     public SendResponse sendSignedTransaction(SendSignedRequest sendRequest) {
128 
129         final byte[][] recipients = Stream.of(sendRequest)
130             .filter(sr -> Objects.nonNull(sr.getTo()))
131             .flatMap(s -> Stream.of(s.getTo()))
132             .map(base64Decoder::decode)
133             .toArray(byte[][]::new);
134 
135         final List<PublicKey> recipientList = Stream
136             .of(recipients)
137             .map(PublicKey::from)
138             .collect(Collectors.toList());
139 
140         recipientList.addAll(enclave.getForwardingKeys());
141 
142         MessageHash messageHash = new MessageHash(sendRequest.getHash());
143 
144         EncryptedRawTransaction encryptedRawTransaction = encryptedRawTransactionDAO.retrieveByHash(messageHash)
145             .orElseThrow(() -> new TransactionNotFoundException("Raw Transaction with hash " + messageHash + " was not found"));
146 
147         final EncodedPayload payload
148             = enclave.encryptPayload(encryptedRawTransaction.toRawTransaction(), recipientList);
149 
150 
151         final EncryptedTransaction newTransaction
152             = new EncryptedTransaction(messageHash, this.payloadEncoder.encode(payload));
153 
154         this.encryptedTransactionDAO.save(newTransaction);
155 
156         recipientList.forEach(recipient -> {
157             final EncodedPayload toPublish = payloadEncoder.forRecipient(payload, recipient);
158             payloadPublisher.publishPayload(toPublish, recipient);
159         });
160 
161         final byte[] key = messageHash.getHashBytes();
162 
163         final String encodedKey = base64Decoder.encodeToString(key);
164 
165         return new SendResponse(encodedKey);
166     }

168     @Override
169     @Transactional
170     public ResendResponse resend(ResendRequest request) {
171 
172         final byte[] publicKeyData = base64Decoder.decode(request.getPublicKey());
173         PublicKey recipientPublicKey = PublicKey.from(publicKeyData);
174         if (request.getType() == ResendRequestType.ALL) {
175 
176             encryptedTransactionDAO
177                 .retrieveAllTransactions()
178                 .stream()
179                 .map(EncryptedTransaction::getEncodedPayload)
180                 .map(payloadEncoder::decode)
181                 .filter(payload -> {
182                     final boolean isRecipient = payload.getRecipientKeys().contains(recipientPublicKey);
183                     final boolean isSender = Objects.equals(payload.getSenderKey(), recipientPublicKey);
184                     return isRecipient || isSender;
185                 }).forEach(payload -> {
186                     if (Objects.equals(payload.getSenderKey(), recipientPublicKey)) {
187                         final PublicKey decryptedKey = searchForRecipientKey(payload).orElseThrow(
188                             () -> {
189                                 final MessageHash hash = MessageHashFactory.create()
190                                     .createFromCipherText(payload.getCipherText());
191                                 return new KeyNotFoundException("No key found as recipient of message " + hash);
192                             }
193                         );
194                         payload.getRecipientKeys().add(decryptedKey);
195                     }
196 
197                     try {
198                         payloadPublisher.publishPayload(payload, recipientPublicKey);
199                     } catch (PublishPayloadException ex) {
200                         LOGGER.warn("Unable to publish payload to recipient {} during resend", recipientPublicKey.encodeToBase64());
201                     }
202 
203                 });
204 
205             return new ResendResponse();
206         } else {
207 
208             final byte[] hashKey = base64Decoder.decode(request.getKey());
209             final MessageHash messageHash = new MessageHash(hashKey);
210 
211             final EncryptedTransaction encryptedTransaction = encryptedTransactionDAO
212                 .retrieveByHash(messageHash)
213                 .orElseThrow(() -> new TransactionNotFoundException("Message with hash " + messageHash + " was not found"));
214 
215             final EncodedPayload payload = payloadEncoder.decode(encryptedTransaction.getEncodedPayload());
216 
217             final EncodedPayload returnValue;
218             if (Objects.equals(payload.getSenderKey(), recipientPublicKey)) {
219                 final PublicKey decryptedKey = searchForRecipientKey(payload).orElseThrow(RuntimeException::new);
220                 payload.getRecipientKeys().add(decryptedKey);
221                 returnValue = payload;
222             } else {
223                 //this is our tx
224                 returnValue = payloadEncoder.forRecipient(payload, recipientPublicKey);
225             }
226 
227             return new ResendResponse(payloadEncoder.encode(returnValue));
228         }
229     }

231     @Override
232     public MessageHash storePayload(byte[] input) {
233 
234         final EncodedPayload payload = payloadEncoder.decode(input);
235 
236         final MessageHash transactionHash = Optional.of(payload)
237             .map(EncodedPayload::getCipherText)
238             .map(messageHashFactory::createFromCipherText).get();
239 
240         if (enclave.getPublicKeys().contains(payload.getSenderKey())) {
241 
242             this.resendManager.acceptOwnMessage(input);
243 
244         } else {
245 
246             //this is a tx from someone else
247             this.encryptedTransactionDAO.save(new EncryptedTransaction(transactionHash, input));
248             LOGGER.info("Stored payload with hash {}", transactionHash);
249 
250         }
251 
252         return transactionHash;
253     }

266     @Override
267     @Transactional
268     public ReceiveResponse receive(ReceiveRequest request) {
269 
270         final byte[] key = base64Decoder.decode(request.getKey());
271 
272         final Optional<byte[]> to = Optional
273                 .ofNullable(request.getTo())
274                 .filter(str -> !str.isEmpty())
275                 .map(base64Decoder::decode);
276 
277         final MessageHash hash = new MessageHash(key);
278         LOGGER.info("Lookup transaction {}",hash);
279         
280         final EncryptedTransaction encryptedTransaction = encryptedTransactionDAO
281                 .retrieveByHash(hash)
282                 .orElseThrow(() -> new TransactionNotFoundException("Message with hash " + hash + " was not found"));
283 
284         final EncodedPayload payload = Optional.of(encryptedTransaction)
285                 .map(EncryptedTransaction::getEncodedPayload)
286                 .map(payloadEncoder::decode)
287                 .orElseThrow(() -> new IllegalStateException("Unable to decode previously encoded payload"));
288 
289         PublicKey recipientKey = to.map(PublicKey::from)
290             .orElse(searchForRecipientKey(payload)
291                 .orElseThrow(() -> new NoRecipientKeyFoundException("No suitable recipient keys found to decrypt payload for : " + hash))
292             );
293 
294         byte[] response = enclave.unencryptTransaction(payload, recipientKey);
295 
296         return new ReceiveResponse(response);
297 
298     }
299 
300     private Optional<PublicKey> searchForRecipientKey(final EncodedPayload payload) {
301         for (final PublicKey potentialMatchingKey : enclave.getPublicKeys()) {
302             try {
303                 enclave.unencryptTransaction(payload, potentialMatchingKey);
304                 return Optional.of(potentialMatchingKey);
305             } catch (EnclaveException | IndexOutOfBoundsException | NaclException ex) {
306                 LOGGER.debug("Attempted payload decryption using wrong key, discarding.");
307             }
308         }
309         return Optional.empty();
310     }

312     @Override
313     @Transactional
314     public StoreRawResponse store(StoreRawRequest storeRequest) {
315 
316         RawTransaction rawTransaction = enclave.encryptRawPayload(storeRequest.getPayload(),
317             storeRequest.getFrom().map(PublicKey::from).orElseGet(enclave::defaultPublicKey));
318         MessageHash hash = messageHashFactory.createFromCipherText(rawTransaction.getEncryptedPayload());
319 
320         EncryptedRawTransaction encryptedRawTransaction = new EncryptedRawTransaction(hash,
321             rawTransaction.getEncryptedPayload(),
322             rawTransaction.getEncryptedKey(),
323             rawTransaction.getNonce().getNonceBytes(),
324             rawTransaction.getFrom().getKeyBytes());
325 
326         encryptedRawTransactionDAO.save(encryptedRawTransaction);
327 
328         return new StoreRawResponse(encryptedRawTransaction.getHash().getHashBytes());
329     }
