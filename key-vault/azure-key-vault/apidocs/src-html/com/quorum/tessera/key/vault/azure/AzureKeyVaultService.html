001package com.quorum.tessera.key.vault.azure;
002
003import com.microsoft.azure.keyvault.models.SecretBundle;
004import com.microsoft.azure.keyvault.requests.SetSecretRequest;
005import com.quorum.tessera.config.AzureKeyVaultConfig;
006import com.quorum.tessera.config.vault.data.AzureGetSecretData;
007import com.quorum.tessera.config.vault.data.AzureSetSecretData;
008import com.quorum.tessera.config.vault.data.GetSecretData;
009import com.quorum.tessera.config.vault.data.SetSecretData;
010import com.quorum.tessera.key.vault.KeyVaultException;
011import com.quorum.tessera.key.vault.KeyVaultService;
012import com.quorum.tessera.key.vault.VaultSecretNotFoundException;
013
014import java.util.Objects;
015
016public class AzureKeyVaultService implements KeyVaultService {
017    private String vaultUrl;
018    private AzureKeyVaultClientDelegate azureKeyVaultClientDelegate;
019
020    AzureKeyVaultService(AzureKeyVaultConfig keyVaultConfig, AzureKeyVaultClientDelegate azureKeyVaultClientDelegate) {
021        Objects.requireNonNull(keyVaultConfig);
022        Objects.requireNonNull(azureKeyVaultClientDelegate);
023
024        this.vaultUrl = keyVaultConfig.getUrl();
025        this.azureKeyVaultClientDelegate = azureKeyVaultClientDelegate;
026    }
027
028    @Override
029    public String getSecret(GetSecretData getSecretData) {
030        if(!(getSecretData instanceof AzureGetSecretData)) {
031            throw new KeyVaultException("Incorrect data type passed to AzureKeyVaultService.  Type was " + getSecretData.getType());
032        }
033
034        AzureGetSecretData azureGetSecretData = (AzureGetSecretData) getSecretData;
035
036        SecretBundle secretBundle;
037
038        if(azureGetSecretData.getSecretVersion() != null) {
039            secretBundle = azureKeyVaultClientDelegate.getSecret(vaultUrl, azureGetSecretData.getSecretName(), azureGetSecretData.getSecretVersion());
040        }
041        else {
042            secretBundle = azureKeyVaultClientDelegate.getSecret(vaultUrl, azureGetSecretData.getSecretName());
043        }
044
045        if(secretBundle == null) {
046            throw new VaultSecretNotFoundException("Azure Key Vault secret " + azureGetSecretData.getSecretName() + " was not found in vault " + vaultUrl);
047        }
048
049        return secretBundle.value();
050    }
051
052    @Override
053    public Object setSecret(SetSecretData setSecretData) {
054        if(!(setSecretData instanceof AzureSetSecretData)) {
055            throw new KeyVaultException("Incorrect data type passed to AzureKeyVaultService.  Type was " + setSecretData.getType());
056        }
057
058        AzureSetSecretData azureSetSecretData = (AzureSetSecretData) setSecretData;
059
060        SetSecretRequest setSecretRequest = new SetSecretRequest.Builder(vaultUrl, azureSetSecretData.getSecretName(), azureSetSecretData.getSecret()).build();
061
062        return this.azureKeyVaultClientDelegate.setSecret(setSecretRequest);
063    }
064}
