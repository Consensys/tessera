001package com.quorum.tessera.ssl.context;
002
003import com.quorum.tessera.config.SslConfig;
004import com.quorum.tessera.config.util.EnvironmentVariableProvider;
005import com.quorum.tessera.config.util.EnvironmentVariableProviderFactory;
006import com.quorum.tessera.config.util.EnvironmentVariables;
007import com.quorum.tessera.ssl.context.model.SSLContextProperties;
008import com.quorum.tessera.ssl.exception.TesseraSecurityException;
009import com.quorum.tessera.ssl.strategy.TrustMode;
010import org.bouncycastle.operator.OperatorCreationException;
011
012import javax.net.ssl.SSLContext;
013import java.io.IOException;
014import java.nio.file.Path;
015import java.nio.file.Paths;
016import java.security.GeneralSecurityException;
017import java.util.Optional;
018
019public class ClientSSLContextFactoryImpl implements ClientSSLContextFactory {
020
021    private static final String DEFAULT_KNOWN_SERVER_FILEPATH = "knownServers";
022
023    private static final EnvironmentVariableProvider envVarProvider = EnvironmentVariableProviderFactory.load().create();
024
025    @Override
026    public SSLContext from(String address, SslConfig sslConfig) {
027
028        TrustMode trustMode = TrustMode
029            .getValueIfPresent(sslConfig.getClientTrustMode().name())
030            .orElse(TrustMode.NONE);
031
032        final Path knownServersFile = Optional.ofNullable(sslConfig.getKnownServersFile())
033            .orElse(Paths.get(DEFAULT_KNOWN_SERVER_FILEPATH));
034
035        final SSLContextProperties properties = new SSLContextProperties(
036            address,
037            sslConfig.getClientKeyStore(),
038            getClientKeyStorePassword(sslConfig),
039            sslConfig.getClientTlsKeyPath(),
040            sslConfig.getClientTlsCertificatePath(),
041            sslConfig.getClientTrustStore(),
042            getClientTrustStorePassword(sslConfig),
043            sslConfig.getClientTrustCertificates(),
044            knownServersFile
045        );
046
047        try {
048            return trustMode.createSSLContext(properties);
049        } catch (IOException | OperatorCreationException | GeneralSecurityException ex) {
050            throw new TesseraSecurityException(ex);
051        }
052    }
053
054    // TODO - Package private for testing, refactor so this can be made private
055    String getClientKeyStorePassword(SslConfig sslConfig) {
056        return getPreferredPassword(sslConfig.getClientKeyStorePassword(), sslConfig.getEnvironmentVariablePrefix(), EnvironmentVariables.CLIENT_KEYSTORE_PWD);
057    }
058
059    // TODO - Package private for testing, refactor so this can be made private
060    String getClientTrustStorePassword(SslConfig sslConfig) {
061        return getPreferredPassword(sslConfig.getClientTrustStorePassword(), sslConfig.getEnvironmentVariablePrefix(), EnvironmentVariables.CLIENT_TRUSTSTORE_PWD);
062    }
063
064    // Return the prefixed env var value if set, else return the config value, else return the global env var value
065    private String getPreferredPassword(String configPassword, String envVarPrefix, String envVar) {
066        String password = envVarProvider.getEnv(envVarPrefix + "_" + envVar);
067
068        if(password != null) {
069            return password;
070        } else if(configPassword != null) {
071            return configPassword;
072        }
073
074        return envVarProvider.getEnv(envVar);
075    }
076
077}
