001package com.quorum.tessera.ssl.context;
002
003import com.quorum.tessera.ssl.trust.CompositeTrustManager;
004import com.quorum.tessera.ssl.trust.TrustAllManager;
005import com.quorum.tessera.ssl.trust.TrustOnFirstUseManager;
006import com.quorum.tessera.ssl.trust.WhiteListTrustManager;
007import com.quorum.tessera.ssl.util.TlsUtils;
008import org.bouncycastle.operator.OperatorCreationException;
009
010import javax.net.ssl.KeyManager;
011import javax.net.ssl.SSLContext;
012import javax.net.ssl.TrustManager;
013import java.io.IOException;
014import java.nio.file.Path;
015import java.security.GeneralSecurityException;
016import java.security.NoSuchAlgorithmException;
017import java.security.SecureRandom;
018import java.util.Arrays;
019import java.util.List;
020import java.util.Objects;
021
022public class SSLContextBuilder {
023
024    private static final String PROTOCOL = "TLS";
025
026    private String address;
027    private Path keyStore;
028    private String keyStorePassword;
029    private Path key;
030    private Path certificate;
031    private Path trustStore;
032    private String trustStorePassword;
033    private List<Path> trustedCertificates;
034
035    private SSLContext sslContext;
036
037    private SSLContextBuilder(String address,
038                              Path keyStore,
039                              String keyStorePassword,
040                              Path trustStore,
041                              String trustStorePassword) throws NoSuchAlgorithmException {
042        this.address = address;
043        this.keyStore = keyStore;
044        this.keyStorePassword = keyStorePassword;
045        this.trustStore = trustStore;
046        this.trustStorePassword = trustStorePassword;
047
048        this.sslContext = SSLContext.getInstance(PROTOCOL);
049    }
050
051
052    public static SSLContextBuilder createBuilder(String address, Path keyStore, String keyStorePassword, Path trustStore, String trustStorePassword) throws NoSuchAlgorithmException {
053        return new SSLContextBuilder(
054            address,
055            keyStore,
056            keyStorePassword,
057            trustStore,
058            trustStorePassword);
059    }
060
061    public SSLContextBuilder fromPemFiles(Path key, Path certificate, List<Path> trustedCertificates) {
062        this.key = key;
063        this.certificate = certificate;
064        this.trustedCertificates = trustedCertificates;
065
066        return this;
067    }
068
069    public SSLContext build() {
070        return sslContext;
071    }
072
073
074    public SSLContextBuilder forWhiteList(Path knownHosts) throws GeneralSecurityException, IOException, OperatorCreationException {
075
076        sslContext.init(buildKeyManagers(), new TrustManager[]{new WhiteListTrustManager(knownHosts)}, null);
077
078        return this;
079    }
080
081
082    public SSLContextBuilder forCASignedCertificates() throws GeneralSecurityException, IOException, OperatorCreationException {
083
084        final KeyManager[] keyManagers = buildKeyManagers();
085
086        final TrustManager[] trustManagers = buildTrustManagers();
087
088        sslContext.init(keyManagers, trustManagers, new SecureRandom());
089
090        return this;
091    }
092
093
094    public SSLContextBuilder forAllCertificates() throws GeneralSecurityException, IOException, OperatorCreationException {
095
096        sslContext.init(buildKeyManagers(), new TrustManager[]{new TrustAllManager()}, null);
097
098        return this;
099    }
100
101
102    public SSLContextBuilder forTrustOnFirstUse(Path knownHostsFile) throws GeneralSecurityException, IOException, OperatorCreationException {
103
104        final KeyManager[] keyManagers = buildKeyManagers();
105
106        sslContext.init(keyManagers,
107            new TrustManager[]{new TrustOnFirstUseManager(knownHostsFile)}, null);
108
109        return this;
110    }
111
112    public SSLContextBuilder forCAOrTOFU(Path knownHostsFile) throws GeneralSecurityException, IOException, OperatorCreationException {
113
114        final KeyManager[] keyManagers = buildKeyManagers();
115
116        final TrustManager[] trustManagersFromTrustStore = buildTrustManagers();
117
118        final int newLength = trustManagersFromTrustStore.length + 1;
119
120        final TrustManager[] trustManagers = Arrays.copyOf(trustManagersFromTrustStore, newLength);
121
122        trustManagers[newLength - 1] = new TrustOnFirstUseManager(knownHostsFile);
123
124        sslContext.init(keyManagers, new TrustManager[]{new CompositeTrustManager(trustManagers)}, new SecureRandom());
125
126        return this;
127
128    }
129
130
131    private KeyManager[] buildKeyManagers() throws GeneralSecurityException, IOException, OperatorCreationException {
132
133        if (Objects.nonNull(this.keyStore)) {
134            if (!this.keyStore.toFile().exists()) {
135                TlsUtils.create().generateKeyStoreWithSelfSignedCertificate(this.address, this.keyStore, this.keyStorePassword);
136            }
137            return SSLKeyStoreLoader.fromJksKeyStore(this.keyStore, this.keyStorePassword);
138        } else {
139            return SSLKeyStoreLoader.fromPemKeyFile(this.key, this.certificate);
140        }
141    }
142
143
144    private TrustManager[] buildTrustManagers() throws GeneralSecurityException, IOException {
145
146        if (Objects.nonNull(this.trustStore)) {
147            return SSLKeyStoreLoader.fromJksTrustStore(this.trustStore, this.trustStorePassword);
148        } else {
149            return SSLKeyStoreLoader.fromPemCertificatesFile(this.trustedCertificates);
150        }
151    }
152
153}
