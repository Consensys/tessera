209    static Path toPath(String workDir, String value) {
210        final Path path;
211
212        if(Optional.ofNullable(workDir).isPresent() && Optional.ofNullable(value).isPresent()) {
213            path = Paths.get(workDir, value);
214        } else if(Optional.ofNullable(value).isPresent()) {
215            path = Paths.get(value);
216        } else {
217            path = null;
218        }
219
220        return path;
221    }

223    public Config build() {
224
225        boolean generateKeyStoreIfNotExisted = false;
226
227        SslConfig sslConfig = new SslConfig(
228                sslAuthenticationMode,
229                generateKeyStoreIfNotExisted,
230                toPath(workDir, sslServerKeyStorePath),
231                sslServerKeyStorePassword,
232                toPath(workDir, sslServerTrustStorePath),
233                sslServerTrustStorePassword,
234                sslServerTrustMode,
235                toPath(workDir, sslClientKeyStorePath),
236                sslClientKeyStorePassword,
237                toPath(workDir, sslClientTrustStorePath),
238                sslClientTrustStorePassword,
239                sslClientTrustMode,
240                toPath(workDir, sslKnownClientsFile),
241                toPath(workDir, sslKnownServersFile),
242                sslServerTrustCertificates.stream()
243                        .filter(Objects::nonNull)
244                        .map(v -> toPath(workDir, v))
245                        .collect(Collectors.toList()),
246                sslClientTrustCertificates.stream()
247                        .filter(Objects::nonNull)
248                        .map(v -> toPath(workDir, v))
249                        .collect(Collectors.toList()),
250                toPath(workDir, sslServerTlsKeyPath),
251                toPath(workDir, sslServerTlsCertificatePath),
252                toPath(workDir, sslClientTlsKeyPath),
253                toPath(workDir, sslClientTlsCertificatePath),
254                null
255        );
256
257
258
259        //TODO must add P2P and Q2T server configs. Maybe ThirdParty too - in disabled state.
260        //serverHostname, serverPort, 50521, CommunicationType.REST, sslConfig, null, null, null
261        final DeprecatedServerConfig serverConfig = new DeprecatedServerConfig();
262        serverConfig.setPort(serverPort);
263        serverConfig.setHostName(serverHostname);
264        serverConfig.setCommunicationType(CommunicationType.REST);
265        serverConfig.setSslConfig(sslConfig);
266        
267        final List<Peer> peerList;
268        if(peers != null) {
269            peerList = peers.stream()
270                            .map(Peer::new)
271                            .collect(Collectors.toList());
272        } else {
273            peerList = null;
274        }
275
276
277        final List<String> forwardingKeys;
278        if(alwaysSendTo != null) {
279            List<String> keyList = new ArrayList<>();
280
281            for(String keyPath : alwaysSendTo) {
282                try {
283                    List<String> keysFromFile = Files.readAllLines(toPath(workDir, keyPath));
284                    keyList.addAll(keysFromFile);
285                } catch (IOException e) {
286                    System.err.println("Error reading alwayssendto file: " + e.getMessage());
287                }
288            }
289
290            forwardingKeys = keyList.stream()
291                .collect(Collectors.toList());
292        } else {
293            forwardingKeys = Collections.emptyList();
294        }
295
296        Config config = new Config();
297        config.setServer(serverConfig);
298        config.setJdbcConfig(jdbcConfig);
299        config.setPeers(peerList);
300        config.setAlwaysSendTo(forwardingKeys);
301        config.setUnixSocketFile(toPath(workDir, unixSocketFile));
302        config.setUseWhiteList(useWhiteList);
303        config.setKeys(keyData);
304        config.setDisablePeerDiscovery(false);
305        return config;
306    }
