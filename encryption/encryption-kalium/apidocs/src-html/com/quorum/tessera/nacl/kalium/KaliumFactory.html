001package com.quorum.tessera.nacl.kalium;
002
003import com.quorum.tessera.nacl.NaclFacade;
004import com.quorum.tessera.nacl.NaclFacadeFactory;
005import org.abstractj.kalium.NaCl;
006import org.slf4j.Logger;
007import org.slf4j.LoggerFactory;
008
009public class KaliumFactory implements NaclFacadeFactory {
010
011    private static final Logger LOGGER = LoggerFactory.getLogger(KaliumFactory.class);
012
013    @Override
014    public NaclFacade create() {
015        LOGGER.debug("Creating a Kalium implementation of NaclFacadeFactory");
016
017        final NaCl.Sodium sodium = NaCl.sodium();
018
019        return new Kalium(sodium);
020    }
021
022}
