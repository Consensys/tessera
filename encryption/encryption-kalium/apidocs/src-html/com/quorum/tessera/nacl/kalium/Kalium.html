019/**
020 * An implementation of the {@link NaclFacade} using the Kalium and libsodium binding
021 */
022public class Kalium implements NaclFacade {
023
024    private static final Logger LOGGER = LoggerFactory.getLogger(Kalium.class);
025
026    private static final String REDACTED = "REDACTED";
027
028    private final NaCl.Sodium sodium;
029
030    public Kalium(final NaCl.Sodium sodium) {
031        this.sodium = Objects.requireNonNull(sodium, "Kalium sodium implementation was null");
032
033        LOGGER.info("Initialising Sodium...");
034        this.sodium.sodium_init();
035        LOGGER.info("Sodium initialised");
036    }

038    @Override
039    public SharedKey computeSharedKey(final PublicKey publicKey, final PrivateKey privateKey) {
040        final byte[] output = new byte[CRYPTO_BOX_CURVE25519XSALSA20POLY1305_BEFORENMBYTES];
041
042        LOGGER.info("Computing the shared key for public key {} and private key {}", publicKey, REDACTED);
043        LOGGER.debug("Computing the shared key for public key {} and private key {}", publicKey, privateKey);
044        final int sodiumResult = this.sodium.crypto_box_curve25519xsalsa20poly1305_beforenm(
045            output, publicKey.getKeyBytes(), privateKey.getKeyBytes()
046        );
047
048        if (sodiumResult == -1) {
049            LOGGER.warn("Could not compute the shared key for pub {} and priv {}", publicKey, REDACTED);
050            LOGGER.debug("Could not compute the shared key for pub {} and priv {}", publicKey, privateKey);
051            throw new NaclException("Kalium could not compute the shared key");
052        }
053
054        final SharedKey sharedKey = SharedKey.from(output);
055
056        LOGGER.info("Computed shared key {} for pub {} and priv {}", sharedKey, publicKey, REDACTED);
057        LOGGER.debug("Computed shared key {} for pub {} and priv {}", sharedKey, publicKey, privateKey);
058
059        return sharedKey;
060    }
061
062    @Override
063    public byte[] seal(final byte[] message, final Nonce nonce, final PublicKey publicKey, final PrivateKey privateKey) {
064        /*
065         * The Kalium library uses the C API
066         * which expects the first CRYPTO_BOX_CURVE25519XSALSA20POLY1305_ZEROBYTES bytes to be zero
067         */
068        final byte[] paddedMessage = pad(message, CRYPTO_BOX_CURVE25519XSALSA20POLY1305_ZEROBYTES);
069        final byte[] output = new byte[paddedMessage.length];
070
071        LOGGER.info("Sealing message using public key {}", publicKey);
072        LOGGER.debug(
073            "Sealing message {} using nonce {}, public key {} and private key {}",
074            Arrays.toString(message), nonce, publicKey, privateKey
075        );
076
077        final int sodiumResult = sodium.crypto_box_curve25519xsalsa20poly1305(
078            output, paddedMessage, paddedMessage.length, nonce.getNonceBytes(), publicKey.getKeyBytes(), privateKey.getKeyBytes()
079        );
080
081        if (sodiumResult == -1) {
082            LOGGER.warn("Could not create sealed payload using public key {} and private key {}", publicKey, REDACTED);
083            LOGGER.debug("Could not create sealed payload using public key {} and private key {}", publicKey, privateKey);
084            throw new NaclException("Kalium could not seal the payload using the provided keys directly");
085        }
086
087        LOGGER.info("Created sealed payload for public key {}", publicKey);
088        LOGGER.debug(
089            "Created sealed payload {} using nonce {}, public key {} and private key {}",
090            Arrays.toString(output), nonce, publicKey, privateKey
091        );
092
093        /*
094         * NaCL C API states that first crypto_secretbox_BOXZEROBYTES must be zero
095         * but these are not part of the message, and must be stripped out
096         */
097        return extract(output, CRYPTO_BOX_CURVE25519XSALSA20POLY1305_BOXZEROBYTES);
098    }

100    @Override
101    public byte[] open(final byte[] cipherText, final Nonce nonce, final PublicKey publicKey, final PrivateKey privateKey) {
102        LOGGER.info("Opening message using public key {}", publicKey);
103        LOGGER.debug(
104            "Opening message {} using nonce {}, public key {} and private key {}",
105            Arrays.toString(cipherText), nonce, publicKey, privateKey
106        );
107
108        /*
109         * NaCL C API states that first crypto_secretbox_BOXZEROBYTES must be zero
110         * but these are not part of the ciphertext, and must be added in
111         */
112        final byte[] paddedInput = pad(cipherText, CRYPTO_BOX_CURVE25519XSALSA20POLY1305_BOXZEROBYTES);
113        final byte[] paddedOutput = new byte[paddedInput.length];
114
115        final int sodiumResult = sodium.crypto_box_curve25519xsalsa20poly1305_open(
116            paddedOutput, paddedInput, paddedInput.length, nonce.getNonceBytes(), publicKey.getKeyBytes(), privateKey.getKeyBytes()
117        );
118
119        if (sodiumResult == -1) {
120            LOGGER.warn("Could not open sealed payload using public key {} and private key {}", publicKey, REDACTED);
121            LOGGER.debug("Could not opern sealed payload using public key {} and private key {}", publicKey, privateKey);
122            throw new NaclException("Kalium could not open the payload using the provided keys directly");
123        }
124
125        LOGGER.info("Opened sealed payload for public key {}", publicKey);
126        LOGGER.debug(
127            "Opened payload {} using nonce {}, public key {} and private key {} to get result {}",
128            Arrays.toString(cipherText), nonce, publicKey, privateKey, Arrays.toString(paddedOutput)
129        );
130
131        return extract(paddedOutput, CRYPTO_BOX_CURVE25519XSALSA20POLY1305_ZEROBYTES);
132    }

134    @Override
135    public byte[] sealAfterPrecomputation(final byte[] message, final Nonce nonce, final SharedKey sharedKey) {
136        /*
137         * The Kalium library uses the C API
138         * which expects the first CRYPTO_BOX_CURVE25519XSALSA20POLY1305_ZEROBYTES bytes to be zero
139         */
140        final byte[] paddedMessage = pad(message, CRYPTO_BOX_CURVE25519XSALSA20POLY1305_ZEROBYTES);
141        final byte[] output = new byte[paddedMessage.length];
142
143        LOGGER.info("Sealing message using public key {}", sharedKey);
144        LOGGER.debug(
145            "Sealing message {} using nonce {} and shared key {}", Arrays.toString(message), nonce, sharedKey
146        );
147
148        final int sodiumResult = this.sodium.crypto_box_curve25519xsalsa20poly1305_afternm(
149            output, paddedMessage, paddedMessage.length, nonce.getNonceBytes(), sharedKey.getKeyBytes()
150        );
151
152        if (sodiumResult == -1) {
153            LOGGER.warn("Could not create sealed payload using shared key {}", sharedKey);
154            LOGGER.debug("Could not create sealed payload using shared key {}", sharedKey);
155            throw new NaclException("Kalium could not seal the payload using the shared key");
156        }
157
158        LOGGER.info("Created sealed payload for shared key {}", sharedKey);
159        LOGGER.debug(
160            "Created sealed payload {} using nonce {} and shared key {}", Arrays.toString(output), nonce, sharedKey
161        );
162
163        /*
164         * NaCL C API states that first crypto_secretbox_BOXZEROBYTES must be zero
165         * but these are not part of the message, and must be stripped out
166         */
167        return extract(output, CRYPTO_BOX_CURVE25519XSALSA20POLY1305_BOXZEROBYTES);
168    }

170    @Override
171    public byte[] openAfterPrecomputation(final byte[] encryptedPayload, final Nonce nonce, final SharedKey sharedKey) {
172        LOGGER.info("Opening message using shared key {}", sharedKey);
173        LOGGER.debug(
174            "Opening message {} using nonce {} and shared key {}", Arrays.toString(encryptedPayload), nonce, sharedKey
175        );
176
177        /*
178         * NaCL C API states that first crypto_secretbox_BOXZEROBYTES must be zero
179         * but these are not part of the ciphertext, and must be added in
180         */
181        final byte[] paddedInput = pad(encryptedPayload, CRYPTO_BOX_CURVE25519XSALSA20POLY1305_BOXZEROBYTES);
182        final byte[] paddedOutput = new byte[paddedInput.length];
183
184        final int sodiumResult = this.sodium.crypto_box_curve25519xsalsa20poly1305_open_afternm(
185            paddedOutput, paddedInput, paddedInput.length, nonce.getNonceBytes(), sharedKey.getKeyBytes()
186        );
187
188        if (sodiumResult == -1) {
189            LOGGER.warn("Could not open sealed payload using shared key {}", sharedKey);
190            LOGGER.debug("Could not open sealed payload using shared key {}", sharedKey);
191            throw new NaclException("Kalium could not open the payload using the shared key");
192        }
193
194        LOGGER.info("Opened sealed payload for shared key {}", sharedKey);
195        LOGGER.debug(
196            "Opened payload {} using nonce {}, public key {} and private key {} to get result {}",
197            Arrays.toString(encryptedPayload), nonce, sharedKey, REDACTED, Arrays.toString(paddedOutput)
198        );
199
200        return extract(paddedOutput, CRYPTO_BOX_CURVE25519XSALSA20POLY1305_ZEROBYTES);
201    }

203    @Override
204    public Nonce randomNonce() {
205
206        final byte[] nonceBytes = new byte[CRYPTO_BOX_CURVE25519XSALSA20POLY1305_NONCEBYTES];
207
208        this.sodium.randombytes(nonceBytes, nonceBytes.length);
209
210        final Nonce nonce = new Nonce(nonceBytes);
211
212        LOGGER.debug("Generated random nonce {}", nonce);
213
214        return nonce;
215    }

217    @Override
218    public KeyPair generateNewKeys() {
219        final byte[] publicKey = new byte[CRYPTO_BOX_CURVE25519XSALSA20POLY1305_PUBLICKEYBYTES];
220        final byte[] privateKey = new byte[CRYPTO_BOX_CURVE25519XSALSA20POLY1305_SECRETKEYBYTES];
221
222        LOGGER.info("Generating new keypair...");
223
224        final int sodiumResult = this.sodium.crypto_box_curve25519xsalsa20poly1305_keypair(publicKey, privateKey);
225
226        if (sodiumResult == -1) {
227            LOGGER.warn("Unable to generate a new keypair!");
228            throw new NaclException("Kalium could not generate a new public/private keypair");
229        }
230
231        final PublicKey pubKey = PublicKey.from(publicKey);
232        final PrivateKey privKey = PrivateKey.from(privateKey);
233
234        LOGGER.info("Generated public key {} and private key {}", pubKey, REDACTED);
235        LOGGER.debug("Generated public key {} and private key {}", pubKey, privKey);
236
237        return new KeyPair(pubKey, privKey);
238    }

240    /**
241     * Left-pads a given message with padSize amount of zeros
242     *
243     * @param input   the message to be padded
244     * @param padSize the amount of left-padding to apply
245     * @return the padded message
246     */
247    private byte[] pad(final byte[] input, final int padSize) {
248        final byte[] paddedMessage = new byte[padSize + input.length];
249        System.arraycopy(input, 0, paddedMessage, padSize, input.length);
250
251        return paddedMessage;
252    }
253
254    /**
255     * Removes left-padding from a given message to tune of padSize
256     *
257     * @param input   The message from which to remove left-padding
258     * @param padSize The amount of left-padding to remove
259     * @return The trimmed message
260     */
261    private byte[] extract(final byte[] input, final int padSize) {
262        final byte[] extractedMessage = new byte[input.length - padSize];
263        System.arraycopy(input, padSize, extractedMessage, 0, extractedMessage.length);
264
265        return extractedMessage;
266    }

268    @Override
269    public SharedKey createSingleKey() {
270        LOGGER.info("Generating random key");
271
272        final byte[] keyBytes = new byte[CRYPTO_BOX_CURVE25519XSALSA20POLY1305_PUBLICKEYBYTES];
273
274        this.sodium.randombytes(keyBytes, keyBytes.length);
275
276        final SharedKey key = SharedKey.from(keyBytes);
277
278        LOGGER.info("Random key generated");
279        LOGGER.debug("Generated key with value {}", key);
280
281        return key;
282    }
