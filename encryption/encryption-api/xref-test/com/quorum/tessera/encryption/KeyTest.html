3   import java.lang.reflect.Method;
4   import java.util.Arrays;
5   import java.util.Base64;
6   import java.util.List;
7   import static org.assertj.core.api.Assertions.*;
8   import org.junit.Test;
9   import org.junit.runner.RunWith;
10  import org.junit.runners.Parameterized;
11  
12  @RunWith(Parameterized.class)
13  public class KeyTest {
14  
15      private final Class<? extends Key> type;
16  
17      private final Method factoryMethod;
18  
19      public KeyTest(Class<? extends Key> type) throws Exception {
20          this.type = type;
21          factoryMethod = type.getDeclaredMethod("from", byte[].class);
22      }
23  
24      @Test
25      public void doTests() throws Exception {
26  
27          byte[] data = "SOME_DATA".getBytes();
28  
29          String base64Value = Base64.getEncoder().encodeToString(data);
30  
31          Key key = (Key) factoryMethod.invoke(null, data);
32  
33          assertThat(key).isNotNull();
34  
35          assertThat(key.encodeToBase64()).isEqualTo(base64Value);
36  
37          assertThat(key.toString()).isEqualTo(type.getSimpleName() + "[" + base64Value + "]");
38  
39          assertThat(key.hashCode()).isEqualTo(Arrays.hashCode(data));
40  
41          Key secondKey = (Key) factoryMethod.invoke(null, "OTHERDATA".getBytes());
42  
43          assertThat(key).isNotEqualTo(secondKey);
44  
45          BogusKey bogusKey = new BogusKey(data);
46          assertThat(key).isNotEqualTo(bogusKey);
47          assertThat(key.encodeToBase64()).isEqualTo(bogusKey.encodeToBase64());
48  
49          Key otherKeyWithSameData = (Key) factoryMethod.invoke(null, data);
50          assertThat(key).isEqualTo(otherKeyWithSameData);
51      }
52  
53      @Parameterized.Parameters
54      public static List<Class<? extends Key>> cases() {
55          return Arrays.asList(MasterKey.class, SharedKey.class, PrivateKey.class, PublicKey.class);
56      }
57  
58      static class BogusKey extends BaseKey {
59  
60          BogusKey(byte[] keyBytes) {
61              super(keyBytes);
62          }
63  
64      }
65  
66  }
