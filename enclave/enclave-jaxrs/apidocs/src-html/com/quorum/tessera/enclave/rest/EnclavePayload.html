001package com.quorum.tessera.enclave.rest;
002
003import java.io.Serializable;
004import java.util.List;
005import javax.xml.bind.annotation.XmlMimeType;
006import javax.xml.bind.annotation.XmlRootElement;
007
008@XmlRootElement
009public class EnclavePayload implements Serializable {
010
011    @XmlMimeType("base64Binary")
012    private byte[] data;
013
014    @XmlMimeType("base64Binary")
015    private byte[] senderKey;
016
017    @XmlMimeType("base64Binary")
018    private List<byte[]> recipientPublicKeys;
019
020    public byte[] getData() {
021        return data;
022    }
023
024    public void setData(byte[] data) {
025        this.data = data;
026    }
027
028    public byte[] getSenderKey() {
029        return senderKey;
030    }
031
032    public void setSenderKey(byte[] senderKey) {
033        this.senderKey = senderKey;
034    }
035
036    public List<byte[]> getRecipientPublicKeys() {
037        return recipientPublicKeys;
038    }
039
040    public void setRecipientPublicKeys(List<byte[]> recipientPublicKeys) {
041        this.recipientPublicKeys = recipientPublicKeys;
042    }
043
044}
