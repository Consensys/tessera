001package com.quorum.tessera.enclave.rest;
002
003import javax.ws.rs.core.Application;
004import java.util.Objects;
005import java.util.Set;
006import java.util.stream.Collectors;
007import java.util.stream.Stream;
008
009public class EnclaveApplication extends Application implements com.quorum.tessera.config.apps.EnclaveApp {
010
011    private final EnclaveResource resource;
012
013    public EnclaveApplication(final EnclaveResource resource) {
014        this.resource = Objects.requireNonNull(resource);
015    }
016
017    @Override
018    public Set<Object> getSingletons() {
019        return Stream.of(resource, new DefaultExceptionMapper()).collect(Collectors.toSet());
020    }
021
022}
