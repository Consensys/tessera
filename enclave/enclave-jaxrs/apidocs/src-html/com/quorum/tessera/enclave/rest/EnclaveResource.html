001package com.quorum.tessera.enclave.rest;
002
003import com.quorum.tessera.enclave.Enclave;
004import com.quorum.tessera.enclave.EncodedPayload;
005import com.quorum.tessera.enclave.PayloadEncoder;
006import com.quorum.tessera.enclave.RawTransaction;
007import com.quorum.tessera.encryption.PublicKey;
008import com.quorum.tessera.nacl.Nonce;
009import com.quorum.tessera.service.Service;
010
011import javax.ws.rs.*;
012import javax.ws.rs.core.MediaType;
013import javax.ws.rs.core.Response;
014import javax.ws.rs.core.StreamingOutput;
015import java.util.List;
016import java.util.Objects;
017import java.util.Optional;
018import java.util.stream.Collectors;
019import javax.json.Json;
020import javax.ws.rs.core.Response.Status;
021
022@Path("/")
023public class EnclaveResource {
024
025    private final Enclave enclave;
026
027    private final PayloadEncoder payloadEncoder = PayloadEncoder.create();
028
029    public EnclaveResource(Enclave enclave) {
030        this.enclave = Objects.requireNonNull(enclave);
031    }
032
033    @GET
034    @Path("ping")
035    public Response ping() {
036        Service.Status status = enclave.status();
037        Status httpStatus;
038        if (status == Service.Status.STARTED) {
039            httpStatus = Status.OK;
040        } else {
041            httpStatus = Status.SERVICE_UNAVAILABLE;
042        }
043        return Response.status(httpStatus).entity(status.name()).build();
044    }
045
046    @Produces(MediaType.APPLICATION_OCTET_STREAM)
047    @GET
048    @Path("default")
049    public Response defaultPublicKey() {
050        final StreamingOutput streamingOutput = out -> out.write(enclave.defaultPublicKey().getKeyBytes());
051        return Response.ok(streamingOutput)
052                .build();
053    }
054
055    @GET
056    @Produces("application/json")
057    @Path("forwarding")
058    public Response getForwardingKeys() {
059
060        List<String> body = enclave.getForwardingKeys()
061                .stream()
062                .map(PublicKey::encodeToBase64)
063                .collect(Collectors.toList());
064
065        return Response.ok(Json.createArrayBuilder(body).build().toString(), MediaType.APPLICATION_JSON_TYPE)
066                .build();
067
068    }
069
070    @GET
071    @Produces("application/json")
072    @Path("public")
073    public Response getPublicKeys() {
074
075        List<String> body = enclave.getPublicKeys()
076                .stream()
077                .map(PublicKey::encodeToBase64)
078                .collect(Collectors.toList());
079
080        return Response.ok(Json.createArrayBuilder(body).build().toString(), MediaType.APPLICATION_JSON_TYPE)
081                .build();
082    }
083
084    @POST
085    @Path("encrypt")
086    @Consumes(MediaType.APPLICATION_JSON)
087    @Produces(MediaType.APPLICATION_OCTET_STREAM)
088    public Response encryptPayload(EnclavePayload payload) {
089
090        PublicKey senderKey = PublicKey.from(payload.getSenderKey());
091
092        List<PublicKey> recipientPublicKeys = payload.getRecipientPublicKeys()
093                .stream()
094                .map(PublicKey::from)
095                .collect(Collectors.toList());
096
097        EncodedPayload outcome = enclave.encryptPayload(payload.getData(), senderKey, recipientPublicKeys);
098
099        byte[] response = payloadEncoder.encode(outcome);
100        final StreamingOutput streamingOutput = out -> out.write(response);
101        return Response.ok(streamingOutput)
102                .build();
103    }
104
105    @POST
106    @Path("encrypt/raw")
107    @Consumes(MediaType.APPLICATION_JSON)
108    @Produces(MediaType.APPLICATION_OCTET_STREAM)
109    public Response encryptPayload(EnclaveRawPayload enclaveRawPayload) {
110
111        byte[] encryptedPayload = enclaveRawPayload.getEncryptedPayload();
112        byte[] encryptedKey = enclaveRawPayload.getEncryptedKey();
113        Nonce nonce = new Nonce(enclaveRawPayload.getNonce());
114        PublicKey from = PublicKey.from(enclaveRawPayload.getFrom());
115
116        List<PublicKey> recipientPublicKeys = enclaveRawPayload.getRecipientPublicKeys().stream()
117                .map(PublicKey::from).collect(Collectors.toList());
118
119        RawTransaction rawTransaction = new RawTransaction(encryptedPayload, encryptedKey, nonce, from);
120
121        EncodedPayload outcome = enclave.encryptPayload(rawTransaction, recipientPublicKeys);
122
123        byte[] response = payloadEncoder.encode(outcome);
124        final StreamingOutput streamingOutput = out -> out.write(response);
125        return Response.ok(streamingOutput)
126                .build();
127    }
128
129    @POST
130    @Path("encrypt/toraw")
131    @Consumes(MediaType.APPLICATION_JSON)
132    @Produces(MediaType.APPLICATION_JSON)
133    public Response encryptRawPayload(EnclavePayload payload) {
134
135        RawTransaction rawTransaction = enclave.encryptRawPayload(payload.getData(), PublicKey.from(payload.getSenderKey()));
136
137        EnclaveRawPayload enclaveRawPayload = new EnclaveRawPayload();
138        enclaveRawPayload.setFrom(rawTransaction.getFrom().getKeyBytes());
139        enclaveRawPayload.setNonce(rawTransaction.getNonce().getNonceBytes());
140        enclaveRawPayload.setEncryptedPayload(rawTransaction.getEncryptedPayload());
141        enclaveRawPayload.setEncryptedKey(rawTransaction.getEncryptedKey());
142
143        return Response.ok(enclaveRawPayload).build();
144
145    }
146
147    @POST
148    @Path("unencrypt")
149    @Consumes(MediaType.APPLICATION_JSON)
150    @Produces(MediaType.APPLICATION_OCTET_STREAM)
151    public Response unencryptTransaction(EnclaveUnencryptPayload enclaveUnencryptPayload) {
152
153        EncodedPayload payload = payloadEncoder.decode(enclaveUnencryptPayload.getData());
154        PublicKey providedKey = Optional.ofNullable(enclaveUnencryptPayload.getProvidedKey())
155            .map(PublicKey::from)
156            .orElse(null);
157
158        byte[] response = enclave.unencryptTransaction(payload, providedKey);
159
160        final StreamingOutput streamingOutput = out -> out.write(response);
161        return Response.ok(streamingOutput).build();
162
163    }
164
165    @POST
166    @Path("addRecipient")
167    @Consumes(MediaType.APPLICATION_JSON)
168    @Produces(MediaType.APPLICATION_OCTET_STREAM)
169    public Response createNewRecipientBox(EnclaveUnencryptPayload enclaveUnencryptPayload) {
170
171        EncodedPayload payload = payloadEncoder.decode(enclaveUnencryptPayload.getData());
172        PublicKey providedKey = PublicKey.from(enclaveUnencryptPayload.getProvidedKey());
173
174        byte[] response = enclave.createNewRecipientBox(payload, providedKey);
175
176        final StreamingOutput streamingOutput = out -> out.write(response);
177        return Response.ok(streamingOutput).build();
178
179    }
180
181}
