001package com.quorum.tessera.enclave;
002
003import com.quorum.tessera.encryption.PublicKey;
004
005/**
006 * Encodes and decodes a {@link EncodedPayload} to and from its binary
007 * representation
008 */
009public interface PayloadEncoder {
010
011    /**
012     * Encodes the payload to a byte array
013     *
014     * @param payload the payload to encode
015     * @return the byte array representing the encoded payload
016     */
017    byte[] encode(EncodedPayload payload);
018
019    /**
020     * Decodes a byte array back into an encrypted payload
021     *
022     * @param input The byte array to decode into an EncodedPayload
023     * @return the decoded payload
024     */
025    EncodedPayload decode(byte[] input);
026
027
028    /**
029     * Strips a payload of any data that isn't relevant to the given recipient
030     * Used to format a payload before it is sent to the target node
031     *
032     * @param input     the full payload from which data needs to be stripped
033     * @param recipient the recipient to retain information about
034     * @return a payload which contains a subset of data from the input, which is relevant to the recipient
035     */
036    EncodedPayload forRecipient(EncodedPayload input, PublicKey recipient);
037
038    static PayloadEncoder create() {
039        return new PayloadEncoderImpl();
040    }
041
042}
