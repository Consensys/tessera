10      private UnsupportedKeyPair keyPair;
11  
12      @Before
13      public void setUp() {
14          this.keyPair = new UnsupportedKeyPair(null, null, null, null, null, null, null, null, null, null, null, null, null, null);
15      }
16  
17      @Test
18      public void getPasswordAlwaysReturnsNull() {
19          assertThat(keyPair.getPassword()).isNull();
20  
21          keyPair.withPassword("password");
22  
23          assertThat(keyPair.getPassword()).isNull();
24      }
25  
26      @Test
27      public void versionSetters() {
28          assertThat(keyPair.getHashicorpVaultSecretVersion()).isNull();
29          assertThat(keyPair.getAzureVaultPublicKeyVersion()).isNull();
30          assertThat(keyPair.getAzureVaultPrivateKeyVersion()).isNull();
31  
32          keyPair.setHashicorpVaultSecretVersion("1");
33          keyPair.setAzureVaultPublicKeyVersion("pubVer");
34          keyPair.setAzureVaultPrivateKeyVersion("privVer");
35  
36          assertThat(keyPair.getHashicorpVaultSecretVersion()).isEqualTo("1");
37          assertThat(keyPair.getAzureVaultPublicKeyVersion()).isEqualTo("pubVer");
38          assertThat(keyPair.getAzureVaultPrivateKeyVersion()).isEqualTo("privVer");
39      }
