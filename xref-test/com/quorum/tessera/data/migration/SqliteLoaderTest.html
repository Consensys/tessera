22  public class SqliteLoaderTest {
23  
24      private Path dbfilePath;
25  
26      private SqliteLoader loader;
27  
28      private Map<String, String> fixtures;
29  
30      //CREATE TABLE ENCRYPTED_TRANSACTION (ID BIGINT NOT NULL, ENCODED_PAYLOAD LONGVARBINARY NOT NULL, HASH LONGVARBINARY NOT NULL UNIQUE, PRIMARY KEY (ID))
31      @Before
32      public void doGenerateSample() throws Exception {
33          fixtures = new LinkedHashMap<>();
34          for (int i = 0; i < 10; i++) {
35              fixtures.put(UUID.randomUUID().toString(), UUID.randomUUID().toString());
36          }
37  
38          dbfilePath = Files.createTempFile("sample", ".db");
39          try (Connection conn = DriverManager.getConnection("jdbc:sqlite:" + dbfilePath);
40               Statement statement = conn.createStatement();) {
41              statement.execute("CREATE TABLE payload (key LONGVARBINARY,bytes LONGVARBINARY)");
42              try (PreparedStatement insertStatement = conn.prepareStatement("INSERT INTO payload (key,bytes) values (?,?)")) {
43                  for (Entry<String, String> entry : fixtures.entrySet()) {
44                      insertStatement.setString(1, entry.getKey());
45                      insertStatement.setString(2, entry.getValue());
46                      insertStatement.executeUpdate();
47                  }
48  
49              }
50          }
51  
52          loader = new SqliteLoader();
53      }
54  
55      @After
56      public void onTearDown() throws Exception {
57          Files.deleteIfExists(dbfilePath);
58      }
59  
60      @Test
61      public void load() throws IOException {
62  
63          Map<byte[], byte[]> results = loader.load(dbfilePath);
64  
65          assertThat(results).hasSize(fixtures.size());
66  
67          Map<String, String> resultz = results.entrySet().stream()
68              .collect(Collectors.toMap(entry -> new String(entry.getKey()), entry -> new String(entry.getValue())));
69  
70          assertThat(resultz).containsAllEntriesOf(fixtures);
71  
72      }
73  
74  }
