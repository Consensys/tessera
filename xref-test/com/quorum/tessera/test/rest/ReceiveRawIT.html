39      public void init() {

41          this.partyOne = partyHelper.findByAlias("A");
42          this.partyTwo = partyHelper.findByAlias("B");
43          
44          SendRequest sendRequest = new SendRequest();
45          sendRequest.setPayload(PAYLOAD);
46          sendRequest.setTo(partyTwo.getPublicKey());
47          sendRequest.setFrom(partyOne.getPublicKey());
48  
49          final Response response = client.target(partyOne.getQ2TUri())
50              .path("/send")
51              .request()
52              .post(Entity.entity(sendRequest, MediaType.APPLICATION_JSON));
53  
54          final SendResponse result = response.readEntity(SendResponse.class);
55  
56          this.hash = result.getKey();

60      @Test
61      public void fetchExistingTransactionUsingOwnKey() {
62  
63          final Response response = client.target(partyOne.getQ2TUri())
64              .path(RECEIVE_PATH)
65              .request()
66              .header(C11N_KEY, this.hash)
67              .header(C11N_TO, partyOne.getPublicKey())
68              .buildGet()
69              .invoke();
70  
71          //validate result
72  
73          assertThat(response).isNotNull();
74          assertThat(response.getStatus()).isEqualTo(200);
75  
76          final byte[] result = response.readEntity(byte[].class);
77  
78          assertThat(result).isEqualTo(PAYLOAD);
79  
80      }

115
