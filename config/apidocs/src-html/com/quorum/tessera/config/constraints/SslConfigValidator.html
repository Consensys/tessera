001package com.quorum.tessera.config.constraints;
002
003import com.quorum.tessera.config.SslAuthenticationMode;
004import com.quorum.tessera.config.SslConfig;
005import com.quorum.tessera.config.SslConfigType;
006import com.quorum.tessera.config.SslTrustMode;
007import com.quorum.tessera.config.util.EnvironmentVariableProvider;
008import com.quorum.tessera.config.util.EnvironmentVariableProviderFactory;
009import com.quorum.tessera.config.util.EnvironmentVariables;
010
011import javax.validation.ConstraintValidator;
012import javax.validation.ConstraintValidatorContext;
013import java.nio.file.Files;
014import java.util.Objects;
015
016public class SslConfigValidator implements ConstraintValidator<ValidSsl,SslConfig> {
017
018    private EnvironmentVariableProvider envVarProvider = EnvironmentVariableProviderFactory.load().create();
019
020    @Override
021    public boolean isValid(SslConfig sslConfig, ConstraintValidatorContext context) {
022
023        context.disableDefaultConstraintViolation();
024
025        if (Objects.isNull(sslConfig)) {
026            return true;
027        }
028
029        if (sslConfig.getTls() == SslAuthenticationMode.STRICT) {
030
031            if (!sslConfig.isGenerateKeyStoreIfNotExisted()) {
032
033                if (!isServerKeyStoreConfigValid(sslConfig, context) ||
034                    !isClientKeyStoreConfigValid(sslConfig, context)) {
035                    return false;
036                }
037            }
038
039            if (!isTrustModeConfigValid(sslConfig, context)) {
040                return false;
041            }
042
043            if (!isServerConfigValidForWhiteListMode(sslConfig, context)) {
044                return false;
045            }
046
047            if (!isServerConfigValidForCAMode(sslConfig, context)) {
048                return false;
049            }
050
051            if (!isClientConfigValidForWhiteListMode(sslConfig, context)) {
052                return false;
053            }
054
055            if (!isClientConfigValidForCAMode(sslConfig, context)) {
056                return false;
057            }
058        }
059
060        return true;
061    }
062
063    private boolean isServerKeyStoreConfigValid(SslConfig sslConfig, ConstraintValidatorContext context) {
064        if (SslConfigType.CLIENT_ONLY == sslConfig.getSslConfigType()) {
065            return true;
066        }
067        if (Objects.isNull(sslConfig.getServerKeyStore()) ||
068            !isPasswordProvided(sslConfig.getServerKeyStorePassword(), sslConfig.getEnvironmentVariablePrefix(), EnvironmentVariables.SERVER_KEYSTORE_PWD) ||
069            Files.notExists(sslConfig.getServerKeyStore())) {
070            if (Objects.isNull(sslConfig.getServerTlsKeyPath()) ||
071                Objects.isNull(sslConfig.getServerTlsCertificatePath()) ||
072                Files.notExists(sslConfig.getServerTlsKeyPath()) ||
073                Files.notExists(sslConfig.getServerTlsCertificatePath())) {
074                setMessage("Server keystore configuration not valid. " +
075                    "Please ensure keystore file exists or keystore password not null, " +
076                    "otherwise please set keystore generation flag to true to have keystore created", context);
077                return false;
078            }
079        }
080        return true;
081    }
082
083    private boolean isClientKeyStoreConfigValid(SslConfig sslConfig, ConstraintValidatorContext context) {
084        if (SslConfigType.SERVER_ONLY == sslConfig.getSslConfigType()) {
085            return true;
086        }
087        if (Objects.isNull(sslConfig.getClientKeyStore()) ||
088            !isPasswordProvided(sslConfig.getClientKeyStorePassword(), sslConfig.getEnvironmentVariablePrefix(), EnvironmentVariables.CLIENT_KEYSTORE_PWD) ||
089            Files.notExists(sslConfig.getClientKeyStore())) {
090            if (Objects.isNull(sslConfig.getClientTlsKeyPath()) ||
091                Objects.isNull(sslConfig.getClientTlsCertificatePath()) ||
092                Files.notExists(sslConfig.getClientTlsKeyPath()) ||
093                Files.notExists(sslConfig.getClientTlsCertificatePath())) {
094                setMessage("Client keystore configuration not valid. " +
095                    "Please ensure keystore file exists or keystore password not null, " +
096                    "otherwise please set keystore generation flag to true to have keystore created", context);
097                return false;
098            }
099        }
100        return true;
101    }
102
103    private boolean isTrustModeConfigValid(SslConfig sslConfig, ConstraintValidatorContext context) {
104        if ((Objects.isNull(sslConfig.getServerTrustMode()) && sslConfig.getSslConfigType() != SslConfigType.CLIENT_ONLY)
105            ||
106            (Objects.isNull(sslConfig.getClientTrustMode()) && sslConfig.getSslConfigType() != SslConfigType.SERVER_ONLY)) {
107            setMessage("Trust mode does not have valid value. Please check server/client trust mode config", context);
108            return false;
109        }
110        return true;
111    }
112
113    private boolean isServerConfigValidForWhiteListMode(SslConfig sslConfig, ConstraintValidatorContext context) {
114        if (sslConfig.getServerTrustMode() == SslTrustMode.WHITELIST) {
115            if (Objects.isNull(sslConfig.getKnownClientsFile()) ||
116                Files.notExists(sslConfig.getKnownClientsFile())) {
117                setMessage("Known clients file not found. If server trust mode is WHITELIST, known clients file must be provided", context);
118                return false;
119            }
120        }
121        return true;
122    }
123
124    private boolean isServerConfigValidForCAMode(SslConfig sslConfig, ConstraintValidatorContext context) {
125        if (sslConfig.getServerTrustMode() == SslTrustMode.CA) {
126            if (Objects.isNull(sslConfig.getServerTrustStore()) ||
127                !isPasswordProvided(sslConfig.getServerTrustStorePassword(), sslConfig.getEnvironmentVariablePrefix(), EnvironmentVariables.SERVER_TRUSTSTORE_PWD) ||
128                Files.notExists(sslConfig.getServerTrustStore())) {
129                if (Objects.isNull(sslConfig.getServerTrustCertificates())) {
130                    setMessage("Trust store config not valid. If server trust mode is CA, trust store must exist and not be null", context);
131                    return false;
132                }
133            }
134        }
135        return true;
136    }
137
138    private boolean isClientConfigValidForWhiteListMode(SslConfig sslConfig, ConstraintValidatorContext context) {
139        if (sslConfig.getClientTrustMode() == SslTrustMode.WHITELIST) {
140            if (Objects.isNull(sslConfig.getKnownServersFile()) ||
141                Files.notExists(sslConfig.getKnownServersFile())) {
142                setMessage("Known servers file not found. If client trust mode is WHITELIST, known servers file must be provided", context);
143                return false;
144            }
145        }
146        return true;
147    }
148
149    private boolean isClientConfigValidForCAMode(SslConfig sslConfig, ConstraintValidatorContext context) {
150        if (sslConfig.getClientTrustMode() == SslTrustMode.CA) {
151            if (Objects.isNull(sslConfig.getClientTrustStore()) ||
152                !isPasswordProvided(sslConfig.getClientTrustStorePassword(), sslConfig.getEnvironmentVariablePrefix(), EnvironmentVariables.CLIENT_TRUSTSTORE_PWD) ||
153                Files.notExists(sslConfig.getClientTrustStore())) {
154                if (Objects.isNull(sslConfig.getClientTrustCertificates())) {
155                    setMessage("Trust store config not valid. If client trust mode is CA, trust store must exist and not be null", context);
156                    return false;
157                }
158            }
159        }
160        return true;
161    }
162
163    private boolean isPasswordProvided(String configPassword, String envVarPrefix, String envVar) {
164        return configPassword != null || envVarProvider.hasEnv(envVar) || envVarProvider.hasEnv(envVarPrefix + "_" + envVar);
165    }
166
167    private void setMessage(final String message, ConstraintValidatorContext context) {
168        context
169            .buildConstraintViolationWithTemplate(message)
170            .addConstraintViolation();
171    }
172}
