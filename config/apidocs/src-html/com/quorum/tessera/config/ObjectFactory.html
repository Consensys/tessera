001package com.quorum.tessera.config;
002
003import javax.xml.bind.JAXBElement;
004import javax.xml.bind.annotation.XmlElementDecl;
005import javax.xml.bind.annotation.XmlRegistry;
006import javax.xml.namespace.QName;
007
008@XmlRegistry
009public class ObjectFactory {
010
011    private static final QName QNAME = new QName("http://tessera.github.com/config", "configuration");
012
013    @XmlElementDecl(namespace = "http://tessera.github.com/config", name = "configuration")
014    public JAXBElement<Config> createConfiguration(Config value) {
015        return new JAXBElement<>(QNAME, Config.class, null, value);
016    }
017
018}
