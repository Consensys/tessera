001package com.quorum.tessera.config.keypairs;
002
003import com.quorum.tessera.config.KeyDataConfig;
004import com.quorum.tessera.config.PrivateKeyData;
005import com.quorum.tessera.config.constraints.ValidBase64;
006import com.quorum.tessera.config.keys.KeyEncryptorFactory;
007import com.quorum.tessera.nacl.NaclException;
008
009import javax.validation.constraints.NotNull;
010import javax.validation.constraints.Pattern;
011import javax.xml.bind.annotation.XmlElement;
012
013import static com.quorum.tessera.config.PrivateKeyType.UNLOCKED;
014import javax.validation.constraints.Size;
015
016public class InlineKeypair implements ConfigKeyPair {
017
018    @XmlElement
019    private final String publicKey;
020
021    @NotNull
022    @XmlElement(name = "config")
023    private final KeyDataConfig privateKeyConfig;
024
025    private String password;
026
027    private String cachedValue;
028
029    public InlineKeypair(final String publicKey, final KeyDataConfig privateKeyConfig) {
030        this.publicKey = publicKey;
031        this.privateKeyConfig = privateKeyConfig;
032    }
033
034    public KeyDataConfig getPrivateKeyConfig() {
035        return this.privateKeyConfig;
036    }
037
038    @Override
039    @Size(min = 1)
040    @NotNull
041    @ValidBase64(message = "Invalid Base64 key provided")
042    public String getPublicKey() {
043        return this.publicKey;
044    }
045
046    @Override
047    @NotNull
048    @Size(min = 1)
049    @ValidBase64(message = "Invalid Base64 key provided")
050    @Pattern(regexp = "^((?!NACL_FAILURE).)*$", message = "Could not decrypt the private key with the provided password, please double check the passwords provided")
051    @Pattern(regexp = "^((?!MISSING_PASSWORD).)*$", message = "{InlineKeyData.missingPassword.message}")
052    public String getPrivateKey() {
053        final PrivateKeyData pkd = privateKeyConfig.getPrivateKeyData();
054
055        if (privateKeyConfig.getType() == UNLOCKED) {
056            return privateKeyConfig.getValue();
057        }
058
059        if (this.cachedValue == null) {
060            if (password == null) {
061                this.cachedValue = "MISSING_PASSWORD";
062            } else {
063                try {
064                    this.cachedValue = KeyEncryptorFactory.create().decryptPrivateKey(pkd, password).encodeToBase64();
065                } catch (final NaclException ex) {
066                    this.cachedValue = "NACL_FAILURE";
067                }
068            }
069        }
070
071        return this.cachedValue;
072    }
073
074    @Override
075    public void withPassword(final String password) {
076        this.password = password;
077    }
078
079    @Override
080    public String getPassword() {
081        return this.password;
082    }
083
084}
