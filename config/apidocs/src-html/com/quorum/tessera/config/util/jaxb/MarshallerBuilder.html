001package com.quorum.tessera.config.util.jaxb;
002
003import com.quorum.tessera.jaxb.JaxbCallback;
004import com.quorum.tessera.config.util.JaxbUtil;
005
006import javax.xml.bind.JAXBContext;
007import javax.xml.bind.Marshaller;
008
009public class MarshallerBuilder {
010
011    private MarshallerBuilder() {
012    }
013
014    public static MarshallerBuilder create() {
015        return new MarshallerBuilder();
016    }
017
018    private boolean beanvalidation = true;
019
020    private MediaType mediaType = MediaType.JSON;
021
022    public MarshallerBuilder withoutBeanValidation() {
023        this.beanvalidation = false;
024        return this;
025    }
026
027    public MarshallerBuilder withXmlMediaType() {
028        this.mediaType = MediaType.XML;
029        return this;
030    }
031
032    public Marshaller build() {
033
034        return JaxbCallback.execute(() -> {
035
036            JAXBContext jAXBContext = JAXBContext.newInstance(JaxbUtil.JAXB_CLASSES);
037
038            Marshaller marshaller = jAXBContext.createMarshaller();
039            if (!beanvalidation) {
040                Enum enu = Enum.valueOf(Class.class.cast(marshaller
041                    .getProperty("eclipselink.beanvalidation.mode")
042                    .getClass()), "NONE");
043
044                marshaller.setProperty("eclipselink.beanvalidation.mode", enu);
045            }
046            marshaller.setProperty("eclipselink.media-type", mediaType.getValue());
047            marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, true);
048
049            if (mediaType == MediaType.JSON) {
050                marshaller.setProperty("eclipselink.json.include-root", false);
051            }
052            return marshaller;
053        });
054
055    }
056
057}
