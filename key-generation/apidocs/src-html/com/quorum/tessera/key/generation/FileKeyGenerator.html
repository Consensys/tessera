001package com.quorum.tessera.key.generation;
002
003import com.quorum.tessera.config.ArgonOptions;
004import com.quorum.tessera.config.KeyData;
005import com.quorum.tessera.config.KeyDataConfig;
006import com.quorum.tessera.config.PrivateKeyData;
007import com.quorum.tessera.config.keypairs.FilesystemKeyPair;
008import com.quorum.tessera.config.keys.KeyEncryptor;
009import com.quorum.tessera.config.util.JaxbUtil;
010import com.quorum.tessera.config.util.PasswordReader;
011import com.quorum.tessera.encryption.KeyPair;
012import com.quorum.tessera.io.IOCallback;
013import com.quorum.tessera.nacl.NaclFacade;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017import java.nio.file.Files;
018import java.nio.file.Path;
019import java.nio.file.Paths;
020import java.util.Base64;
021import java.util.Objects;
022
023import static com.quorum.tessera.config.PrivateKeyType.LOCKED;
024import static com.quorum.tessera.config.PrivateKeyType.UNLOCKED;
025import static java.nio.charset.StandardCharsets.UTF_8;
026import static java.nio.file.StandardOpenOption.CREATE_NEW;
027
028public class FileKeyGenerator implements KeyGenerator {
029
030    private static final Logger LOGGER = LoggerFactory.getLogger(FileKeyGenerator.class);
031
032    private static final String EMPTY_FILENAME = "";
033
034    private final NaclFacade nacl;
035
036    private final KeyEncryptor keyEncryptor;
037
038    private final PasswordReader passwordReader;
039
040    public FileKeyGenerator(final NaclFacade nacl, final KeyEncryptor keyEncryptor, final PasswordReader passwordReader) {
041        this.nacl = Objects.requireNonNull(nacl);
042        this.keyEncryptor = Objects.requireNonNull(keyEncryptor);
043        this.passwordReader = Objects.requireNonNull(passwordReader);
044    }
045
046    @Override
047    public FilesystemKeyPair generate(final String filename, final ArgonOptions encryptionOptions, final KeyVaultOptions keyVaultOptions) {
048
049        final String password = this.passwordReader.requestUserPassword();
050
051        final KeyPair generated = this.nacl.generateNewKeys();
052
053        final String publicKeyBase64 = Base64.getEncoder().encodeToString(generated.getPublicKey().getKeyBytes());
054
055        final KeyData finalKeys = new KeyData();
056
057        if (!password.isEmpty()) {
058
059            final PrivateKeyData encryptedPrivateKey = this.keyEncryptor.encryptPrivateKey(
060                generated.getPrivateKey(), password, encryptionOptions
061            );
062
063            finalKeys.setConfig(
064                new KeyDataConfig(
065                    new PrivateKeyData(
066                        null,
067                        encryptedPrivateKey.getSnonce(),
068                        encryptedPrivateKey.getAsalt(),
069                        encryptedPrivateKey.getSbox(),
070                        encryptedPrivateKey.getArgonOptions()
071                    ),
072                    LOCKED
073                )
074            );
075
076            LOGGER.info("Newly generated private key has been encrypted");
077
078        } else {
079            
080            String keyData = Base64.getEncoder().encodeToString(generated.getPrivateKey().getKeyBytes());
081
082            finalKeys.setConfig(new KeyDataConfig(new PrivateKeyData(keyData, null, null, null, null), UNLOCKED));
083
084        }
085
086        finalKeys.setPrivateKey(generated.getPrivateKey().encodeToBase64());
087        finalKeys.setPublicKey(publicKeyBase64);
088
089        final String privateKeyJson = JaxbUtil.marshalToString(finalKeys.getConfig());
090
091        final Path resolvedPath = Paths.get(filename).toAbsolutePath();
092        final Path parentPath;
093
094        if(EMPTY_FILENAME.equals(filename)) {
095            parentPath = resolvedPath;
096        } else {
097            parentPath = resolvedPath.getParent();
098        }
099
100        final Path publicKeyPath = parentPath.resolve(filename + ".pub");
101        final Path privateKeyPath = parentPath.resolve(filename + ".key");
102
103        IOCallback.execute(() -> Files.write(publicKeyPath, publicKeyBase64.getBytes(UTF_8), CREATE_NEW));
104        IOCallback.execute(() -> Files.write(privateKeyPath, privateKeyJson.getBytes(UTF_8), CREATE_NEW));
105
106        LOGGER.info("Saved public key to {}", publicKeyPath.toAbsolutePath().toString());
107        LOGGER.info("Saved private key to {}", privateKeyPath.toAbsolutePath().toString());
108
109        final FilesystemKeyPair keyPair = new FilesystemKeyPair(publicKeyPath, privateKeyPath);
110
111        keyPair.withPassword(password);
112
113        return keyPair;
114    }
115
116}
