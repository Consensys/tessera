001package com.quorum.tessera.data.migration;
002
003import java.io.IOException;
004import java.nio.file.Path;
005import java.util.Collections;
006import java.util.HashMap;
007import java.util.Map;
008import java.util.Optional;
009
010public interface StoreLoader {
011
012    Map<byte[], byte[]> load(Path input) throws IOException;
013
014    Map<StoreType, StoreLoader> LOOKUP = Collections.unmodifiableMap(
015        new HashMap<StoreType, StoreLoader>() {{
016            put(StoreType.BDB, new BdbDumpFile());
017            put(StoreType.DIR, new DirectoryStoreFile());
018            put(StoreType.SQLITE, new SqliteLoader());
019        }}
020    );
021
022    static StoreLoader create(StoreType storeType) {
023        return Optional.ofNullable(LOOKUP.get(storeType)).get();
024    }
025
026}
