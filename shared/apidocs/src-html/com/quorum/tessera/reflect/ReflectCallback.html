001package com.quorum.tessera.reflect;
002
003import org.slf4j.Logger;
004import org.slf4j.LoggerFactory;
005
006@FunctionalInterface
007public interface ReflectCallback<T> {
008
009    Logger LOGGER = LoggerFactory.getLogger(ReflectCallback.class);
010
011    T doExecute() throws ReflectiveOperationException;
012
013    static <T> T execute(ReflectCallback<T> callback) {
014        try {
015            return callback.doExecute();
016        } catch (ReflectiveOperationException ex) {
017            LOGGER.error(null, ex);
018            throw new ReflectException(ex);
019        }
020    }
021
022}
