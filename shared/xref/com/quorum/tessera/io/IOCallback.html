1   package com.quorum.tessera.io;

3   import org.slf4j.Logger;
4   import org.slf4j.LoggerFactory;
5   
6   import java.io.IOException;
7   import java.io.UncheckedIOException;
8   
9   /**
10   * Callback and template function to uncheck IO exceptions
11   *
12   * @param <T> the function to execute that throws an {@link IOException}
13   */
14  @FunctionalInterface
15  public interface IOCallback<T> {
16  
17      Logger LOGGER = LoggerFactory.getLogger(IOCallback.class);
18  
19      T doExecute() throws IOException;
20  
21      static <T> T execute(IOCallback<T> callback) {
22          try {
23              return callback.doExecute();
24          } catch (final IOException ex) {
25              LOGGER.debug(null, ex);
26              throw new UncheckedIOException(ex);
27          }
28      }
29  
30  }
