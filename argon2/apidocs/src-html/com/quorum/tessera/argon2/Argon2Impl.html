001package com.quorum.tessera.argon2;
002
003import de.mkammerer.argon2.Argon2Advanced;
004import de.mkammerer.argon2.Argon2Factory;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import java.util.Arrays;
009
010public class Argon2Impl implements Argon2 {
011
012    private static final Logger LOGGER = LoggerFactory.getLogger(Argon2Impl.class);
013
014    private static final ArgonOptions DEFAULT_OPTIONS = new ArgonOptions("i", 10, 1048576, 4);
015
016    @Override
017    public ArgonResult hash(final ArgonOptions options, final String password, final byte[] salt) {
018        final Argon2Advanced argon2 = this.getArgon2Instance(options.getAlgorithm());
019
020        final byte[] hash = argon2.rawHash(
021            options.getIterations(), options.getMemory(), options.getParallelism(), password, salt
022        );
023
024        LOGGER.debug("Argon2 hash produced the array {}", Arrays.toString(hash));
025
026        return new ArgonResult(options, salt, hash);
027    }
028
029    @Override
030    public ArgonResult hash(final String password, final byte[] salt) {
031        return this.hash(DEFAULT_OPTIONS, password, salt);
032    }
033
034    /**
035     * The string form of the algorithm to use.
036     * If an invalid algorithm is chosen, a default of Argon2i is chosen.
037     *
038     * @param algorithm the algorithm to use
039     * @return an instance of the chosen algorithm
040     */
041    private Argon2Advanced getArgon2Instance(final String algorithm) {
042        LOGGER.debug("Searching for the Argon2 algorithm {}", algorithm);
043
044        switch (algorithm) {
045            case "d":
046                return Argon2Factory.createAdvanced(Argon2Factory.Argon2Types.ARGON2d);
047            case "id":
048                return Argon2Factory.createAdvanced(Argon2Factory.Argon2Types.ARGON2id);
049            case "i":
050                return Argon2Factory.createAdvanced(Argon2Factory.Argon2Types.ARGON2i);
051            default:
052                throw new IllegalArgumentException("Invalid Argon2 algorithm " + algorithm);
053        }
054    }
055
056}
