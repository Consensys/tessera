27      @Before
28      public void setUp() {
29          this.delegateFactory = mock(KeyValueOperationsDelegateFactory.class);
30          this.delegate = mock(KeyValueOperationsDelegate.class);
31          when(delegateFactory.create(anyString())).thenReturn(delegate);
32  
33          this.keyVaultService = new HashicorpKeyVaultService(delegateFactory);
34      }

36      @Test
37      public void getSecret() {
38          HashicorpGetSecretData getSecretData = mock(HashicorpGetSecretData.class);
39  
40          when(getSecretData.getSecretEngineName()).thenReturn("secretEngine");
41          when(getSecretData.getSecretName()).thenReturn("secretName");
42          when(getSecretData.getValueId()).thenReturn("keyId");
43  
44          Versioned versionedResponse = mock(Versioned.class);
45  
46          when(delegate.get(any(HashicorpGetSecretData.class))).thenReturn(versionedResponse);
47  
48          when(versionedResponse.hasData()).thenReturn(true);
49  
50          Map responseData = mock(Map.class);
51          when(versionedResponse.getData()).thenReturn(responseData);
52          when(responseData.containsKey("keyId")).thenReturn(true);
53          String keyValue = "keyvalue";
54          when(responseData.get("keyId")).thenReturn(keyValue);
55  
56          String result = keyVaultService.getSecret(getSecretData);
57  
58          assertThat(result).isEqualTo(keyValue);
59      }
60  
61      @Test
62      public void getSecretThrowsExceptionIfNullRetrievedFromVault() {
63          HashicorpGetSecretData getSecretData = new HashicorpGetSecretData("engine", "secretName", "id", 0);
64  
65          when(delegate.get(getSecretData)).thenReturn(null);
66  
67          Throwable ex = catchThrowable(() -> keyVaultService.getSecret(getSecretData));
68  
69          assertThat(ex).isExactlyInstanceOf(HashicorpVaultException.class);
70          assertThat(ex).hasMessage("No data found at engine/secretName");
71      }
72  
73      @Test
74      public void getSecretThrowsExceptionIfNoDataRetrievedFromVault() {

76  
77          Versioned versionedResponse = mock(Versioned.class);
78          when(versionedResponse.hasData()).thenReturn(false);
79  
80          when(delegate.get(getSecretData)).thenReturn(versionedResponse);
81

95          Map responseData = mock(Map.class);

99          when(delegate.get(getSecretData)).thenReturn(versionedResponse);

107     @Test
108     public void setSecretReturnsMetadataObject() {
109         HashicorpSetSecretData setSecretData = new HashicorpSetSecretData("engine", "name", Collections.emptyMap());
110 
111         Versioned.Metadata metadata = mock(Versioned.Metadata.class);
112         when(delegate.set(setSecretData)).thenReturn(metadata);
113 
114         Object result = keyVaultService.setSecret(setSecretData);
115 
116         assertThat(result).isInstanceOf(Versioned.Metadata.class);
117         assertThat(result).isEqualTo(metadata);

120     @Test
121     public void setSecretIfNullPointerExceptionThenHashicorpExceptionThrown() {
122         HashicorpSetSecretData setSecretData = mock(HashicorpSetSecretData.class);
123 
124         when(delegate.set(any(HashicorpSetSecretData.class))).thenThrow(new NullPointerException());
125 
126         Throwable ex = catchThrowable(() -> keyVaultService.setSecret(setSecretData));
127 
128         assertThat(ex).isExactlyInstanceOf(HashicorpVaultException.class);
129         assertThat(ex.getMessage()).isEqualTo("Unable to save generated secret to vault.  Ensure that the secret engine being used is a v2 kv secret engine");
