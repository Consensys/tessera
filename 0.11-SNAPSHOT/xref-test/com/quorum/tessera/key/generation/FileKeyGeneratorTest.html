56          this.keyEncryptor = mock(KeyEncryptor.class);
57          this.passwordReader = mock(PasswordReader.class);
58  
59          when(passwordReader.requestUserPassword()).thenReturn("");

70      @Test
71      public void generateFromKeyDataUnlockedPrivateKey() throws IOException {
72  
73          doReturn(keyPair).when(nacl).generateNewKeys();
74  
75          String filename = UUID.randomUUID().toString();
76          final Path tmpDir = Files.createTempDirectory("keygen").toAbsolutePath().resolve(filename);
77  
78          final FilesystemKeyPair generated = generator.generate(tmpDir.toString(), null, null);
79  
80          assertThat(generated).isInstanceOf(FilesystemKeyPair.class);
81          assertThat(generated.getPublicKey()).isEqualTo("cHVibGljS2V5");
82          assertThat(generated.getPrivateKey()).isEqualTo("cHJpdmF0ZUtleQ==");
83          assertThat(generated.getInlineKeypair().getPrivateKeyConfig().getType()).isEqualTo(UNLOCKED);
84  
85          verify(nacl).generateNewKeys();
86      }

88      @Test
89      public void generateFromKeyDataLockedPrivateKey() throws IOException {
90  
91          when(passwordReader.requestUserPassword()).thenReturn("PASSWORD");
92  
93          final Path tempFolder = Files.createTempDirectory(UUID.randomUUID().toString());
94          final String keyFilesName = tempFolder.resolve(UUID.randomUUID().toString()).toString();
95  
96          doReturn(keyPair).when(nacl).generateNewKeys();
97  
98          final ArgonOptions argonOptions = new ArgonOptions("id", 1, 1, 1);
99  
100         final PrivateKeyData encryptedPrivateKey = new PrivateKeyData(null, null, null, null, argonOptions);
101 
102         doReturn(encryptedPrivateKey).when(keyEncryptor).encryptPrivateKey(any(PrivateKey.class), anyString(), eq(null));
103 
104         final PrivateKeyData encryptedKey = new PrivateKeyData(null, "snonce", "salt", "sbox", argonOptions);
105 
106         doReturn(encryptedKey).when(keyEncryptor).encryptPrivateKey(any(PrivateKey.class), anyString(), eq(null));
107 
108         final FilesystemKeyPair generated = generator.generate(keyFilesName, null, null);
109 
110         final KeyDataConfig pkd = generated.getInlineKeypair().getPrivateKeyConfig();
111         assertThat(generated.getPublicKey()).isEqualTo("cHVibGljS2V5");
112         assertThat(pkd.getSbox()).isEqualTo("sbox");
113         assertThat(pkd.getSnonce()).isEqualTo("snonce");
114         assertThat(pkd.getAsalt()).isEqualTo("salt");
115         assertThat(pkd.getType()).isEqualTo(PrivateKeyType.LOCKED);
116 
117         verify(keyEncryptor).encryptPrivateKey(any(PrivateKey.class), anyString(), eq(null));
118         verify(nacl).generateNewKeys();
119     }

154     @Test
155     public void providingPathThatExistsThrowsError() throws IOException {
156         final Path tempFolder = Files.createTempDirectory(UUID.randomUUID().toString());
157         final String keyFilesName = tempFolder.resolve("key").toString();
158         tempFolder.toFile().setWritable(false);
159 
160         doReturn(keyPair).when(nacl).generateNewKeys();
161 
162         doReturn(new PrivateKeyData("", "", "", "", new ArgonOptions("", 1, 1, 1)))
163                 .when(keyEncryptor)
164                 .encryptPrivateKey(any(PrivateKey.class), anyString(), eq(null));
165 
166         final Throwable throwable = catchThrowable(() -> generator.generate(keyFilesName, null, null));
167 
168         assertThat(throwable).isInstanceOf(UncheckedIOException.class);
169 
170         assertThat(Files.exists(tempFolder.resolve("key.pub"))).isFalse();
171         assertThat(Files.exists(tempFolder.resolve("key.key"))).isFalse();
172 
173         verify(nacl).generateNewKeys();
