5   import com.quorum.tessera.config.ServerConfig;
6   import org.junit.After;

8   import org.junit.Test;
9   
10  import javax.validation.ConstraintValidatorContext;
11  import java.util.ArrayList;
12  import java.util.HashMap;
13  import java.util.Map;
14  import java.util.List;
15  import java.util.stream.Collectors;
16  
17  import static org.assertj.core.api.Assertions.assertThat;
18  import static org.mockito.ArgumentMatchers.anyString;
19  import static org.mockito.Mockito.*;
20  
21  public class ServerConfigsValidatorTest {
22  
23      private Map<AppType, ServerConfig> serverConfigs = new HashMap<>();
24  
25      private ConstraintValidatorContext cvc;
26  
27      private ServerConfigsValidator validator;
28  
29      @Before
30      public void onSetUp() {
31          cvc = mock(ConstraintValidatorContext.class);
32  
33          for (AppType appType : AppType.values()) {
34  
35              ServerConfig serverConfig = new ServerConfig();
36              serverConfig.setApp(appType);
37              serverConfig.setEnabled(true);
38              serverConfig.setServerAddress("localhost:123");
39              serverConfig.setCommunicationType(CommunicationType.REST);
40              serverConfig.setSslConfig(null);
41              serverConfig.setInfluxConfig(null);
42              serverConfig.setBindingAddress(null);
43  
44              serverConfigs.put(appType, serverConfig);
45          }
46  
47          validator = new ServerConfigsValidator();
48  
49          when(cvc.buildConstraintViolationWithTemplate(anyString()))
50                  .thenReturn(mock(ConstraintValidatorContext.ConstraintViolationBuilder.class));
51      }
52  
53      @After
54      public void onTearDown() {
55          verifyNoMoreInteractions(cvc);
56      }
57  
58      @Test
59      public void isValidWhenServerConfigsIsNull() {
60          assertThat(validator.isValid(null, cvc)).isTrue();
61      }
62  
63      @Test
64      public void isValidWhenValidDataIsSupplied() {
65          List<ServerConfig> serverConfigList = serverConfigList();
66  
67          assertThat(validator.isValid(serverConfigList, cvc)).isTrue();
68      }
69  
70      @Test
71      public void isNotValidWhenNoP2PServersAreEnabled() {
72  
73          serverConfigs.get(AppType.P2P).setEnabled(false);
74  
75          assertThat(validator.isValid(serverConfigList(), cvc)).isFalse();
76          verify(cvc).disableDefaultConstraintViolation();
77          verify(cvc).buildConstraintViolationWithTemplate(eq("Only one P2P server must be configured and enabled."));
78      }
79  
80      @Test
81      public void isNotValidWhenNoP2PServersAreDefined() {
82          List<ServerConfig> serverConfigList =
83                  serverConfigList().stream().filter(s -> s.getApp() != AppType.P2P).collect(Collectors.toList());
84          assertThat(validator.isValid(serverConfigList, cvc)).isFalse();
85          verify(cvc).disableDefaultConstraintViolation();
86          verify(cvc).buildConstraintViolationWithTemplate(eq("Only one P2P server must be configured and enabled."));
87      }
88  
89      @Test
90      public void isNotValidWhenTwoOrMoreP2PServersAreDefinedAndEnabled() {
91          List<ServerConfig> serverConfigList = serverConfigList();
92          serverConfigList.add(serverConfigs.get(AppType.P2P));
93          assertThat(validator.isValid(serverConfigList, cvc)).isFalse();
94          verify(cvc).disableDefaultConstraintViolation();
95          verify(cvc).buildConstraintViolationWithTemplate(eq("Only one P2P server must be configured and enabled."));
96      }
97  
98      @Test
99      public void isNotValidWhenNoQ2TServersAreEnabled() {
100         serverConfigs.get(AppType.Q2T).setEnabled(false);
101         assertThat(validator.isValid(serverConfigList(), cvc)).isFalse();
102         verify(cvc).disableDefaultConstraintViolation();
103         verify(cvc).buildConstraintViolationWithTemplate(eq("At least one Q2T server must be configured and enabled."));
104     }
105 
106     @Test
107     public void isNotValidWhenNoQ2TServersAreDefined() {
108         List<ServerConfig> serverConfigList =
109                 serverConfigList().stream().filter(s -> s.getApp() != AppType.Q2T).collect(Collectors.toList());
110 
111         assertThat(validator.isValid(serverConfigList, cvc)).isFalse();
112         verify(cvc).disableDefaultConstraintViolation();
113         verify(cvc).buildConstraintViolationWithTemplate(eq("At least one Q2T server must be configured and enabled."));
114     }
115 
116     private List<ServerConfig> serverConfigList() {
117         return new ArrayList<>(serverConfigs.values());
118     }
119 }
