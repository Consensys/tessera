20      @Rule
21      public SystemErrRule systemErrRule = new SystemErrRule().enableLog();

23      private final ConfigBuilder builderWithValidValues = FixtureUtil.builderWithValidValues();

25      @Test
26      public void nullIsNullAndNotAStringWithTheValueOfNull() {
27          assertThat(ConfigBuilder.toPath(null, null)).isNull();
28          assertThat(ConfigBuilder.toPath(null, "test")).isNotNull();
29      }
30  
31      @Test
32      public void buildValid() {
33          final Config result = this.builderWithValidValues.build();
34  
35          assertThat(result).isNotNull();
36  
37          assertThat(result.getUnixSocketFile()).isEqualTo(Paths.get("somepath.ipc"));
38  
39          assertThat(result.getKeys().getKeyData()).hasSize(1);
40          final ConfigKeyPair keyData = result.getKeys().getKeyData().get(0);
41          assertThat(keyData).isNotNull().extracting("privateKeyPath").containsExactly(Paths.get("private"));
42          assertThat(keyData).isNotNull().extracting("publicKeyPath").containsExactly(Paths.get("public"));
43  
44          final DeprecatedServerConfig serverConfig = result.getServer();
45          assertThat(serverConfig).isNotNull();
46          assertThat(serverConfig.getPort()).isEqualTo(892);
47          assertThat(serverConfig.getHostName()).isEqualTo("http://bogus.com");
48          assertThat(serverConfig.getBindingAddress()).isEqualTo("http://bogus.com:892");
49  
50          final SslConfig sslConfig = serverConfig.getSslConfig();
51          assertThat(sslConfig).isNotNull();
52  
53          assertThat(sslConfig.getClientKeyStorePassword()).isEqualTo("sslClientKeyStorePassword");
54          assertThat(sslConfig.getClientKeyStore()).isEqualTo(Paths.get("sslClientKeyStorePath"));
55          assertThat(sslConfig.getClientTlsKeyPath()).isEqualTo(Paths.get("sslClientTlsKeyPath"));
56          assertThat(sslConfig.getServerTrustCertificates()).containsExactly(Paths.get("sslServerTrustCertificates"));
57  
58          assertThat(result.getJdbcConfig().getUsername()).isEqualTo("jdbcUsername");
59          assertThat(result.getJdbcConfig().getPassword()).isEqualTo("jdbcPassword");
60          assertThat(result.getJdbcConfig().getUrl()).isEqualTo("jdbc:bogus");
61  
62      }
63  
64      @Test
65      public void influxHostNameEmptyThenInfluxConfigIsNull() {
66          final Config result = builderWithValidValues.build();
67  
68          assertThat(result.getServer().getInfluxConfig()).isNull();
69      }

71      @Test
72      public void alwaysSendToFileNotFoundPrintsErrorMessageToTerminal() {
73          List<String> alwaysSendTo = new ArrayList<>();
74          alwaysSendTo.add("doesntexist.txt");
75          alwaysSendTo.add("alsodoesntexist.txt");
76  
77          final ConfigBuilder builder = builderWithValidValues.alwaysSendTo(alwaysSendTo);
78          builder.build();
79  
80          assertThat(systemErrRule.getLog())
81              .isEqualTo("Error reading alwayssendto file: doesntexist.txt\nError reading alwayssendto file: alsodoesntexist.txt\n");
82      }

90      }
