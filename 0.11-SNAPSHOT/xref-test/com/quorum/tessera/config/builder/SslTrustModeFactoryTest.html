3   import com.quorum.tessera.config.SslTrustMode;
4   import java.util.EnumMap;
5   import static org.assertj.core.api.Assertions.assertThat;

8   public class SslTrustModeFactoryTest {
9   
10      @Test
11      public void resolveSslTrustModeNone() {
12  
13          assertThat(SslTrustModeFactory.resolveByLegacyValue(null)).isEqualTo(SslTrustMode.NONE);
14          assertThat(SslTrustModeFactory.resolveByLegacyValue("BOGUS")).isEqualTo(SslTrustMode.NONE);
15      }
16  
17      @Test
18      public void resolveSslTrustMode() {
19  
20          java.util.Map<SslTrustMode, String> fixtures = new EnumMap<>(SslTrustMode.class);
21  
22          fixtures.put(SslTrustMode.CA, "ca");
23          fixtures.put(SslTrustMode.TOFU, "tofu");
24          fixtures.put(SslTrustMode.CA_OR_TOFU, "ca-or-tofu");
25          fixtures.put(SslTrustMode.NONE, "none");
26  
27          for (SslTrustMode mode : fixtures.keySet()) {
28              SslTrustMode result = SslTrustModeFactory.resolveByLegacyValue(fixtures.get(mode));
29              assertThat(result).isEqualTo(mode);
30          }
31      }
32  
33      @Test
34      public void resolveSslTrustModeForCaOrTofu() {
35          SslTrustMode result = SslTrustModeFactory.resolveByLegacyValue("ca-or-tofu");
36          assertThat(result).isEqualTo(SslTrustMode.CA_OR_TOFU);
37  
38      }
39  }
