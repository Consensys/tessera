255     @Test
256     public void keyGetsUpdated() throws IOException, ParseException {
257         final KeyDataConfig startingKey = JaxbUtil.unmarshal(
258             getClass().getResourceAsStream("/lockedprivatekey.json"), KeyDataConfig.class
259         );
260 
261         final Path key = Files.createTempFile("key", ".key");
262         Files.write(key, JaxbUtil.marshalToString(startingKey).getBytes());
263 
264         final String[] args = new String[]{
265             "--keys.keyData.privateKeyPath", key.toString(),
266             "--keys.passwords", "testpassword",
267             "--keys.keyData.config.data.aopts.algorithm", "id",
268             "--keys.keyData.config.data.aopts.memory", "2048",
269             "--keys.keyData.config.data.aopts.iterations", "1",
270             "--keys.keyData.config.data.aopts.parallelism", "1"
271         };
272         final CommandLine commandLine = new DefaultParser().parse(options, args);
273 
274         this.parser.parse(commandLine);
275 
276         final KeyDataConfig endingKey = JaxbUtil.unmarshal(Files.newInputStream(key), KeyDataConfig.class);
277 
278         assertThat(endingKey.getSbox()).isNotEqualTo(startingKey.getSbox());
279         assertThat(endingKey.getSnonce()).isNotEqualTo(startingKey.getSnonce());
280         assertThat(endingKey.getAsalt()).isNotEqualTo(startingKey.getAsalt());
