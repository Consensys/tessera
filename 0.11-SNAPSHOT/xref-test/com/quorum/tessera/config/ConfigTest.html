30          Peer peer = new Peer("Junit");
31          config.addPeer(peer);
32          assertThat(config.getPeers()).containsOnly(peer);

48          Config config = new Config();
49          ServerConfig serverConfig = new ServerConfig();

51          serverConfig.setEnabled(true);
