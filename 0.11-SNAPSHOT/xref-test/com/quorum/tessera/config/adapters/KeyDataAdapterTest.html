21      @Test
22      public void marshallDirectKeys() {

24          final KeyData expected = new KeyData();
25          expected.setPublicKey("PUB");
26          expected.setPrivateKey("PRIV");
27  
28          final KeyData marshalledKey = adapter.marshal(keys);

136
