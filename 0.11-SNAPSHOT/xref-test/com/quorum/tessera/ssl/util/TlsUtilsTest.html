3   import org.bouncycastle.operator.OperatorCreationException;
4   import org.junit.Test;

6   import java.io.IOException;
7   import java.io.InputStream;
8   import java.nio.file.Files;
9   import java.nio.file.Path;
10  import java.security.*;
11  import java.security.cert.Certificate;
12  import java.security.cert.CertificateException;
13  import java.security.cert.X509Certificate;
14  
15  import static org.assertj.core.api.Java6Assertions.assertThat;
16  
17  public class TlsUtilsTest {
18  
19      private static final String PASSWORD = "quorum";
20  
21      private static final String ALIAS = "tessera";
22  
23      @Test
24      public void testGenerateKeys() throws OperatorCreationException, InvalidKeyException, NoSuchAlgorithmException, IOException, SignatureException, NoSuchProviderException, CertificateException, KeyStoreException {
25  
26          final Path privateKeyFile = Files.createTempFile("privatekey", ".tmp");
27          Files.deleteIfExists(privateKeyFile);
28  
29          assertThat(Files.exists(privateKeyFile)).isFalse();
30  
31          TlsUtils.create().generateKeyStoreWithSelfSignedCertificate("https://localhost:8080", privateKeyFile, PASSWORD);
32  
33          assertThat(Files.exists(privateKeyFile)).isTrue();
34  
35          //Read keystore from created file
36          final KeyStore keyStore = KeyStore.getInstance("JKS");
37  
38          try (InputStream in = Files.newInputStream(privateKeyFile)) {
39              keyStore.load(in, PASSWORD.toCharArray());
40          }
41  
42          assertThat(keyStore.containsAlias(ALIAS)).isTrue();
43  
44          Certificate certificate = keyStore.getCertificate(ALIAS);
45  
46          assertThat(certificate).isNotNull();
47          assertThat(certificate).isInstanceOf(X509Certificate.class);
48          assertThat(((X509Certificate) certificate))
49              .extracting("info").isNotNull()
50              .extracting("issuer").isNotNull()
51              .extracting("names").size().isEqualTo(1);
52  
53      }
54  }
