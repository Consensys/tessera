11      @Test
12      public void inputDataIsFoundInStream() {
13  
14          final byte[] inputStreamBytes = ("TRY1" + System.lineSeparator()).getBytes();
15  
16          final ByteArrayInputStream byteArrayInputStream = new ByteArrayInputStream(inputStreamBytes);
17  
18          final PasswordReader passwordReader = new InputStreamPasswordReader(byteArrayInputStream);
19  
20          final String password = passwordReader.readPasswordFromConsole();
21  
22          assertThat(password).isEqualTo("TRY1");
23  
24      }
25  
26      @Test
27      public void testNoLineFoundInStream() {
28          final byte[] inputStreamBytes = new byte[0];
29          final ByteArrayInputStream byteArrayInputStream = new ByteArrayInputStream(inputStreamBytes);
30          final PasswordReader passwordReader = new InputStreamPasswordReader(byteArrayInputStream);
31  
32          final String password = passwordReader.readPasswordFromConsole();
33          assertThat(password).isEmpty();
