14  public class PasswordReaderTest {
15  
16      private final SystemAdapter systemAdapter = SystemAdapter.INSTANCE;
17  
18      @Before
19      public void onSetup() {
20          assertThat(systemAdapter).isInstanceOf(MockSystemAdapter.class);
21  
22          MockSystemAdapter.class.cast(systemAdapter).setErrPrintStream(mock(PrintStream.class));
23          MockSystemAdapter.class.cast(systemAdapter).setOutPrintStream(mock(PrintStream.class));
24      }
25  
26      @Test
27      public void passwordsNotMatchingCausesRetry() {
28          final byte[] systemInBytes = (
29              "TRY1" + System.lineSeparator() +
30              "TRY2" + System.lineSeparator() +
31              "TRY3" + System.lineSeparator() +
32              "TRY3" + System.lineSeparator()
33          ).getBytes();
34  
35          final ByteArrayInputStream byteArrayInputStream = new ByteArrayInputStream(systemInBytes);
36  
37          final PasswordReader passwordReader = new InputStreamPasswordReader(byteArrayInputStream);
38  
39          final String password = passwordReader.requestUserPassword();
40  
41          assertThat(password).isEqualTo("TRY3");
42      }
