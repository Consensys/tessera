32      @Before
33      public void setUp() throws Exception {
34  
35          URL url = EnclaveRestIT.class.getResource("/sample-config.json");
36  
37          CliResult cliResult = CliDelegate.INSTANCE.execute("-configfile", url.getFile());
38  
39          EnclaveFactory enclaveFactory = EnclaveFactory.create();
40  
41          Config config = cliResult.getConfig().get();
42          this.enclave = enclaveFactory.createLocal(config);
43  
44          jersey = Util.create(enclave);
45          jersey.setUp();
46  
47          enclaveClient = new RestfulEnclaveClient(jersey.client(), jersey.target().getUri());
48      }
49  
50      @After
51      public void tearDown() throws Exception {
52          jersey.tearDown();
53      }
54  
55      @Test
56      public void defaultPublicKey() {
57          PublicKey result = enclaveClient.defaultPublicKey();
58  
59          assertThat(result).isNotNull();
60          assertThat(result.encodeToBase64()).isEqualTo("/+UuD63zItL1EbjxkKUljMgG8Z1w0AJ8pNOR4iq2yQc=");
61      }
62  
63      @Test
64      public void forwardingKeys() {
65          Set<PublicKey> result = enclaveClient.getForwardingKeys();
66  
67          assertThat(result).isEmpty();
68      }
69  
70      @Test
71      public void getPublicKeys() {
72          Set<PublicKey> result = enclaveClient.getPublicKeys();
73  
74          assertThat(result).hasSize(1);
75  
76          assertThat(result.iterator().next().encodeToBase64()).isEqualTo("/+UuD63zItL1EbjxkKUljMgG8Z1w0AJ8pNOR4iq2yQc=");
77      }
78  
79      @Test
80      public void status() {
81          assertThat(enclaveClient.status()).isEqualTo(Service.Status.STARTED);
82      }
