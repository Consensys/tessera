3   import com.quorum.tessera.data.EncryptedRawTransactionDAO;

7   import org.springframework.context.annotation.Bean;
8   import org.springframework.context.annotation.ComponentScan;
9   import org.springframework.context.annotation.Configuration;
10  import org.springframework.orm.jpa.JpaTransactionManager;
11  import org.springframework.orm.jpa.LocalContainerEntityManagerFactoryBean;
12  import org.springframework.orm.jpa.vendor.EclipseLinkJpaDialect;
13  import org.springframework.orm.jpa.vendor.EclipseLinkJpaVendorAdapter;
14  
15  import javax.persistence.EntityManagerFactory;
16  import javax.sql.DataSource;
17  import java.util.HashMap;
18  
19  @Configuration
20  @ComponentScan(basePackages = "com.quorum.tessera.data.jpatest")
21  public abstract class JpaConfig {
22  
23      @Bean
24      public EncryptedTransactionDAO someDAO() {
25          return new EncryptedTransactionDAOImpl();
26      }
27  
28      @Bean
29      public EncryptedRawTransactionDAO encryptedRawTransactionDAO() {
30          return new EncryptedRawTransactionDAOImpl();
31      }
32  
33      @Bean
34      public JpaTransactionManager jpaTransactionManager(final EntityManagerFactory entityManagerFactory) {
35          return new JpaTransactionManager(entityManagerFactory);
36      }
37  
38      @Bean
39      public abstract DataSource dataSource();
40  
41      @Bean
42      public LocalContainerEntityManagerFactoryBean entityManagerFactoryBean(final DataSource dataSource) {
43          LocalContainerEntityManagerFactoryBean localContainerEntityManagerFactoryBean =
44                  new LocalContainerEntityManagerFactoryBean();
45          localContainerEntityManagerFactoryBean.setDataSource(dataSource);
46          localContainerEntityManagerFactoryBean.setJpaDialect(new EclipseLinkJpaDialect());
47          localContainerEntityManagerFactoryBean.setJpaVendorAdapter(new EclipseLinkJpaVendorAdapter());
48  
49          localContainerEntityManagerFactoryBean.setJpaPropertyMap(
50                  new HashMap<String, String>() {
51                      {
52                          put("eclipselink.logging.logger", "org.eclipse.persistence.logging.slf4j.SLF4JLogger");
53                          put("eclipselink.weaving", "false");
54                          //  put("eclipselink.logging.level.sql","FINE");
55                          // put("eclipselink.logging.parameters","true");
56                          put("javax.persistence.schema-generation.database.action", "create");
57                          put("javax.persistence.schema-generation.scripts.action", "create");
58                          put(
59                                  "javax.persistence.schema-generation.scripts.create-target",
60                                  "target/" + getCreateScriptName());
61                      }
62                  });
63  
64          /*
65             <property name="jpaVendorAdapter">
66                 <bean class="org.springframework.orm.jpa.vendor.EclipseLinkJpaVendorAdapter">
67                     <property name="showSql" value="true"/>
68                     <property name="generateDdl" value="true"/>
69                     <property name="databasePlatform" value="org.eclipse.persistence.platform.database.H2Platform"/>
70                 </bean>
71             </property>
72          */
73          return localContainerEntityManagerFactoryBean;
74      }
75  
76      public final String getCreateScriptName() {
77          return String.join(".", getClass().getSimpleName(), "ddl");
78      }
79  }
