65          assertThat(def).isEqualToComparingFieldByFieldRecursively(arg);

68      @Test
69      public void nullParametersReturnsFalse() {
70          // First go
71          command.setConfigMixin(null);
72          final CliResult resultOne = command.call();
73          assertThat(resultOne).isEqualToComparingFieldByField(new CliResult(1, true, null));
74  
75          // Second go
76          final ConfigurationMixin configurationMixin = new ConfigurationMixin();
77          command.setConfigMixin(configurationMixin);
78          final CliResult resultTwo = command.call();
79          assertThat(resultTwo).isEqualToComparingFieldByField(new CliResult(1, true, null));
80  
81          // Third go
82          final Config config = new Config();
83          configurationMixin.setConfig(config);
84          final CliResult resultThree = command.call();
85          assertThat(resultThree).isEqualToComparingFieldByField(new CliResult(1, true, null));
86      }

88      @Test
89      public void addPeer() throws Exception {
90          final Entity entity = Entity.entity(new Peer("http://junit.com:8989"), APPLICATION_JSON);
91  
92          final URI uri = UriBuilder.fromPath("/result").build();
93          when(invocationBuilder.put(entity)).thenReturn(Response.created(uri).build());
94  
95          final Path configFile = ElUtil.createAndPopulatePaths(getClass().getResource("/sample-config.json"));
96          final ConfigurationMixin mixin = new ConfigurationMixin();
97          try (InputStream in = Files.newInputStream(configFile)) {
98              final Config config = ConfigFactory.create().create(in, Collections.emptyList());
99              mixin.setConfig(config);
100         }
101 
102         command.setConfigMixin(mixin);
103         command.setPeerUrl("http://junit.com:8989");
104 
105         final CliResult result = command.call();
106         assertThat(result).isEqualToComparingFieldByField(new CliResult(0, true, null));
107 
108         verify(invocationBuilder).put(entity);
109     }
