118     @Test
119     public void toUsingSameInfoFromFixture() {
120 
121         final PartyInfo partyInfo = partyInfoParser.from(dataOne);
122         final byte[] result = partyInfoParser.to(partyInfo);
123 
124         final ByteBuffer byteBuffer = ByteBuffer.wrap(result);
125 
126         assertThat(result).isNotEmpty();
127         assertThat(byteBuffer.getLong()).isEqualTo(21L);
128 
129         byte[] urlData = new byte[21];
130         byteBuffer.get(urlData);
131 
132         final String url = new String(urlData);
133         assertThat(url).isEqualTo(partyInfo.getUrl());
134 
135         long numberOfRecipients = byteBuffer.getLong();
136         assertThat(numberOfRecipients).isEqualTo(1L);
137 
138         long keyByteLength = byteBuffer.getLong();
139         assertThat(keyByteLength).isEqualTo(32L);
140 
141         final byte[] keyData = new byte[32];
142         byteBuffer.get(keyData);
143         assertThat(keyData).hasSize(32)
144                 .isEqualTo(partyInfo.getRecipients().iterator().next().getKey().getKeyBytes());
145 
146         long recipientUrlLength = byteBuffer.getLong();
147         assertThat(recipientUrlLength).isEqualTo(21L);
148 
149         byte[] recipientUrlData = new byte[21];
150         byteBuffer.get(recipientUrlData);
151 
152         String recipientUrl = new String(recipientUrlData);
153         assertThat(recipientUrl).isEqualTo(partyInfo.getRecipients().iterator().next().getUrl());
154 
155         long partyCount = byteBuffer.getLong();
156         assertThat(partyCount).isEqualTo(1L);
157 
158         long partyUrlLength = byteBuffer.getLong();
159         assertThat(partyUrlLength).isEqualTo(22L);
160 
161         byte[] partyUrlData = new byte[22];
162         byteBuffer.get(partyUrlData);
163 
164         String partyUrl = new String(partyUrlData);
165         assertThat(partyUrl).isEqualTo(partyInfo.getParties().iterator().next().getUrl());
166 
167     }
