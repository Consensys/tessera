140     @Test
141     public void autoDiscoveryDisabledOnlyKnownKeysAdded() {
142 
143         when(configService.isDisablePeerDiscovery()).thenReturn(true);
144 
145         Recipient known = new Recipient(PublicKey.from("known".getBytes()), "http://other-node.com:8080");
146         Recipient unknown = new Recipient(PublicKey.from("unknown".getBytes()), "http://unknown.com:8080");

148         final PartyInfo forUpdate =

157         final ArgumentCaptor<PartyInfo> captor = ArgumentCaptor.forClass(PartyInfo.class);
158 
159         verify(partyInfoStore).getPartyInfo();
160         verify(partyInfoStore, times(2)).store(captor.capture());
161         verify(configService).featureToggles();
162 
163         final List<Recipient> allRegisteredKeys =
164                 captor.getAllValues().stream().map(PartyInfo::getRecipients).flatMap(Set::stream).collect(toList());
165 
166         assertThat(allRegisteredKeys)
167                 .hasSize(3)
168                 .containsExactlyInAnyOrder(
169                         new Recipient(PublicKey.from("some-key".getBytes()), URI + "/"),
170                         new Recipient(PublicKey.from("another-public-key".getBytes()), URI + "/"),
171                         new Recipient(PublicKey.from("known".getBytes()), "http://other-node.com:8080"));

279     public void attemptToUpdateRecipientWithExistingKeyWithNewUrlIsIgnoredIfToggleDisabled() {
280         // setup services
281         final FeatureToggles featureToggles = new FeatureToggles();
282         featureToggles.setEnableRemoteKeyValidation(false);
283         when(configService.featureToggles()).thenReturn(featureToggles);
284 
285         // setup data
286         final String uri = "http://localhost:8080";
287 
288         final PublicKey testKey = PublicKey.from("some-key".getBytes());
289         final PartyInfo initial = new PartyInfo(uri, singleton(new Recipient(testKey, uri)), emptySet());
290         when(partyInfoStore.getPartyInfo()).thenReturn(initial);
291 
292         final PublicKey extraKey = PublicKey.from("some-other-key".getBytes());
293 
294         final Set<Recipient> newRecipients =
295                 new HashSet<>(
296                         Arrays.asList(
297                                 new Recipient(testKey, "http://other.com"),
298                                 new Recipient(extraKey, "http://some-other-url.com")));
299         final PartyInfo updated = new PartyInfo(uri, newRecipients, emptySet());
300 
301         // call it
302         final PartyInfo updatedInfo = partyInfoService.updatePartyInfo(updated);
303
