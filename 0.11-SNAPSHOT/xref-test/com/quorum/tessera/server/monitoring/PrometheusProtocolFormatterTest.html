1   package com.quorum.tessera.server.monitoring;
2   
3   import com.quorum.tessera.config.AppType;
4   import org.junit.Before;
5   import org.junit.Test;
6   
7   import java.util.ArrayList;
8   import java.util.List;
9   
10  import static org.assertj.core.api.Assertions.assertThat;
11  
12  public class PrometheusProtocolFormatterTest {
13  
14      private PrometheusProtocolFormatter protocolFormatter;
15  
16      private List<MBeanMetric> mockMetrics;
17  
18      @Before
19      public void setUp() {
20          this.protocolFormatter = new PrometheusProtocolFormatter();
21          this.mockMetrics = new ArrayList<>();
22      }
23  
24      @Test
25      public void noArgResourceResponseCorrectlyFormatted() {
26          mockMetrics.add(new MBeanResourceMetric("GET->upCheck()#a10a4f8d", "AverageTime[ms]_total", "100"));
27  
28          AppType type = AppType.P2P;
29  
30          String expectedResponse = "tessera_P2P_GET_upCheck_AverageTime_ms 100";
31  
32          assertThat(protocolFormatter.format(mockMetrics, type)).isEqualTo(expectedResponse);
33      }
34  
35      @Test
36      public void singleArgResourceResponseCorrectlyFormatted() {
37          mockMetrics.add(new MBeanResourceMetric("POST->resend(ResendRequest)#8ca0a760", "RequestRate[requestsPerSeconds]", "1.3"));
38  
39          AppType type = AppType.P2P;
40  
41          String expectedResponse = "tessera_P2P_POST_resend_ResendRequest_RequestRate_requestsPerSeconds 1.3";
42  
43          assertThat(protocolFormatter.format(mockMetrics, type)).isEqualTo(expectedResponse);
44      }
45  
46      @Test
47      public void singleArrayArgResourceResponseCorrectlyFormatted() {
48          mockMetrics.add(new MBeanResourceMetric("POST->push(byte[])#7f702b7e", "MinTime[ms]_total", "3.4"));
49  
50          AppType type = AppType.P2P;
51  
52          String expectedResponse = "tessera_P2P_POST_push_byte_MinTime_ms 3.4";
53  
54          assertThat(protocolFormatter.format(mockMetrics, type)).isEqualTo(expectedResponse);
55      }
56  
57      @Test
58      public void multipleArgResourceResponseCorrectlyFormatted() {
59          mockMetrics.add(new MBeanResourceMetric("GET->receiveRaw(String;String)#fc8f8357", "AverageTime[ms]_total", "5.2"));
60  
61          AppType type = AppType.P2P;
62  
63          String expectedResponse = "tessera_P2P_GET_receiveRaw_StringString_AverageTime_ms 5.2";
64  
65          assertThat(protocolFormatter.format(mockMetrics, type)).isEqualTo(expectedResponse);
66      }
67  
68      @Test
69      public void multipleMetricsResponseCorrectlyFormatted() {
70          mockMetrics.add(new MBeanResourceMetric("GET->upCheck()#a10a4f8d", "AverageTime[ms]_total", "100"));
71          mockMetrics.add(new MBeanResourceMetric("POST->resend(ResendRequest)#8ca0a760", "RequestRate[requestsPerSeconds]", "1.3"));
72  
73          AppType type = AppType.P2P;
74  
75          String expectedResponse = "tessera_P2P_GET_upCheck_AverageTime_ms 100" + "\n" +
76              "tessera_P2P_POST_resend_ResendRequest_RequestRate_requestsPerSeconds 1.3";
77  
78          assertThat(protocolFormatter.format(mockMetrics, type)).isEqualTo(expectedResponse);
79      }
80  
81      @Test
82      public void noMetricsToFormatIsHandled() {
83          AppType type = AppType.P2P;
84  
85          assertThat(protocolFormatter.format(mockMetrics, type)).isEmpty();
86      }
87  }
