37      @Test
38      public void metricNameDoesNotEndWithTotalSoIsNotIncluded() throws MalformedObjectNameException, IntrospectionException, ReflectionException, AttributeNotFoundException, MBeanException, InstanceNotFoundException {
39          ObjectName mBeanName = new ObjectName("domain", "key", "value");
40          names.add(mBeanName);
41  
42          AppType appType = AppType.P2P;
43          ObjectName objName = new ObjectName("org.glassfish.jersey:type=P2PRestApp,subType=Resources,resource=com.quorum.tessera.api.*,executionTimes=RequestTimes,detail=methods,method=*");
44  
45          when(mBeanServer.queryNames(objName, null)).thenReturn(names);
46  
47          String attributeName = "name";
48          MBeanAttributeInfo[] mBeanAttributes = {new MBeanAttributeInfo(attributeName, "type", "desc", true, false, false)};
49          MBeanInfo mBeanInfo = new MBeanInfo(null, null, mBeanAttributes, null, null, null);
50  
51          when(mBeanServer.getMBeanInfo(mBeanName)).thenReturn(mBeanInfo);
52          when(mBeanServer.getAttribute(any(ObjectName.class), any(String.class))).thenReturn(1);
53  
54          List<MBeanMetric> metrics = metricsEnquirer.getMBeanMetrics(appType);
55  
56          assertThat(metrics.size()).isEqualTo(0);
57      }
