23  public class JerseyServerIT {
24  
25      private URI serverUri = URI.create("http://localhost:8080");
26  
27      private JerseyServer server;
28  
29      @Before
30      public void onSetUp() throws Exception {
31          System.setProperty("sun.net.http.allowRestrictedHeaders", "true");
32  
33          ServerConfig serverConfig = new ServerConfig();
34          serverConfig.setCommunicationType(CommunicationType.REST);
35          serverConfig.setServerAddress("http://localhost:8080");
36          
37          CrossDomainConfig crossDomainConfig = new CrossDomainConfig();
38          crossDomainConfig.setAllowedOrigins(Arrays.asList("*.acme.com", "*.other.com"));
39          serverConfig.setCrossDomainConfig(crossDomainConfig);
40  
41          JaxbUtil.marshalWithNoValidation(serverConfig, System.out);
42  
43          Application sample = new SampleApplication();
44          server = new JerseyServer(serverConfig, sample);
45  
46          server.start();
47      }
48  
49      @After
50      public void onTearDown() throws Exception {
51          server.stop();
52      }
53  
54      @Test
55      public void ping() {
56          
57          MultivaluedMap<String, Object> headers = new MultivaluedHashMap<>();
58          headers.add("Origin", "*.acme.com");
59  
60          Response result = ClientBuilder.newClient()
61              .target(serverUri)
62              .path("ping")
63              .request()
64              .headers(headers)
65              .get();
66  
67          assertThat(result.getHeaderString("Access-Control-Allow-Origin"))
68              .isEqualTo("*.acme.com");
69  
70          assertThat(result.getHeaderString("Access-Control-Allow-Credentials"))
71              .isEqualTo("true");
72  
73          assertThat(result.getStatus()).isEqualTo(200);
74          assertThat(result.readEntity(String.class)).isEqualTo("HEllow");
75      }
76  
77      @Test
78      public void create() {
79  
80          SamplePayload payload = new SamplePayload();
81          payload.setValue("Hellow");
82  
83          Response result = ClientBuilder.newClient()
84              .target(serverUri)
85              .path("create")
86              .request()
87              .post(Entity.entity(payload, MediaType.APPLICATION_JSON));
88  
89          assertThat(result.getStatus()).isEqualTo(201);
90          assertThat(result.getLocation()).isNotNull();
91  
92          Response result2 = ClientBuilder.newClient()
93              .target(result.getLocation())
94              .request(MediaType.APPLICATION_JSON)
95              .get();
96  
97          SamplePayload p = result2.readEntity(SamplePayload.class);
98          assertThat(p).isNotNull();
99          assertThat(p.getValue()).isEqualTo("Hellow");
100 
101         Response result3 = ClientBuilder.newClient()
102             .target(serverUri)
103             .path(p.getId()).request().delete();
104 
105         assertThat(result3.getStatus()).isEqualTo(200);
106         SamplePayload deleted = result3.readEntity(SamplePayload.class);
107         assertThat(deleted.getValue()).isEqualTo("Hellow");
108     }
109 }
