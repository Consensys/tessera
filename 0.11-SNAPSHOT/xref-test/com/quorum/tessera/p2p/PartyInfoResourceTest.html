115     @Test
116     public void partyInfo() {
117 
118         String url = "http://www.bogus.com";
119 
120         PublicKey myKey = PublicKey.from("myKey".getBytes());
121 
122         PublicKey recipientKey = PublicKey.from("recipientKey".getBytes());
123 
124         String message = "I love sparrows";
125 
126         byte[] payload = message.getBytes();
127 
128         Recipient recipient = new Recipient(recipientKey, url);
129 
130         Set<Recipient> recipientList = Collections.singleton(recipient);
131 
132         PartyInfo partyInfo = new PartyInfo(url, recipientList, Collections.emptySet());
133 
134         when(partyInfoParser.from(payload)).thenReturn(partyInfo);
135 
136         when(enclave.defaultPublicKey()).thenReturn(myKey);
137 
138         when(partyInfoParser.to(partyInfo)).thenReturn(payload);
139 
140         EncodedPayload encodedPayload = mock(EncodedPayload.class);
141 
142         List<String> uuidList = new ArrayList<>();
143         doAnswer(
144                         (invocation) -> {
145                             byte[] d = invocation.getArgument(0);
146                             uuidList.add(new String(d));
147                             return encodedPayload;
148                         })
149                 .when(enclave)
150                 .encryptPayload(any(byte[].class), any(PublicKey.class), anyList());
151 
152         when(payloadEncoder.encode(encodedPayload)).thenReturn(payload);
153 
154         WebTarget webTarget = mock(WebTarget.class);
155         when(restClient.target(url)).thenReturn(webTarget);
156         when(webTarget.path(anyString())).thenReturn(webTarget);
157         Invocation.Builder invocationBuilder = mock(Invocation.Builder.class);
158         when(webTarget.request()).thenReturn(invocationBuilder);
159 
160         Response response = mock(Response.class);
161         when(response.getStatus()).thenReturn(200);
162 
163         doAnswer((invocation) -> uuidList.get(0)).when(response).readEntity(String.class);
164 
165         when(invocationBuilder.post(any(Entity.class))).thenReturn(response);
166 
167         when(partyInfoService.updatePartyInfo(any(PartyInfo.class))).thenReturn(partyInfo);
168 
169         Response result = partyInfoResource.partyInfo(payload);
170 
171         assertThat(result.getStatus()).isEqualTo(200);
172 
173         verify(partyInfoParser).from(payload);
174         verify(enclave).defaultPublicKey();
175         verify(enclave).encryptPayload(any(byte[].class), any(PublicKey.class), anyList());
176         verify(payloadEncoder).encode(encodedPayload);
177         verify(restClient).target(url);
178         verify(partyInfoService).updatePartyInfo(any(PartyInfo.class));
179     }

321     @Test
322     public void validationDisabledPassesAllKeysToStore() {
323         this.partyInfoResource =
324                 new PartyInfoResource(partyInfoService, partyInfoParser, restClient, enclave, payloadEncoder, false);
325 
326         final byte[] payload = "Test message".getBytes();
327 
328         final String url = "http://www.bogus.com";
329         final String otherurl = "http://www.randomaddress.com";
330         final PublicKey recipientKey = PublicKey.from("recipientKey".getBytes());
331         final Set<Recipient> recipientList =
332                 new HashSet<>(Arrays.asList(new Recipient(recipientKey, url), new Recipient(recipientKey, otherurl)));
333         final PartyInfo partyInfo = new PartyInfo(url, recipientList, Collections.emptySet());
334 
335         final ArgumentCaptor<PartyInfo> captor = ArgumentCaptor.forClass(PartyInfo.class);
336         final byte[] serialisedData = "SERIALISED".getBytes();
337 
338         when(partyInfoParser.from(payload)).thenReturn(partyInfo);
339         when(partyInfoService.getPartyInfo()).thenReturn(partyInfo);
340         when(partyInfoParser.to(captor.capture())).thenReturn(serialisedData);
341 
342         final Response callResponse = partyInfoResource.partyInfo(payload);
343         final byte[] data = (byte[]) callResponse.getEntity();
344 
345         assertThat(captor.getValue().getUrl()).isEqualTo(url);
346         assertThat(captor.getValue().getRecipients()).isEmpty();
347         assertThat(captor.getValue().getParties()).isEmpty();
348         assertThat(new String(data)).isEqualTo("SERIALISED");
349         verify(partyInfoParser).from(payload);
350         verify(partyInfoParser).to(any(PartyInfo.class));
351         verify(partyInfoService).updatePartyInfo(partyInfo);
352         verify(partyInfoService).getPartyInfo();
353     }
