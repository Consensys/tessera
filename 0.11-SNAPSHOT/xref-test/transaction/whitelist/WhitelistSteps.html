34  public class WhitelistSteps implements En {
35  
36      private static final Logger LOGGER = LoggerFactory.getLogger(WhitelistSteps.class);
37  
38      private static String jarPath =
39              System.getProperty("application.jar", "../../tessera-app/target/tessera-app-0.9-SNAPSHOT-app.jar");
40  
41      private final URL logbackConfigFile = WhitelistSteps.class.getResource("/logback-node.xml");
42  
43      public WhitelistSteps() {
44  
45          try {
46              final Path pid = Files.createTempFile("whitelist", ".pid");
47  
48              ExecutorService executorService = Executors.newCachedThreadPool();
49  
50              Given(
51                      "Node at port {int}",
52                      (Integer port) -> {
53                          ExecutionContext executionContext =
54                                  ExecutionContext.Builder.create()
55                                          .with(CommunicationType.REST)
56                                          .with(DBType.H2)
57                                          .with(EnclaveType.LOCAL)
58                                          .with(SocketType.HTTP)
59                                          .build();
60  
61                          ConfigBuilder whiteListConfigBuilder =
62                                  new ConfigBuilder()
63                                          .withNodeId("whitelist")
64                                          .withNodeNumber(5)
65                                          .withQ2TSocketType(SocketType.HTTP)
66                                          .withExecutionContext(executionContext)
67                                          .withP2pPort(port)
68                                          .withPeer("http://localhost:7000")
69                                          .withKeys(
70                                                  "WxsJ4souK0mptNx1UGw6hb1WNNIbPhLPvW9GoaXau3Q=",
71                                                  "YbOOFA4mwSSdGH6aFfGl2M7N1aiPOj5nHpD7GzJKSiA=");
72  
73                          Config whiteListConfig = whiteListConfigBuilder.build();
74                          whiteListConfig.setUseWhiteList(true);
75  
76                          Path configFile = Paths.get("target").resolve("white-list-config.json");
77  
78                          try (OutputStream out = Files.newOutputStream(configFile)) {
79                              JaxbUtil.marshalWithNoValidation(whiteListConfig, out);
80                          } catch (IOException ex) {
81                              throw new UncheckedIOException(ex);
82                          }
83  
84                          List<String> cmd =
85                                  new ExecArgsBuilder()
86                                          .withMainClass(Main.class)
87                                          .withClassPathItem(Paths.get(jarPath))
88                                          .withConfigFile(configFile)
89                                          .withJvmArg("-Dlogback.configurationFile=" + logbackConfigFile)
90                                          .withJvmArg("-Dnode.number=whitelist")
91                                          .withPidFile(pid)
92                                          .build();
93  
94                          ProcessBuilder processBuilder = new ProcessBuilder(cmd);
95                          processBuilder.redirectErrorStream(true);
96  
97                          Process process = processBuilder.start();
98  
99                          executorService.submit(
100                                 () -> {
101                                     try (BufferedReader reader =
102                                             Stream.of(process.getInputStream())
103                                                     .map(InputStreamReader::new)
104                                                     .map(BufferedReader::new)
105                                                     .findAny()
106                                                     .get()) {
107 
108                                         String line;
109                                         while ((line = reader.readLine()) != null) {
110                                             LOGGER.info("Exec line Whitelist : {}", line);
111                                         }
112 
113                                     } catch (IOException ex) {
114                                         throw new UncheckedIOException(ex);
115                                     }
116                                 });
117 
118                         executorService.submit(
119                                 () -> {
120                                     try {
121                                         process.waitFor();
122                                     } catch (InterruptedException ex) {
123 
124                                     }
125                                 });
126 
127                         ServerStatusCheck serverStatusCheck =
128                                 ServerStatusCheck.create(
129                                         whiteListConfig.getServerConfigs().stream()
130                                                 .filter(s -> s.getApp() == AppType.P2P)
131                                                 .findAny()
132                                                 .get());
133 
134                         Boolean started =
135                                 executorService
136                                         .submit(new ServerStatusCheckExecutor(serverStatusCheck))
137                                         .get(20, TimeUnit.SECONDS);
138 
139                         assertThat(started).isTrue();
140                     });
141 
142             List<Response> responseHolder = new ArrayList<>();
143             When(
144                     "a request is made against the node",
145                     () -> {
146                         Client client = ClientBuilder.newClient();
147                         Response response = client.target("http://localhost:7000").path("upcheck").request().get();
148 
149                         responseHolder.add(response);
150                     });
151           
152             Then(
153                     "the response code is UNAUTHORIZED",
154                     () -> assertThat(responseHolder.get(0).getStatus()).isEqualTo(401));
155 
156             Then(
157                     "the node is stopped",
158                     () -> {
159                         Files.lines(pid)
160                                 .findAny()
161                                 .ifPresent(
162                                         p -> {
163                                             try {
164                                                 ExecUtils.kill(p);
165                                             } catch (IOException | InterruptedException ex) {
166                                             }
167                                         });
168                     });
169 
170         } catch (Exception ex) {
171             throw new RuntimeException(ex);
172         }
173     }
