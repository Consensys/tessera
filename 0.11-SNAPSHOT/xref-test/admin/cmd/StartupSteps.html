8   import java.nio.file.Files;
9   import java.nio.file.Path;
10  import java.util.ArrayList;
11  import java.util.HashMap;
12  import java.util.List;
13  import java.util.Map;
14  import java.util.UUID;
15  import static org.assertj.core.api.Assertions.*;
16  
17  public class StartupSteps implements En {
18  
19      public StartupSteps() {
20  
21          List<Party> partyHolder = new ArrayList<>();
22  
23          Given(
24                  "configuration file with empty public and private key values",
25                  () -> {
26                      URL url = getClass().getResource("/empty-keys-config.json");
27  
28                      Party party = new Party("", url, "X");
29                      Config config = party.getConfig();
30                      assertThat(config.getKeys().getKeyData().get(0).getPrivateKey()).isEmpty();
31                      assertThat(config.getKeys().getKeyData().get(0).getPublicKey()).isEmpty();
32                      partyHolder.add(party);
33                  });
34  
35          Given(
36                  "configuration file with with key paths containing empty values",
37                  () -> {
38                      URL url = getClass().getResource("/empty-keyspath-config.json");
39                      Path emptyKeyFile = Files.createTempFile(UUID.randomUUID().toString(), "");
40                      Path ipcFile = Files.createTempFile(UUID.randomUUID().toString(), "");
41                      Map<String, String> params = new HashMap<>();
42                      params.put("emptyKeyFilePath", emptyKeyFile.toAbsolutePath().toString());
43                      params.put("unixSocketPath", ipcFile.toAbsolutePath().toString());
44                      Path configFile = ElUtil.createTempFileFromTemplate(url, params);
45                      Party party = new Party("", configFile.toUri().toURL(), "X");
46                      Config config = party.getConfig();
47                      assertThat(emptyKeyFile).exists();
48  
49                      partyHolder.add(party);
50                  });
51  
52          List<ExecutionResult> results = new ArrayList<>();
53          When(
54                  "admin user executes start",
55                  () -> {
56                      assertThat(partyHolder).hasSize(1);
57                      Party party = partyHolder.iterator().next();
58                      ExecutionResult result = Utils.start(party);
59                      results.add(result);
60                  });
61  
62          Then(
63                  "node returns error message and exits",
64                  () -> {
65                      assertThat(results).hasSize(1);
66                      ExecutionResult result = results.get(0);
67                      assertThat(result.getExitCode()).isNotEqualTo(0);
68                      assertThat(result.getOutput()).hasSize(2);
69  
70                      assertThat(result.getOutput())
71                              .anyMatch(m -> m.startsWith("Config validation issue: keys.keyData[0].privateKey"));
72  
73                      assertThat(result.getOutput())
74                              .anyMatch(m -> m.startsWith("Config validation issue: keys.keyData[0].publicKey"));
75                  });
76      }
77  }
