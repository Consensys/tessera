8       public static String generate(ExecutionContext executionContext) {
9   
10          List<String> tokens = new ArrayList<>();
11          executionContext.getPrefix()
12              .ifPresent(v -> tokens.add(v));
13          tokens.add(executionContext.getCommunicationType().name().toLowerCase());
14          tokens.add(executionContext.getSocketType().name().toLowerCase());
15  
16          return String.join("-",tokens);
17      }
18      
19      
20      public static String generate(ExecutionContext executionContext,NodeAlias alias) {
21          if(executionContext.isAdmin()) {
22              return "admin";
23          }
24          List<String> tokens = new ArrayList<>();
25          executionContext.getPrefix()
26              .ifPresent(v -> tokens.add(v));
27          tokens.add(executionContext.getCommunicationType().name().toLowerCase());
28          tokens.add(executionContext.getSocketType().name().toLowerCase());
29          tokens.add(executionContext.getEnclaveType().name().toLowerCase());
30          tokens.add(alias.name().toLowerCase());
31  
32          return String.join("-",tokens);
33      }
