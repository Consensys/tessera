1   package suite;
2   
3   import com.quorum.tessera.config.CommunicationType;
4   import com.quorum.tessera.config.ServerConfig;
5   import com.quorum.tessera.io.IOCallback;
6   import java.net.URL;
7   import javax.ws.rs.core.UriBuilder;
8   import org.slf4j.Logger;
9   import org.slf4j.LoggerFactory;
10  
11  public interface ServerStatusCheck {
12  
13      Logger LOGGER = LoggerFactory.getLogger(ServerStatusCheck.class);
14  
15      boolean checkStatus();
16  
17      static ServerStatusCheck create(ServerConfig serverConfig) {
18          CommunicationType communicationType = serverConfig.getCommunicationType();
19  
20          if (communicationType == CommunicationType.REST) {
21  
22              if (serverConfig.isUnixSocket()) {
23                  return new UnixSocketServerStatusCheck(serverConfig.getServerUri());
24              } else {
25                  final URL url =
26                          IOCallback.execute(
27                                  () -> UriBuilder.fromUri(serverConfig.getServerUri()).path("upcheck").build().toURL());
28                  if (serverConfig.isSsl()) {
29                      return new HttpsServerStatusCheck(url, serverConfig.getSslConfig());
30                  } else {
31                      return new HttpServerStatusCheck(url);
32                  }
33              }
34          }
35  
36          if (communicationType == CommunicationType.GRPC) {
37              URL grpcUrl =
38                      IOCallback.execute(
39                              () -> UriBuilder.fromUri(serverConfig.getBindingUri()).path("upcheck").build().toURL());
40              return new GrpcServerStatusCheck(grpcUrl, serverConfig.getApp());
41          }
42  
43          throw new UnsupportedOperationException("Unable to cerate server check for " + serverConfig);
44      }
45  }
