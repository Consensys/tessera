001package com.quorum.tessera.config;
002
003import com.quorum.tessera.config.adapters.PathAdapter;
004import com.quorum.tessera.config.constraints.*;
005
006import javax.validation.Valid;
007import javax.validation.constraints.NotNull;
008import javax.xml.bind.annotation.*;
009import javax.xml.bind.annotation.adapters.XmlJavaTypeAdapter;
010import java.nio.file.Path;
011import java.util.ArrayList;
012import java.util.Collections;
013import java.util.List;
014
015@XmlRootElement
016@XmlAccessorType(XmlAccessType.FIELD)
017@ValidEitherServerConfigsOrServer
018public class Config extends ConfigItem {
019
020    @XmlAttribute private String version = Version.getVersion();
021
022    @NotNull
023    @Valid
024    @XmlElement(name = "jdbc", required = true)
025    private JdbcConfig jdbcConfig;
026
027    @Valid
028    @ValidServerConfigs
029    @XmlElement(name = "serverConfigs", required = true)
030    private List<@Valid @ValidServerConfig ServerConfig> serverConfigs;
031
032    @NotNull
033    @Valid
034    @XmlElement(name = "peer", required = true)
035    private List<Peer> peers;
036
037    @Valid
038    @XmlElement(required = true)
039    @ValidKeyConfiguration
040    @ValidKeyVaultConfiguration
041    private KeyConfiguration keys;
042
043    @NotNull
044    @XmlElement(name = "alwaysSendTo")
045    private List<@ValidBase64 String> alwaysSendTo = new ArrayList<>();
046
047    @ValidPath(checkCanCreate = true)
048    @XmlElement(required = true, type = String.class)
049    @XmlJavaTypeAdapter(PathAdapter.class)
050    private Path unixSocketFile;
051
052    @XmlAttribute private boolean useWhiteList;
053
054    @XmlAttribute private boolean disablePeerDiscovery;
055
056    @XmlElement private DeprecatedServerConfig server;
057
058    @XmlElement private FeatureToggles features = new FeatureToggles();
059
060    @Deprecated
061    public Config(
062            final JdbcConfig jdbcConfig,
063            final List<ServerConfig> serverConfigs,
064            final List<Peer> peers,
065            final KeyConfiguration keyConfiguration,
066            final List<String> alwaysSendTo,
067            final Path unixSocketFile,
068            final boolean useWhiteList,
069            final boolean disablePeerDiscovery) {
070        this.jdbcConfig = jdbcConfig;
071        this.serverConfigs = serverConfigs;
072        this.peers = peers;
073        this.keys = keyConfiguration;
074        this.alwaysSendTo = alwaysSendTo;
075        this.unixSocketFile = unixSocketFile;
076        this.useWhiteList = useWhiteList;
077        this.disablePeerDiscovery = disablePeerDiscovery;
078    }
079
080    public Config() {}
081
082    public JdbcConfig getJdbcConfig() {
083        return this.jdbcConfig;
084    }
085
086    // TODO: Shouldn't need to laziely recalcuate on a getter
087    public List<ServerConfig> getServerConfigs() {
088        if (null != this.serverConfigs) {
089            return this.serverConfigs;
090        }
091        return DeprecatedServerConfig.from(server, unixSocketFile);
092    }
093
094    public boolean isServerConfigsNull() {
095        return null == this.serverConfigs;
096    }
097
098    @Deprecated
099    public Path getUnixSocketFile() {
100        return unixSocketFile;
101    }
102
103    public List<Peer> getPeers() {
104        if (peers == null) {
105            return null;
106        }
107        return Collections.unmodifiableList(peers);
108    }
109
110    public KeyConfiguration getKeys() {
111        return this.keys;
112    }
113
114    public List<String> getAlwaysSendTo() {
115        return this.alwaysSendTo;
116    }
117
118    public boolean isUseWhiteList() {
119        return this.useWhiteList;
120    }
121
122    public boolean isDisablePeerDiscovery() {
123        return disablePeerDiscovery;
124    }
125
126    public void addPeer(Peer peer) {
127        if (peers == null) {
128            this.peers = new ArrayList<>();
129        }
130        this.peers.add(peer);
131    }
132
133    public ServerConfig getP2PServerConfig() {
134        // TODO need to revisit
135        return getServerConfigs().stream()
136                .filter(ServerConfig::isEnabled)
137                .filter(sc -> sc.getApp() == AppType.P2P)
138                .findFirst()
139                .orElse(null);
140    }
141
142    @Deprecated
143    public DeprecatedServerConfig getServer() {
144        return server;
145    }
146
147    @Deprecated
148    public void setServer(DeprecatedServerConfig server) {
149        this.server = server;
150    }
151
152    public void setJdbcConfig(JdbcConfig jdbcConfig) {
153        this.jdbcConfig = jdbcConfig;
154    }
155
156    public void setServerConfigs(List<ServerConfig> serverConfigs) {
157        this.serverConfigs = serverConfigs;
158    }
159
160    public void setPeers(List<Peer> peers) {
161        this.peers = peers;
162    }
163
164    public void setKeys(KeyConfiguration keys) {
165        this.keys = keys;
166    }
167
168    public void setAlwaysSendTo(List<String> alwaysSendTo) {
169        this.alwaysSendTo = alwaysSendTo;
170    }
171
172    @Deprecated
173    public void setUnixSocketFile(Path unixSocketFile) {
174        this.unixSocketFile = unixSocketFile;
175    }
176
177    public void setUseWhiteList(boolean useWhiteList) {
178        this.useWhiteList = useWhiteList;
179    }
180
181    public void setDisablePeerDiscovery(boolean disablePeerDiscovery) {
182        this.disablePeerDiscovery = disablePeerDiscovery;
183    }
184
185    public String getVersion() {
186        return version;
187    }
188
189    public FeatureToggles getFeatures() {
190        return features;
191    }
192
193    public void setFeatures(final FeatureToggles features) {
194        this.features = features;
195    }
196}
