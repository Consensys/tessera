001package com.quorum.tessera.config;
002
003import java.util.Optional;
004import javax.validation.constraints.NotNull;
005import javax.validation.constraints.Pattern;
006import javax.xml.bind.annotation.XmlAccessType;
007import javax.xml.bind.annotation.XmlAccessorType;
008import javax.xml.bind.annotation.XmlAttribute;
009
010@XmlAccessorType(XmlAccessType.FIELD)
011public class ArgonOptions extends ConfigItem {
012
013    @Pattern(regexp = "^(id|i|d)$")
014    @XmlAttribute(name = "variant")
015    private String algorithm;
016
017    @NotNull
018    @XmlAttribute
019    private Integer iterations;
020
021    @NotNull
022    @XmlAttribute
023    private Integer memory;
024
025    @NotNull
026    @XmlAttribute
027    private Integer parallelism;
028
029    public ArgonOptions(String algorithm, Integer iterations, Integer memory, Integer parallelism) {
030        this.algorithm = Optional.ofNullable(algorithm).orElse("id");
031        this.iterations = iterations;
032        this.memory = memory;
033        this.parallelism = parallelism;
034    }
035
036    public ArgonOptions() {
037
038    }
039
040    public String getAlgorithm() {
041        return algorithm;
042    }
043
044    public Integer getIterations() {
045        return iterations;
046    }
047
048    public Integer getMemory() {
049        return memory;
050    }
051
052    public Integer getParallelism() {
053        return parallelism;
054    }
055
056    public void setAlgorithm(String algorithm) {
057        this.algorithm = algorithm;
058    }
059
060    public void setIterations(Integer iterations) {
061        this.iterations = iterations;
062    }
063
064    public void setMemory(Integer memory) {
065        this.memory = memory;
066    }
067
068    public void setParallelism(Integer parallelism) {
069        this.parallelism = parallelism;
070    }
071
072}
