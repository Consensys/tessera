001package com.quorum.tessera.config;
002
003import java.util.Arrays;
004import java.util.List;
005import javax.xml.bind.annotation.XmlAccessType;
006import javax.xml.bind.annotation.XmlAccessorType;
007import javax.xml.bind.annotation.XmlElement;
008
009@XmlAccessorType(XmlAccessType.FIELD)
010public class CrossDomainConfig extends ConfigItem {
011
012    @XmlElement
013    private List<String> allowedMethods = Arrays.asList("GET", "POST", "PUT", "DELETE", "OPTIONS", "HEAD");
014
015    @XmlElement
016    private List<String> allowedOrigins;
017
018    @XmlElement
019    private List<String> allowedHeaders;
020
021    @XmlElement
022    private Boolean allowCredentials = Boolean.TRUE;
023    
024    public List<String> getAllowedOrigins() {
025        return allowedOrigins;
026    }
027
028    public void setAllowedOrigins(List<String> allowedOrigins) {
029        this.allowedOrigins = allowedOrigins;
030    }
031
032    public List<String> getAllowedMethods() {
033        return allowedMethods;
034    }
035
036    public void setAllowedMethods(List<String> allowedMethods) {
037        this.allowedMethods = allowedMethods;
038    }
039
040    public List<String> getAllowedHeaders() {
041        return allowedHeaders;
042    }
043
044    public void setAllowedHeaders(List<String> allowedHeaders) {
045        this.allowedHeaders = allowedHeaders;
046    }
047
048    public Boolean getAllowCredentials() {
049        return allowCredentials;
050    }
051
052    public void setAllowCredentials(Boolean allowCredentials) {
053        this.allowCredentials = allowCredentials;
054    }
055
056}
