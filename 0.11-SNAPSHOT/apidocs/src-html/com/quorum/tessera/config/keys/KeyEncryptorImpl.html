001package com.quorum.tessera.config.keys;
002
003import com.quorum.tessera.argon2.Argon2;
004import com.quorum.tessera.argon2.ArgonResult;
005import com.quorum.tessera.config.ArgonOptions;
006import com.quorum.tessera.config.PrivateKeyData;
007import com.quorum.tessera.encryption.PrivateKey;
008import com.quorum.tessera.encryption.SharedKey;
009import com.quorum.tessera.nacl.NaclFacade;
010import com.quorum.tessera.nacl.Nonce;
011import org.slf4j.Logger;
012import org.slf4j.LoggerFactory;
013
014import java.security.SecureRandom;
015import java.util.Arrays;
016import java.util.Base64;
017import java.util.Objects;
018
019/**
020 * An implementation of {@link KeyEncryptor} that uses Argon2
021 * <p>
022 * The password is hashed using the generated/provided salt to generate a 32
023 * byte hash This hash is then used as the symmetric key to encrypt the private
024 * key
025 */
026public class KeyEncryptorImpl implements KeyEncryptor {
027
028    private static final Logger LOGGER = LoggerFactory.getLogger(KeyEncryptorImpl.class);
029
030    private final Argon2 argon2;
031
032    private final NaclFacade nacl;
033
034    private final Base64.Decoder decoder = Base64.getDecoder();
035
036    private final Base64.Encoder encoder = Base64.getEncoder();
037
038    private final SecureRandom secureRandom = new SecureRandom();
039
040    public KeyEncryptorImpl(final Argon2 argon2, final NaclFacade nacl) {
041        this.argon2 = Objects.requireNonNull(argon2);
042        this.nacl = Objects.requireNonNull(nacl);
043    }
044
045    @Override
046    public PrivateKeyData encryptPrivateKey(final PrivateKey privateKey, final String password, final ArgonOptions argonOptions) {
047
048        LOGGER.info("Encrypting a private key");
049
050        LOGGER.debug("Encrypting private key {} using password {}", privateKey, password);
051
052        final byte[] salt = new byte[KeyEncryptor.SALTLENGTH];
053        this.secureRandom.nextBytes(salt);
054
055        LOGGER.debug("Generated the random salt {}", Arrays.toString(salt));
056
057        final ArgonResult argonResult;
058
059        if (argonOptions == null) {
060            argonResult = this.argon2.hash(password, salt);
061        } else {
062            argonResult = this.argon2.hash(
063                    new com.quorum.tessera.argon2.ArgonOptions(
064                            argonOptions.getAlgorithm(),
065                            argonOptions.getIterations(),
066                            argonOptions.getMemory(),
067                            argonOptions.getParallelism()
068                    ),
069                    password,
070                    salt
071            );
072        }
073
074        final Nonce nonce = this.nacl.randomNonce();
075        LOGGER.debug("Generated the random nonce {}", nonce);
076
077        final byte[] encryptedKey = this.nacl.sealAfterPrecomputation(
078                privateKey.getKeyBytes(), nonce, SharedKey.from(argonResult.getHash())
079        );
080
081        LOGGER.info("Private key encrypted");
082
083        final String snonce = this.encoder.encodeToString(nonce.getNonceBytes());
084        final String asalt = this.encoder.encodeToString(salt);
085        final String sbox = this.encoder.encodeToString(encryptedKey);
086
087        return new PrivateKeyData(
088                null,
089                snonce,
090                asalt,
091                sbox,
092                new ArgonOptions(
093                        argonResult.getOptions().getAlgorithm(),
094                        argonResult.getOptions().getIterations(),
095                        argonResult.getOptions().getMemory(),
096                        argonResult.getOptions().getParallelism()
097                )
098        );
099    }
100
101    @Override
102    public PrivateKey decryptPrivateKey(final PrivateKeyData privateKey, final String password) {
103
104        LOGGER.info("Decrypting private key");
105        LOGGER.debug("Decrypting private key {} using password {}", privateKey.getValue(), password);
106
107        final byte[] salt = this.decoder.decode(privateKey.getAsalt());
108
109        final ArgonResult argonResult = this.argon2.hash(
110                new com.quorum.tessera.argon2.ArgonOptions(
111                        privateKey.getArgonOptions().getAlgorithm(),
112                        privateKey.getArgonOptions().getIterations(),
113                        privateKey.getArgonOptions().getMemory(),
114                        privateKey.getArgonOptions().getParallelism()
115                ),
116                password,
117                salt
118        );
119
120        final byte[] originalKey = this.nacl.openAfterPrecomputation(
121                this.decoder.decode(privateKey.getSbox()),
122                new Nonce(this.decoder.decode(privateKey.getSnonce())),
123                SharedKey.from(argonResult.getHash())
124        );
125
126        PrivateKey outcome = PrivateKey.from(originalKey);
127
128        LOGGER.info("Decrypted private key");
129        LOGGER.debug("Decrypted private key {}", outcome.encodeToBase64());
130
131        return outcome;
132    }
133
134}
