001package com.quorum.tessera.config;
002
003import javax.validation.constraints.NotNull;
004import javax.xml.bind.annotation.XmlAccessType;
005import javax.xml.bind.annotation.XmlAccessorType;
006import javax.xml.bind.annotation.XmlElement;
007
008@XmlAccessorType(XmlAccessType.FIELD)
009public class JdbcConfig extends ConfigItem {
010
011    @XmlElement private String username;
012
013    @XmlElement private String password;
014
015    @NotNull
016    @XmlElement(required = true)
017    private String url;
018
019    /** Auto create tables if no exists */
020    @XmlElement(defaultValue = "false")
021    private boolean autoCreateTables;
022
023    public JdbcConfig(String username, String password, String url) {
024        this.username = username;
025        this.password = password;
026        this.url = url;
027    }
028
029    public JdbcConfig() {
030        this(null, null, null);
031    }
032
033    public String getUsername() {
034        return username;
035    }
036
037    public String getPassword() {
038        return password;
039    }
040
041    public String getUrl() {
042        return url;
043    }
044
045    public void setUsername(String username) {
046        this.username = username;
047    }
048
049    public void setPassword(String password) {
050        this.password = password;
051    }
052
053    public void setUrl(String url) {
054        this.url = url;
055    }
056
057    public boolean isAutoCreateTables() {
058        return autoCreateTables;
059    }
060
061    public void setAutoCreateTables(boolean autoCreateTables) {
062        this.autoCreateTables = autoCreateTables;
063    }
064}
