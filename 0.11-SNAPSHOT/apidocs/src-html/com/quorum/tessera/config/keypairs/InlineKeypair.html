001package com.quorum.tessera.config.keypairs;
002
003import com.quorum.tessera.config.KeyDataConfig;
004import com.quorum.tessera.config.PrivateKeyData;
005import com.quorum.tessera.config.constraints.ValidBase64;
006import com.quorum.tessera.config.keys.KeyEncryptorFactory;
007import com.quorum.tessera.nacl.NaclException;
008
009import javax.validation.constraints.NotNull;
010import javax.validation.constraints.Pattern;
011import javax.validation.constraints.Size;
012import javax.xml.bind.annotation.XmlElement;
013
014import java.util.Objects;
015
016import static com.quorum.tessera.config.PrivateKeyType.UNLOCKED;
017
018public class InlineKeypair implements ConfigKeyPair {
019
020    @XmlElement
021    private final String publicKey;
022
023    @NotNull
024    @XmlElement(name = "config")
025    private final KeyDataConfig privateKeyConfig;
026
027    private String password;
028
029    private String cachedValue;
030
031    private String cachedPassword;
032
033    public InlineKeypair(final String publicKey, final KeyDataConfig privateKeyConfig) {
034        this.publicKey = publicKey;
035        this.privateKeyConfig = privateKeyConfig;
036    }
037
038    public KeyDataConfig getPrivateKeyConfig() {
039        return this.privateKeyConfig;
040    }
041
042    @Override
043    @Size(min = 1)
044    @NotNull
045    @ValidBase64(message = "Invalid Base64 key provided")
046    public String getPublicKey() {
047        return this.publicKey;
048    }
049
050    @Override
051    @NotNull
052    @Size(min = 1)
053    @ValidBase64(message = "Invalid Base64 key provided")
054    @Pattern(regexp = "^((?!NACL_FAILURE).)*$", message = "Could not decrypt the private key with the provided password, please double check the passwords provided")
055    public String getPrivateKey() {
056        final PrivateKeyData pkd = privateKeyConfig.getPrivateKeyData();
057
058        if (privateKeyConfig.getType() == UNLOCKED) {
059            return privateKeyConfig.getValue();
060        }
061
062        if (this.cachedValue == null || !Objects.equals(this.cachedPassword, this.password)) {
063            if (password != null) {
064                try {
065                    this.cachedValue = KeyEncryptorFactory.create().decryptPrivateKey(pkd, password).encodeToBase64();
066                } catch (final NaclException ex) {
067                    this.cachedValue = "NACL_FAILURE";
068                }
069            }
070        }
071
072        this.cachedPassword = this.password;
073
074        return this.cachedValue;
075    }
076
077    @Override
078    public void withPassword(final String password) {
079        this.password = password;
080    }
081
082    @Override
083    public String getPassword() {
084        return this.password;
085    }
086
087}
