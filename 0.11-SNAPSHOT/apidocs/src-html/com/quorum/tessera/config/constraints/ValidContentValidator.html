001package com.quorum.tessera.config.constraints;
002
003import com.quorum.tessera.io.FilesDelegate;
004import java.nio.file.Files;
005import javax.validation.ConstraintValidator;
006import javax.validation.ConstraintValidatorContext;
007import java.nio.file.Path;
008import java.util.List;
009import java.util.Objects;
010import java.util.stream.Collectors;
011
012public class ValidContentValidator implements ConstraintValidator<ValidContent, Path> {
013
014    private ValidContent config;
015
016    @Override
017    public void initialize(ValidContent constraintAnnotation) {
018        this.config = constraintAnnotation;
019    }
020
021    @Override
022    public boolean isValid(Path path, ConstraintValidatorContext context) {
023
024        if (Objects.isNull(path)) {
025            return true;
026        }
027
028        if (!Files.exists(path)) {
029            return true;
030        }
031
032        List<String> lines = FilesDelegate.create().lines(path)
033                .filter(line -> !Objects.equals("", line))
034                .collect(Collectors.toList());
035        
036        return lines.size() >= config.minLines() && lines.size() <= config.maxLines();
037
038    }
039
040}
