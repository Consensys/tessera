001package com.quorum.tessera.config.constraints;
002
003import javax.validation.ConstraintValidator;
004import javax.validation.ConstraintValidatorContext;
005import java.util.Base64;
006import java.util.Objects;
007
008import static com.quorum.tessera.config.adapters.KeyDataAdapter.NACL_FAILURE_TOKEN;
009
010public class Base64Validator implements ConstraintValidator<ValidBase64, String> {
011
012    @Override
013    public boolean isValid(String value, ConstraintValidatorContext cvc) {
014        if(Objects.isNull(value)) {
015            return true;
016        }
017
018        if(value.startsWith(NACL_FAILURE_TOKEN)) {
019            return true;
020        }
021        
022        try {
023            Base64.getDecoder().decode(value);
024            return true;
025        } catch (IllegalArgumentException ex) {
026            return false;
027        }
028    }
029
030}
