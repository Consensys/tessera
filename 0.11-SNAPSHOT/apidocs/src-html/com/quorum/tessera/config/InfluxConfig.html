001package com.quorum.tessera.config;
002
003import com.quorum.tessera.config.constraints.ValidServerAddress;
004import com.quorum.tessera.config.constraints.ValidSsl;
005
006import javax.validation.Valid;
007import javax.validation.constraints.NotNull;
008import javax.xml.bind.annotation.XmlAccessType;
009import javax.xml.bind.annotation.XmlAccessorType;
010import javax.xml.bind.annotation.XmlElement;
011import java.net.URI;
012import java.util.Objects;
013
014@XmlAccessorType(XmlAccessType.FIELD)
015public class InfluxConfig extends ConfigItem {
016
017    @ValidServerAddress(
018        message = "Server Address is invalid",
019        supportedSchemes = {"http", "https"}
020    )
021    @NotNull
022    @XmlElement
023    private String serverAddress;
024
025    @NotNull
026    @XmlElement(required = true)
027    private Long pushIntervalInSecs;
028
029    @NotNull
030    @XmlElement(required = true)
031    private String dbName;
032
033    @Valid
034    @XmlElement
035    @ValidSsl
036    private SslConfig sslConfig;
037
038    public InfluxConfig() {
039
040    }
041
042    public boolean isSsl() {
043        return Objects.nonNull(sslConfig) && sslConfig.getTls() == SslAuthenticationMode.STRICT;
044    }
045
046    public URI getServerUri() {
047        try {
048            return URI.create(serverAddress);
049        } catch (IllegalArgumentException ex) {
050            throw new ConfigException(ex);
051        }
052    }
053
054    public String getServerAddress() {
055        return serverAddress;
056    }
057
058    public Long getPushIntervalInSecs() {
059        return pushIntervalInSecs;
060    }
061
062    public String getDbName() {
063        return dbName;
064    }
065
066    public SslConfig getSslConfig() {
067        return sslConfig;
068    }
069
070    public void setServerAddress(String serverAddress) {
071        this.serverAddress = serverAddress;
072    }
073
074    public void setPushIntervalInSecs(Long pushIntervalInSecs) {
075        this.pushIntervalInSecs = pushIntervalInSecs;
076    }
077
078    public void setDbName(String dbName) {
079        this.dbName = dbName;
080    }
081
082    public void setSslConfig(SslConfig sslConfig) {
083        this.sslConfig = sslConfig;
084    }
085}
