001package com.quorum.tessera.config;
002
003import com.quorum.tessera.config.constraints.ValidSsl;
004
005import javax.validation.Valid;
006import javax.validation.constraints.NotNull;
007import javax.xml.bind.annotation.XmlAccessType;
008import javax.xml.bind.annotation.XmlAccessorType;
009import javax.xml.bind.annotation.XmlElement;
010import java.nio.file.Path;
011import java.util.ArrayList;
012import java.util.Collections;
013import java.util.List;
014
015@Deprecated
016@XmlAccessorType(XmlAccessType.FIELD)
017public class DeprecatedServerConfig extends ConfigItem {
018
019    @NotNull
020    @XmlElement(required = true)
021    private String hostName;
022
023    @NotNull
024    @XmlElement
025    private Integer port;
026
027    @XmlElement
028    private Integer grpcPort;
029
030    @XmlElement
031    private CommunicationType communicationType;
032
033    @Valid
034    @XmlElement
035    @ValidSsl
036    private SslConfig sslConfig;
037
038    @Valid
039    @XmlElement
040    private InfluxConfig influxConfig;
041
042    @XmlElement
043    private String bindingAddress;
044
045    public DeprecatedServerConfig() {
046    }
047
048    public String getHostName() {
049        return hostName;
050    }
051
052    public void setHostName(String hostName) {
053        this.hostName = hostName;
054    }
055
056    public Integer getPort() {
057        return port;
058    }
059
060    public void setPort(Integer port) {
061        this.port = port;
062    }
063
064    public Integer getGrpcPort() {
065        return grpcPort;
066    }
067
068    public void setGrpcPort(Integer grpcPort) {
069        this.grpcPort = grpcPort;
070    }
071
072    public CommunicationType getCommunicationType() {
073        return communicationType;
074    }
075
076    public void setCommunicationType(CommunicationType communicationType) {
077        this.communicationType = communicationType;
078    }
079
080    public SslConfig getSslConfig() {
081        return sslConfig;
082    }
083
084    public void setSslConfig(SslConfig sslConfig) {
085        this.sslConfig = sslConfig;
086    }
087
088    public InfluxConfig getInfluxConfig() {
089        return influxConfig;
090    }
091
092    public void setInfluxConfig(InfluxConfig influxConfig) {
093        this.influxConfig = influxConfig;
094    }
095
096    public String getBindingAddress() {
097        if (bindingAddress == null) {
098            this.bindingAddress = hostName + ":" + port;
099        }
100        return bindingAddress;
101    }
102
103    public void setBindingAddress(String bindingAddress) {
104        this.bindingAddress = bindingAddress;
105    }
106
107    public static List<ServerConfig> from(DeprecatedServerConfig server, Path unixSocketFile) {
108        if (null == server) {
109            return Collections.emptyList();
110        }
111
112        ServerConfig q2tConfig = new ServerConfig();
113        q2tConfig.setEnabled(true);
114        q2tConfig.setApp(AppType.Q2T);
115        q2tConfig.setCommunicationType(CommunicationType.REST);
116        String uriValue = "unix:"+ String.valueOf(unixSocketFile);
117        q2tConfig.setServerAddress(uriValue);
118        q2tConfig.setInfluxConfig(server.getInfluxConfig());
119
120        ServerConfig p2pConfig = new ServerConfig();
121        p2pConfig.setEnabled(true);
122        p2pConfig.setApp(AppType.P2P);
123
124        if (server.getCommunicationType() == CommunicationType.GRPC) {
125            p2pConfig.setServerAddress(server.getHostName() +":"+ server.getGrpcPort());
126            p2pConfig.setCommunicationType(CommunicationType.GRPC);
127        } else {
128            p2pConfig.setServerAddress(server.getHostName() +":"+ server.getPort());
129            p2pConfig.setCommunicationType(CommunicationType.REST);
130        }
131        p2pConfig.setInfluxConfig(server.getInfluxConfig());
132        p2pConfig.setSslConfig(server.getSslConfig());
133        p2pConfig.setBindingAddress(server.getBindingAddress());
134
135        List<ServerConfig> srvConfigs = new ArrayList<>();
136        srvConfigs.add(q2tConfig);
137        srvConfigs.add(p2pConfig);
138
139        return srvConfigs;
140    }
141
142}
