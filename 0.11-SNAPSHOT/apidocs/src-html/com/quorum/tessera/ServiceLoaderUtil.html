001package com.quorum.tessera;
002
003import java.util.Optional;
004import java.util.ServiceLoader;
005import java.util.stream.Stream;
006import java.util.stream.StreamSupport;
007
008public interface ServiceLoaderUtil {
009
010    static <T> Optional<T> load(Class<T> type) {
011        return ServiceLoaderUtil.loadAll(type).findFirst();
012    }
013
014    static <T> Stream<T> loadAll(Class<T> type) {
015        // TODO: Java 9 defines a native stream method for the service loader, use that instead
016        return StreamSupport.stream(ServiceLoader.load(type).spliterator(), false);
017    }
018}
