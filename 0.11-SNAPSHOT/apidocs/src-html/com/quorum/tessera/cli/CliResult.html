001package com.quorum.tessera.cli;
002
003import com.quorum.tessera.config.Config;
004
005import java.util.Objects;
006import java.util.Optional;
007
008public class CliResult {
009
010    private final Integer status;
011
012    private final boolean suppressStartup;
013
014    private final Config config;
015
016    public CliResult(Integer status, boolean suppressStartup,Config config) {
017        this.status = Objects.requireNonNull(status);
018        this.suppressStartup = suppressStartup;
019        this.config = config;
020    }
021
022    public Integer getStatus() {
023        return status;
024    }
025
026    public boolean isSuppressStartup() {
027        return suppressStartup;
028    }
029
030
031    public Optional<Config> getConfig() {
032        return Optional.ofNullable(config);
033    }
034}
