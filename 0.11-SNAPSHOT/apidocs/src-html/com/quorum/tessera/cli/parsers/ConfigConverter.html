001package com.quorum.tessera.cli.parsers;
002
003import com.quorum.tessera.config.Config;
004import com.quorum.tessera.config.ConfigFactory;
005import picocli.CommandLine;
006
007import java.io.FileNotFoundException;
008import java.io.InputStream;
009import java.nio.file.Files;
010import java.nio.file.Path;
011import java.nio.file.Paths;
012import java.util.Collections;
013
014public class ConfigConverter implements CommandLine.ITypeConverter<Config> {
015
016    @Override
017    public Config convert(final String value) throws Exception {
018        final ConfigFactory configFactory = ConfigFactory.create();
019
020        final Path path = Paths.get(value);
021
022        if (!Files.exists(path)) {
023            throw new FileNotFoundException(String.format("%s not found.", path));
024        }
025
026        try (InputStream in = Files.newInputStream(path)) {
027            return configFactory.create(in, Collections.emptyList());
028        }
029    }
030}
