001package com.quorum.tessera.cli.parsers;
002
003import com.quorum.tessera.config.Config;
004import com.quorum.tessera.config.ConfigFactory;
005import com.quorum.tessera.config.keypairs.ConfigKeyPair;
006import com.quorum.tessera.config.util.ConfigFileStore;
007import com.quorum.tessera.config.util.JaxbUtil;
008import com.quorum.tessera.io.SystemAdapter;
009import org.apache.commons.cli.CommandLine;
010
011import java.io.FileNotFoundException;
012import java.io.IOException;
013import java.io.InputStream;
014import java.io.OutputStream;
015import java.nio.file.Files;
016import java.nio.file.Path;
017import java.nio.file.Paths;
018import java.util.Collections;
019import java.util.List;
020import java.util.Objects;
021
022import static java.nio.file.StandardOpenOption.CREATE_NEW;
023
024public class ConfigurationParser implements Parser<Config> {
025
026    private List<ConfigKeyPair> newlyGeneratedKeys = Collections.emptyList();
027
028    public ConfigurationParser withNewKeys(final List<ConfigKeyPair> newKeys) {
029        this.newlyGeneratedKeys = Objects.requireNonNull(newKeys);
030        return this;
031    }
032
033    @Override
034    public Config parse(final CommandLine commandLine) throws IOException {
035
036        final ConfigFactory configFactory = ConfigFactory.create();
037
038        Config config = null;
039
040        final boolean isGeneratingWithKeyVault = commandLine.hasOption("keygen") && commandLine.hasOption("keygenvaulturl");
041
042        if (commandLine.hasOption("configfile") && !isGeneratingWithKeyVault) {
043            final Path path = Paths.get(commandLine.getOptionValue("configfile"));
044
045            if (!Files.exists(path)) {
046                throw new FileNotFoundException(String.format("%s not found.", path));
047            }
048
049            try (InputStream in = Files.newInputStream(path)) {
050                config = configFactory.create(in, newlyGeneratedKeys);
051            }
052
053            if (!newlyGeneratedKeys.isEmpty()) {
054                //we have generated new keys, so we need to output the new configuration
055                output(commandLine, config);
056            }
057
058            ConfigFileStore.create(path);
059
060        }
061
062        return config;
063    }
064
065    private static void output(CommandLine commandLine, Config config) throws IOException {
066
067        if (commandLine.hasOption("output")) {
068            final Path outputConfigFile = Paths.get(commandLine.getOptionValue("output"));
069
070            try (OutputStream out = Files.newOutputStream(outputConfigFile, CREATE_NEW)) {
071                JaxbUtil.marshal(config, out);
072            }
073        } else {
074            JaxbUtil.marshal(config, SystemAdapter.INSTANCE.out());
075        }
076
077    }
078}
