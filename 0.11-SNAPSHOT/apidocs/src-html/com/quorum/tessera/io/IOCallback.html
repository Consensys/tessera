001package com.quorum.tessera.io;
002
003import org.slf4j.Logger;
004import org.slf4j.LoggerFactory;
005
006import java.io.IOException;
007import java.io.UncheckedIOException;
008
009/**
010 * Callback and template function to uncheck IO exceptions
011 *
012 * @param <T> the function to execute that throws an {@link IOException}
013 */
014@FunctionalInterface
015public interface IOCallback<T> {
016
017    Logger LOGGER = LoggerFactory.getLogger(IOCallback.class);
018
019    T doExecute() throws IOException;
020
021    static <T> T execute(IOCallback<T> callback) {
022        try {
023            return callback.doExecute();
024        } catch (final IOException ex) {
025            LOGGER.debug(null, ex);
026            throw new UncheckedIOException(ex);
027        }
028    }
029
030}
