001package com.quorum.tessera.io;
002
003import com.quorum.tessera.ServiceLoaderUtil;
004import java.io.InputStream;
005import java.nio.file.Files;
006import java.nio.file.LinkOption;
007import java.nio.file.OpenOption;
008import java.nio.file.Path;
009import java.nio.file.attribute.FileAttribute;
010import java.nio.file.attribute.PosixFilePermission;
011import java.util.Set;
012import java.util.stream.Stream;
013
014/**
015 * <p>
016 *  Delegates calls to nio Files functions unchecking IOExceptions
017 *  and providing a means of mocking file system interactions.
018 * </p>
019 * 
020 * @see java.nio.file.Files
021 */
022public interface FilesDelegate {
023
024    default boolean notExists(Path path, LinkOption... options) {
025        return Files.notExists(path, options);
026    }
027
028    default boolean deleteIfExists(Path path) {
029        return IOCallback.execute(() -> Files.deleteIfExists(path));
030    }
031
032    default Path createFile(Path path, FileAttribute... attributes) {
033        return IOCallback.execute(() -> Files.createFile(path, attributes));
034    }
035
036    default InputStream newInputStream(Path path, OpenOption... options) {
037        return IOCallback.execute(() -> Files.newInputStream(path, options));
038    }
039
040    default boolean exists(Path path, LinkOption... options) {
041        return Files.exists(path, options);
042    }
043
044    default byte[] readAllBytes(Path path) {
045        return IOCallback.execute(() -> Files.readAllBytes(path));
046    }
047
048    default Stream<String> lines(Path path) {
049        return IOCallback.execute(() -> Files.lines(path));
050    }
051
052    default Path write(Path path, byte[] bytes, OpenOption... options) {
053        return IOCallback.execute(() -> Files.write(path, bytes, options));
054    }
055
056    default Path setPosixFilePermissions(Path path, Set<PosixFilePermission> perms) {
057        return IOCallback.execute(() -> Files.setPosixFilePermissions(path, perms));
058    }
059
060    static FilesDelegate create() {
061        return ServiceLoaderUtil.load(FilesDelegate.class).orElse(new FilesDelegate() {
062        });
063    }
064
065}
