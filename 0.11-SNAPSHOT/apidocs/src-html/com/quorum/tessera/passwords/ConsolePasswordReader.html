001package com.quorum.tessera.passwords;
002
003import java.io.Console;
004
005public class ConsolePasswordReader implements PasswordReader {
006
007    private final Console console;
008
009    public ConsolePasswordReader(final Console console) {
010        this.console = console;
011    }
012
013    @Override
014    public String readPasswordFromConsole() {
015        final char[] consolePassword = this.console.readPassword();
016        return new String(consolePassword);
017    }
018
019}
