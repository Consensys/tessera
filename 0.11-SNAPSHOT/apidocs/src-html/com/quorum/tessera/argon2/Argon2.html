001package com.quorum.tessera.argon2;
002
003import com.quorum.tessera.ServiceLoaderUtil;
004
005/** Provides hashing functions using the Argon2 class of algorithms. */
006public interface Argon2 {
007
008    /**
009     * Hash the password using the provided options
010     *
011     * @param options the custom options to call Argon2 with
012     * @param password the password to hash
013     * @param salt the salt to apply when hashing
014     * @return the result which contains the output, as well as the input parameters
015     */
016    ArgonResult hash(ArgonOptions options, String password, byte[] salt);
017
018    /**
019     * Hash the password using the given salt and some default options
020     *
021     * @param password the password to hash
022     * @param salt the salt to apply when hashing
023     * @return the result which contains the output, as well as the input parameters
024     */
025    ArgonResult hash(String password, byte[] salt);
026
027    // TODO: move into factory and return the stream itself
028    static Argon2 create() {
029        return ServiceLoaderUtil.loadAll(Argon2.class).findAny().get();
030    }
031}
