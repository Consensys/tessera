001package com.quorum.tessera.encryption;
002
003
004import java.util.Set;
005
006/**
007 * Manages local keys for the running node, include key lookups and key loading/generation
008 */
009public interface KeyManager {
010
011    /**
012     * Fetches the public key that corresponds to the given private key
013     * This means that the returned public key is under control of the current node
014     * (i.e. it is one of our keys)
015     *
016     * @param privateKey the private key for which to find the public key
017     * @return the corresponding public key
018     * @throws RuntimeException if the private key could not be found
019     */
020    PublicKey getPublicKeyForPrivateKey(PrivateKey privateKey);
021
022    /**
023     * Fetches the private key that corresponds to the given public key
024     * This means that the returned private key is under control of the current node
025     * (i.e. it is one of our keys)
026     *
027     * @param publicKey the public key for which to find the private key
028     * @return the corresponding private key
029     * @throws RuntimeException if the public key could not be found
030     */
031    PrivateKey getPrivateKeyForPublicKey(PublicKey publicKey);
032
033    /**
034     * Return a list of all recipients public keys of this node
035     *
036     * @return the set of all public keys
037     */
038    Set<PublicKey> getPublicKeys();
039
040    /**
041     * Returns the default public key of this node if a transaction is
042     * sent that doesn't have a sender.
043     * <p>
044     * Must return the same key each time during a single run,
045     * but no guarantee is made about which key is returned if the
046     * application is restarted.
047     *
048     * @return The public key to use as the default sender
049     */
050    PublicKey defaultPublicKey();
051
052    /**
053     * Returns a set of public keys that all transactions should be sent to in
054     * addition to the declared recipients
055     *
056     * @return the set of keys to forward to
057     */
058    Set<PublicKey> getForwardingKeys();
059
060}
