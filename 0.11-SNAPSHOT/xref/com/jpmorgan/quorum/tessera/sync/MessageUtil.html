1   package com.jpmorgan.quorum.tessera.sync;
2   
3   import com.quorum.tessera.enclave.EncodedPayload;
4   import com.quorum.tessera.enclave.PayloadEncoder;
5   import com.quorum.tessera.encryption.PublicKey;
6   import com.quorum.tessera.partyinfo.PartyInfoParser;
7   import com.quorum.tessera.partyinfo.model.PartyInfo;
8   import java.util.Base64;
9   import java.util.Optional;
10  
11  public class MessageUtil {
12  
13      private static final PartyInfoParser PARTYINFO_PARSER = PartyInfoParser.create();
14  
15      private static final PayloadEncoder PAYLOAD_ENCODER = PayloadEncoder.create();
16  
17      private static final Base64.Decoder BASE64_DECODER = Base64.getDecoder();
18  
19      private static final Base64.Encoder BASE64_ENCODER = Base64.getEncoder();
20  
21      public MessageUtil() {
22          throw new UnsupportedOperationException("This class cannot be instantiated");
23      }
24  
25      public static PublicKey decodePublicKeyFromBase64(String publicKeyData) {
26          return Optional.of(publicKeyData).map(BASE64_DECODER::decode).map(PublicKey::from).get();
27      }
28  
29      public static EncodedPayload decodeTransactionsFromBase64(String transactionData) {
30          return Optional.of(transactionData).map(BASE64_DECODER::decode).map(PAYLOAD_ENCODER::decode).get();
31      }
32  
33      public static PartyInfo decodePartyInfoFromBase64(String partyInfoData) {
34          return Optional.of(partyInfoData).map(BASE64_DECODER::decode).map(PARTYINFO_PARSER::from).get();
35      }
36  
37      public static String encodeToBase64(PartyInfo partyInfo) {
38          return Optional.of(partyInfo).map(PARTYINFO_PARSER::to).map(BASE64_ENCODER::encodeToString).get();
39      }
40  
41      public static String encodeToBase64(EncodedPayload transactions) {
42          return Optional.of(transactions).map(PAYLOAD_ENCODER::encode).map(BASE64_ENCODER::encodeToString).get();
43      }
44  
45      public static String encodeToBase64(PublicKey publicKey) {
46          return publicKey.encodeToBase64();
47      }
48  }
