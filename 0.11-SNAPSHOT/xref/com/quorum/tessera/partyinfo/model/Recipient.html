10  public class Recipient {

17          this.key = key;
18          this.url = url;
19      }
20  
21      public PublicKey getKey() {
22          return key;
23      }
24  
25      public String getUrl() {
26          return url;
27      }
28  
29      @Override
30      public boolean equals(final Object o) {
31          if (!(o instanceof Recipient)) {
32              return false;
33          }
34  
35          final Recipient recipient = (Recipient) o;
36  
37          return Objects.equals(key, recipient.key) &&
38              Objects.equals(url, recipient.url);
39      }
40  
41      @Override
42      public int hashCode() {
43          return Objects.hash(key, url);
44      }
45  }
