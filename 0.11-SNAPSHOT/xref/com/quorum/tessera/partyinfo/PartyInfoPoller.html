1   package com.quorum.tessera.partyinfo;
2   
3   import com.quorum.tessera.partyinfo.model.Party;
4   import com.quorum.tessera.partyinfo.model.PartyInfo;
5   import org.slf4j.Logger;
6   import org.slf4j.LoggerFactory;
7   
8   import java.util.Objects;
9   
10  /**
11   * Polls every so often to all known nodes for any new discoverable nodes. This keeps all nodes up-to date and
12   * discoverable by other nodes
13   */
14  public class PartyInfoPoller implements Runnable {
15  
16      private static final Logger LOGGER = LoggerFactory.getLogger(PartyInfoPoller.class);
17  
18      private final PartyInfoService partyInfoService;
19  
20      private final PartyInfoParser partyInfoParser;
21  
22      private final P2pClient p2pClient;
23  
24      public PartyInfoPoller(final PartyInfoService partyInfoService, final P2pClient p2pClient) {
25          this(partyInfoService, PartyInfoParser.create(), p2pClient);
26      }
27  
28      public PartyInfoPoller(
29              final PartyInfoService partyInfoService, final PartyInfoParser partyInfoParser, final P2pClient p2pClient) {
30          this.partyInfoService = Objects.requireNonNull(partyInfoService);
31          this.partyInfoParser = Objects.requireNonNull(partyInfoParser);
32          this.p2pClient = Objects.requireNonNull(p2pClient);
33      }
34  
35      /**
36       * Iterates over all known parties and contacts them for the current state of their known node discovery list
37       *
38       * <p>For Tessera 0.9 backwards, after contacting the known parties, this poller then updates this nodes list of
39       * data with any new information collected.
40       *
41       * <p>This behaviour is now deprecated since the /partyinfo API call now has been made more strict with node
42       * validation to prevent exploiting the API to attack the Tessera network.
43       *
44       * <p>This call is merely to let its parties know about this node existence, any recipients that want to be added to
45       * this node's PartyInfo will need to make their own partyinfo call and validation
46       */
47      @Override
48      public void run() {
49          final PartyInfo partyInfo = partyInfoService.getPartyInfo();
50          final byte[] encodedPartyInfo = partyInfoParser.to(partyInfo);
51  
52          final String ourUrl = partyInfo.getUrl();
53  
54          partyInfo.getParties().stream()
55                  .map(Party::getUrl)
56                  .filter(url -> !ourUrl.equals(url))
57                  .forEach(url -> pollSingleParty(url, encodedPartyInfo));
58      }
59  
60      /**
61       * Sends a request for node information to a single target
62       *
63       * @param url the target URL to call
64       * @param encodedPartyInfo the encoded current party information
65       */
66      private void pollSingleParty(final String url, final byte[] encodedPartyInfo) {
67          try {
68              p2pClient.sendPartyInfo(url, encodedPartyInfo);
69          } catch (final Exception ex) {
70              LOGGER.warn("Error {} when connecting to {}", ex.getMessage(), url);
71              LOGGER.debug(null, ex);
72          }
73      }
74  }
