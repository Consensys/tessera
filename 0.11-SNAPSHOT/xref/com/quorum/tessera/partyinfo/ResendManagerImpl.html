12  import javax.transaction.Transactional;
13  import java.util.Arrays;
14  import java.util.Optional;
15  
16  public class ResendManagerImpl implements ResendManager {
17  
18      private final EncryptedTransactionDAO encryptedTransactionDAO;
19  
20      private final PayloadEncoder payloadEncoder;
21  
22      private final Enclave enclave;
23  
24      private final MessageHashFactory messageHashFactory = MessageHashFactory.create();
25  
26      public ResendManagerImpl(EncryptedTransactionDAO encryptedTransactionDAO, Enclave enclave) {
27          this(encryptedTransactionDAO, PayloadEncoder.create(), enclave);
28      }
29  
30      public ResendManagerImpl(final EncryptedTransactionDAO dao, final PayloadEncoder encoder, final Enclave enclave) {
31          this.encryptedTransactionDAO = dao;
32          this.payloadEncoder = encoder;
33          this.enclave = enclave;
34      }
35  
36      // TODO: synchronize based on messagehash, so different message don't lock each other
37      @Transactional
38      public synchronized void acceptOwnMessage(final byte[] message) {
39  
40          final EncodedPayload payload = payloadEncoder.decode(message);
41  
42          // check the payload can be decrpyted to ensure it isn't rubbish being sent to us
43          final byte[] newDecrypted = enclave.unencryptTransaction(payload, null);
44  
45          final MessageHash transactionHash =
46                  Optional.of(payload)
47                          .map(EncodedPayload::getCipherText)
48                          .map(messageHashFactory::createFromCipherText)
49                          .get();
50  
51          final PublicKey sender = payload.getSenderKey();
52          if (!enclave.getPublicKeys().contains(sender)) {
53              throw new IllegalArgumentException(
54                      "Message " + transactionHash.toString() + " does not have one the nodes own keys as a sender");
55          }
56  
57          // this is a tx which we created
58          final Optional<EncryptedTransaction> tx = this.encryptedTransactionDAO.retrieveByHash(transactionHash);
59  
60          if (tx.isPresent()) {
61  
62              // we just need to add the recipient
63              final byte[] encodedPayload = tx.get().getEncodedPayload();
64              final EncodedPayload existing = payloadEncoder.decode(encodedPayload);
65  
66              if (!existing.getRecipientKeys().contains(payload.getRecipientKeys().get(0))) {
67                  // lets compare it against another message received before
68                  final byte[] oldDecrypted = enclave.unencryptTransaction(existing, null);
69                  final boolean same =
70                          Arrays.equals(newDecrypted, oldDecrypted)
71                                  && Arrays.equals(payload.getCipherText(), existing.getCipherText());
72  
73                  if (!same) {
74                      throw new IllegalArgumentException("Invalid payload provided");
75                  }
76  
77                  existing.getRecipientKeys().add(payload.getRecipientKeys().get(0));
78                  existing.getRecipientBoxes().add(payload.getRecipientBoxes().get(0));
79  
80                  tx.get().setEncodedPayload(payloadEncoder.encode(existing));
81  
82                  this.encryptedTransactionDAO.save(tx.get());
83              }
84  
85          } else {
86  
87              // we need to recreate this
88              payload.getRecipientKeys().add(sender);
89              byte[] newbox = enclave.createNewRecipientBox(payload, sender);
90              payload.getRecipientBoxes().add(newbox);
91  
92              final byte[] encoded = payloadEncoder.encode(payload);
93  
94              this.encryptedTransactionDAO.save(new EncryptedTransaction(transactionHash, encoded));
95          }
96      }
97  }
