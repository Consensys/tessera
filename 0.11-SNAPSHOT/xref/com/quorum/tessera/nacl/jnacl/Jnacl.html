36      @Override
37      public SharedKey computeSharedKey(final PublicKey publicKey, final PrivateKey privateKey) {
38          final byte[] precomputed = new byte[crypto_secretbox_BEFORENMBYTES];
39  
40          LOGGER.debug("Computing the shared key for public key {} and private key {}", publicKey, privateKey);
41          final int jnaclResult = secretBox.cryptoBoxBeforenm(
42              precomputed, publicKey.getKeyBytes(), privateKey.getKeyBytes()
43          );
44  
45          if(jnaclResult == -1) {
46              LOGGER.warn("Could not compute the shared key for pub {} and priv {}", publicKey, REDACTED);
47              LOGGER.debug("Could not compute the shared key for pub {} and priv {}", publicKey, privateKey);
48              throw new NaclException("JNacl could not compute the shared key");
49          }
50  
51          final SharedKey sharedKey = SharedKey.from(precomputed);
52  
53          LOGGER.debug("Computed shared key {} for pub {} and priv {}", sharedKey, publicKey, privateKey);
54  
55          return sharedKey;
56  
57      }

59      @Override
60      public byte[] seal(final byte[] message, final Nonce nonce, final PublicKey publicKey, final PrivateKey privateKey) {
61  
62          LOGGER.debug("Sealing message using public key {}", publicKey);
63          LOGGER.debug(
64              "Sealing message {} using nonce {}, public key {} and private key {}",
65              Arrays.toString(message), nonce, publicKey, privateKey
66          );
67  
68          try {
69  
70              final NaCl nacl = new NaCl(privateKey.getKeyBytes(), publicKey.getKeyBytes());
71  
72              final byte[] cipherText = nacl.encrypt(message, nonce.getNonceBytes());
73  
74              LOGGER.debug("Created sealed payload for public key {}", publicKey);
75              LOGGER.debug(
76                  "Created sealed payload {} using nonce {}, public key {} and private key {}",
77                  Arrays.toString(cipherText), nonce, publicKey, privateKey
78              );
79  
80              return extract(cipherText, crypto_secretbox_BOXZEROBYTES);
81  
82          } catch (final Exception ex) {
83              throw new NaclException(ex.getMessage());
84          }
85  
86      }

88      @Override
89      public byte[] open(final byte[] cipherText, final Nonce nonce, final PublicKey publicKey, final PrivateKey privateKey) {
90          LOGGER.debug("Opening message using public key {}", publicKey);
91          LOGGER.debug(
92              "Opening message {} using nonce {}, public key {} and private key {}",
93              Arrays.toString(cipherText), nonce, publicKey, privateKey
94          );
95  
96          try {
97  
98              final byte[] paddedInput = pad(cipherText, crypto_secretbox_BOXZEROBYTES);
99  
100             final NaCl nacl = new NaCl(privateKey.getKeyBytes(), publicKey.getKeyBytes());
101 
102             final byte[] plaintext = nacl.decrypt(paddedInput, nonce.getNonceBytes());
103 
104             LOGGER.debug("Created sealed payload for public key {}", publicKey);
105             LOGGER.debug(
106                 "Created sealed payload {} using nonce {}, public key {} and private key {}",
107                 Arrays.toString(cipherText), nonce, publicKey, privateKey
108             );
109 
110             return plaintext;
111         } catch (final Exception ex) {
112             throw new NaclException(ex.getMessage());
113         }
114     }

116     @Override
117     public byte[] sealAfterPrecomputation(final byte[] message, final Nonce nonce, final SharedKey sharedKey) {
118 
119         final byte[] paddedMessage = new byte[message.length + crypto_secretbox_ZEROBYTES];
120         final byte[] output = new byte[message.length + crypto_secretbox_ZEROBYTES];
121 
122         LOGGER.debug("Sealing message using public key {}", sharedKey);
123         LOGGER.debug(
124             "Sealing message {} using nonce {} and shared key {}",
125             Arrays.toString(message), nonce, sharedKey
126         );
127 
128         System.arraycopy(message, 0, paddedMessage, crypto_secretbox_ZEROBYTES, message.length);
129         final int jnaclResult = secretBox.cryptoBoxAfternm(
130             output, paddedMessage, paddedMessage.length, nonce.getNonceBytes(), sharedKey.getKeyBytes()
131         );
132 
133         if(jnaclResult == -1) {
134             LOGGER.warn("Could not create sealed payload using shared key {}", sharedKey);
135             LOGGER.debug("Could not create sealed payload using shared key {}", sharedKey);
136             throw new NaclException("jnacl could not seal the payload using the shared key");
137         }
138 
139         LOGGER.debug("Created sealed payload for shared key {}", sharedKey);
140         LOGGER.debug(
141             "Created sealed payload {} using nonce {} and shared key {}",
142             Arrays.toString(output), nonce, sharedKey
143         );
144 
145         return extract(output, crypto_secretbox_BOXZEROBYTES);
146     }

148     @Override
149     public byte[] openAfterPrecomputation(final byte[] cipherText, final Nonce nonce, final SharedKey sharedKey) {
150         LOGGER.debug("Opening message using shared key {}", sharedKey);
151         LOGGER.debug(
152             "Opening message {} using nonce {} and shared key {}",
153             Arrays.toString(cipherText), nonce, sharedKey
154         );
155 
156         final byte[] paddedInput = pad(cipherText, crypto_secretbox_BOXZEROBYTES);
157         final byte[] paddedOutput = new byte[paddedInput.length];
158 
159         final int jnaclResult = secretBox.cryptoBoxOpenAfternm(
160             paddedOutput, paddedInput, paddedInput.length, nonce.getNonceBytes(), sharedKey.getKeyBytes()
161         );
162 
163         if(jnaclResult == -1) {
164             LOGGER.warn("Could not open sealed payload using shared key {}", sharedKey);
165             LOGGER.debug("Could not open sealed payload using shared key {}", sharedKey);
166             throw new NaclException("jnacl could not open the payload using the shared key");
167         }
168 
169         LOGGER.debug("Opened sealed payload for shared key {}", sharedKey);
170         LOGGER.debug(
171             "Opened payload {} using nonce {}, public key {} and private key {} to get result {}",
172             Arrays.toString(cipherText), nonce, sharedKey, REDACTED, Arrays.toString(paddedOutput)
173         );
174 
175         return extract(paddedOutput, crypto_secretbox_ZEROBYTES);
176     }

191     @Override
192     public KeyPair generateNewKeys() {
193         final byte[] publicKey = new byte[crypto_secretbox_PUBLICKEYBYTES];
194         final byte[] privateKey = new byte[crypto_secretbox_SECRETKEYBYTES];
195 
196         LOGGER.info("Generating new keypair...");
197 
198         final int jnaclResult = secretBox.cryptoBoxKeypair(publicKey, privateKey);
199 
200         if(jnaclResult == -1) {
201             LOGGER.warn("Unable to generate a new keypair!");
202             throw new NaclException("jnacl could not generate a new public/private keypair");
203         }
204 
205         final PublicKey pubKey = PublicKey.from(publicKey);
206         final PrivateKey privKey = PrivateKey.from(privateKey);
207 
208         LOGGER.info("Generated public key {} and private key {}", pubKey, REDACTED);
209         LOGGER.debug("Generated public key {} and private key {}", pubKey, privKey);
210 
211         return new KeyPair(pubKey, privKey);

230     /**
231      * Left-pads a given message with padSize amount of zeros
232      *
233      * @param input the message to be padded
234      * @param padSize the amount of left-padding to apply
235      * @return the padded message
236      */
237     private byte[] pad(final byte[] input, final int padSize) {
238         final byte[] paddedMessage = new byte[padSize + input.length];
239         System.arraycopy(input, 0, paddedMessage, padSize, input.length);
240 
241         return paddedMessage;
242     }
243 
244     /**
245      * Removes left-padding from a given message to tune of padSize
246      *
247      * @param input The message from which to remove left-padding
248      * @param padSize The amount of left-padding to remove
249      * @return The trimmed message
250      */
251     private byte[] extract(final byte[] input, final int padSize) {
252         final byte[] extractedMessage = new byte[input.length - padSize];
253         System.arraycopy(input, padSize, extractedMessage, 0, extractedMessage.length);
254 
255         return extractedMessage;
256     }
