5   import java.util.Optional;

7   public interface JdbcConfigFactory {
8   
9       static JdbcConfig fromLegacyStorageString(String storage) {
10          
11          Optional.ofNullable(storage).orElseThrow(IllegalArgumentException::new);
12          
13          if(storage.startsWith("jdbc")) {
14              return new JdbcConfig(null, null, storage);
15          }
16  
17          if(storage.startsWith("sqlite")) {
18              return new JdbcConfig(null, null, String.format("jdbc:%s", storage));
19          }
20          
21          if(storage.startsWith("memory")) {
22              return new JdbcConfig(null, null, "jdbc:h2:mem:tessera");
23          }
24  
25          throw new UnsupportedOperationException(String.format("%s is not a supported storage option.", storage));
26      }
27  
28  }
