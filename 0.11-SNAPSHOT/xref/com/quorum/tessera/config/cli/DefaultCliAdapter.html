50      public DefaultCliAdapter(final KeyPasswordResolver keyPasswordResolver) {
51          this.keyPasswordResolver = Objects.requireNonNull(keyPasswordResolver);
52      }

54      @Override

56          return this.execute(allParameters);

59      @Override
60      public CliType getType() {

64      @Override
65      public CliResult execute(String... args) throws Exception {
66  
67          Options options = this.buildBaseOptions();
68  
69          Map<String, Class> overrideOptions = OverrideUtil.buildConfigOptions();
70  
71          overrideOptions.forEach((optionName, optionType) -> {
72  
73              final boolean isCollection = optionType.isArray();
74  
75              Option.Builder optionBuilder = Option.builder()
76                  .longOpt(optionName)
77                  .desc(String.format("Override option for %s , type: %s", optionName, optionType.getSimpleName()));
78  
79              if (isCollection) {
80                  optionBuilder.hasArgs().argName(optionType.getSimpleName().toUpperCase() + "...");
81              } else {
82                  optionBuilder.hasArg().argName(optionType.getSimpleName().toUpperCase());
83              }
84  
85              options.addOption(optionBuilder.build());
86  
87          });
88  
89          final List<String> argsList = Arrays.asList(args);
90          if (argsList.contains("help") || argsList.isEmpty()) {
91              HelpFormatter formatter = new HelpFormatter();
92              PrintWriter pw = new PrintWriter(sys().out());
93              formatter.printHelp(pw,
94                  200, "tessera -configfile <PATH> [-keygen <PATH>] [-pidfile <PATH>]",
95                  null, options, formatter.getLeftPadding(),
96                  formatter.getDescPadding(), null, false);
97              pw.flush();
98              return new CliResult(0, true, null);
99          }
100 
101         try {
102 
103             final CommandLine line = new DefaultParser().parse(options, args);
104 
105             final Config config = parseConfig(line);
106 
107             if (Objects.nonNull(config)) {
108 
109                 overrideOptions.forEach((optionName, value) -> {
110                     if (line.hasOption(optionName)) {
111                         String[] values = line.getOptionValues(optionName);
112                         LOGGER.debug("Setting : {} with value(s) {}", optionName, values);
113                         OverrideUtil.setValue(config, optionName, values);
114                         LOGGER.debug("Set : {} with value(s) {}", optionName, values);
115                     }
116                 });
117 
118                 keyPasswordResolver.resolveKeyPasswords(config);
119 
120                 final Set<ConstraintViolation<Config>> violations = validator.validate(config);
121                 if (!violations.isEmpty()) {
122                     throw new ConstraintViolationException(violations);
123                 }
124             }
125 
126             new PidFileParser().parse(line);
127 
128             boolean suppressStartup = line.hasOption("keygen") && Objects.isNull(config);
129 
130             return new CliResult(0, suppressStartup, config);
131 
132         } catch (ParseException exp) {
133             throw new CliException(exp.getMessage());
134         }
135 
136     }
137 
138     private Config parseConfig(CommandLine commandLine) throws IOException {
139 
140         if(commandLine.hasOption("updatepassword")) {
141             new KeyUpdateParser(
142                 KeyEncryptorFactory.create(),
143                 PasswordReaderFactory.create()
144             ).parse(commandLine);
145 
146             //return early so other options don't get processed
147             return null;
148         }
149 
150         final List<ConfigKeyPair> newKeys = new KeyGenerationParser().parse(commandLine);
151 
152         final Config config = new ConfigurationParser().withNewKeys(newKeys).parse(commandLine);
153 
154         if (!commandLine.hasOption("configfile") && !commandLine.hasOption("keygen") && !commandLine.hasOption("updatepassword")) {
155             throw new CliException("One or more: -configfile or -keygen or -updatepassword options are required.");
156         }
157 
158         return config;
159     }

161     private Options buildBaseOptions() {
162

164

171                 .argName("PATH")

177                 .hasArg(false)
178                 .build());
179

182                 .desc("Comma-separated list of paths to save generated key files. Can also be used with keyvault. Number of args equals number of key-pairs generated.")
183                 .hasArgs()
184                 .optionalArg(false)
185                 .argName("PATH")
186                 .build());
187 
188         options.addOption(
189             Option.builder("keygenconfig")
190                 .desc("Path to private key config for generation of missing key files")
191                 .hasArg(true)
192                 .optionalArg(false)
193                 .argName("PATH")
194                 .build());

196         options.addOption(
197             Option.builder("output")
198                 .desc("Generate updated config file with generated keys")
199                 .hasArg(true)

201                 .build());
