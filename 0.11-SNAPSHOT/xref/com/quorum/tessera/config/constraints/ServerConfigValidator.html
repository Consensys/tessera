1   package com.quorum.tessera.config.constraints;
2   
3   import com.quorum.tessera.config.ServerConfig;
4   import org.slf4j.Logger;
5   import org.slf4j.LoggerFactory;
6   
7   import javax.validation.ConstraintValidator;
8   import javax.validation.ConstraintValidatorContext;
9   
10  public class ServerConfigValidator implements ConstraintValidator<ValidServerConfig, ServerConfig> {
11  
12      private static final Logger LOGGER = LoggerFactory.getLogger(ServerConfigValidator.class);
13  
14      @Override
15      public boolean isValid(ServerConfig serverConfig, ConstraintValidatorContext constraintContext) {
16  
17          if(serverConfig == null) {
18              return true;
19          }
20          
21          if (!serverConfig.getApp().getAllowedCommunicationTypes().contains(serverConfig.getCommunicationType())) {
22              LOGGER.debug("Invalid communicationType '" + serverConfig.getCommunicationType() +
23                  "' specified for serverConfig with app " + serverConfig.getApp());
24              constraintContext.disableDefaultConstraintViolation();
25              constraintContext.buildConstraintViolationWithTemplate("Invalid communicationType '" +
26                  serverConfig.getCommunicationType() + "' specified for serverConfig with app " + serverConfig.getApp())
27                  .addConstraintViolation();
28              return false;
29          }
30  
31          return true;
32      }
33  }
