3   import com.quorum.tessera.config.ArgonOptions;
4   import com.quorum.tessera.config.PrivateKeyData;
5   import com.quorum.tessera.encryption.PrivateKey;

7   /**
8    * Provides key encryption and decryption for private keys
9    * into the representation expected by a private key file
10   */
11  public interface KeyEncryptor {
12  
13      int SALTLENGTH = 16;
14  
15      /**
16       * Encrypts a private key using the given password and returns a JSON object that can
17       * be interpreted when decrypting
18       *
19       * @param privateKey the key to encrypt
20       * @param password   the password to encrypt the key with
21       * @return the configuration that can be used to decrypt the private key
22       */
23      PrivateKeyData encryptPrivateKey(PrivateKey privateKey, String password, ArgonOptions argonOptions);
24  
25      /**
26       * Decrypts a private key using the password and information provided by the given
27       * JSON object. What information the object contains is up to the implementor.
28       *
29       * @param privateKeyConfig the configuration used to decrypt the private key
30       * @param password         the password that should be used to decrypt the private key
31       * @return the decrypted private key
32       */
33      PrivateKey decryptPrivateKey(PrivateKeyData privateKeyConfig, String password);
34  
35  }
