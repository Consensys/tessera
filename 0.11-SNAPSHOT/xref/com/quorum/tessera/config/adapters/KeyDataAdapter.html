1   package com.quorum.tessera.config.adapters;
2   
3   import com.quorum.tessera.config.KeyData;
4   import com.quorum.tessera.config.keypairs.*;
5   
6   import javax.xml.bind.annotation.adapters.XmlAdapter;
7   import java.util.Objects;
8   
9   public class KeyDataAdapter extends XmlAdapter<KeyData, ConfigKeyPair> {
10  
11      public static final String NACL_FAILURE_TOKEN = "NACL_FAILURE";
12  
13      @Override
14      public ConfigKeyPair unmarshal(final KeyData keyData) {
15  
16          //case 1, the keys are provided inline
17          if (Objects.nonNull(keyData.getPrivateKey()) && Objects.nonNull(keyData.getPublicKey())) {
18              return new DirectKeyPair(keyData.getPublicKey(), keyData.getPrivateKey());
19          }
20  
21          //case 2, the config is provided inline
22          if (keyData.getPublicKey() != null && keyData.getConfig() != null) {
23              return new InlineKeypair(keyData.getPublicKey(), keyData.getConfig());
24          }
25  
26          //case 3, the Azure Key Vault data is provided
27          if(keyData.getAzureVaultPublicKeyId() != null && keyData.getAzureVaultPrivateKeyId() != null) {
28              return new AzureVaultKeyPair(keyData.getAzureVaultPublicKeyId(), keyData.getAzureVaultPrivateKeyId(), keyData.getAzureVaultPublicKeyVersion(), keyData.getAzureVaultPrivateKeyVersion());
29          }
30  
31          //case 4, the Hashicorp Vault data is provided
32          if(keyData.getHashicorpVaultPublicKeyId() != null && keyData.getHashicorpVaultPrivateKeyId() != null
33              && keyData.getHashicorpVaultSecretEngineName() != null && keyData.getHashicorpVaultSecretName() != null) {
34              return new HashicorpVaultKeyPair(keyData.getHashicorpVaultPublicKeyId(), keyData.getHashicorpVaultPrivateKeyId(), keyData.getHashicorpVaultSecretEngineName(), keyData.getHashicorpVaultSecretName(), keyData.getHashicorpVaultSecretVersion());
35          }
36  
37          //case 5, the keys are provided inside a file
38          if(keyData.getPublicKeyPath() != null && keyData.getPrivateKeyPath() != null) {
39              return new FilesystemKeyPair(keyData.getPublicKeyPath(), keyData.getPrivateKeyPath());
40          }
41  
42          //case 6, the key config specified is invalid
43          return new UnsupportedKeyPair(
44              keyData.getConfig(),
45              keyData.getPrivateKey(),
46              keyData.getPublicKey(),
47              keyData.getPrivateKeyPath(),
48              keyData.getPublicKeyPath(),
49              keyData.getAzureVaultPublicKeyId(),
50              keyData.getAzureVaultPrivateKeyId(),
51              keyData.getAzureVaultPublicKeyVersion(),
52              keyData.getAzureVaultPrivateKeyVersion(),
53              keyData.getHashicorpVaultPublicKeyId(),
54              keyData.getHashicorpVaultPrivateKeyId(),
55              keyData.getHashicorpVaultSecretEngineName(),
56              keyData.getHashicorpVaultSecretName(),
57              keyData.getHashicorpVaultSecretVersion()
58          );
59      }
60  
61      @Override
62      public KeyData marshal(final ConfigKeyPair keyPair) {
63  
64          KeyData keyData = new KeyData();
65  
66          if(keyPair instanceof DirectKeyPair) {
67              DirectKeyPair kp = (DirectKeyPair) keyPair;
68  
69              keyData.setPublicKey(kp.getPublicKey());
70              keyData.setPrivateKey(kp.getPrivateKey());
71              return keyData;
72          }
73  
74          if(keyPair instanceof InlineKeypair) {
75              InlineKeypair kp = (InlineKeypair) keyPair;
76  
77              keyData.setPublicKey(kp.getPublicKey());
78              keyData.setConfig(kp.getPrivateKeyConfig());
79              return keyData;
80          }
81  
82          if(keyPair instanceof AzureVaultKeyPair) {
83              AzureVaultKeyPair kp = (AzureVaultKeyPair) keyPair;
84  
85              keyData.setAzureVaultPublicKeyId(kp.getPublicKeyId());
86              keyData.setAzureVaultPrivateKeyId(kp.getPrivateKeyId());
87              keyData.setAzureVaultPublicKeyVersion(kp.getPublicKeyVersion());
88              keyData.setAzureVaultPrivateKeyVersion(kp.getPrivateKeyVersion());
89              return keyData;
90          }
91  
92          if(keyPair instanceof HashicorpVaultKeyPair) {
93              HashicorpVaultKeyPair kp = (HashicorpVaultKeyPair) keyPair;
94  
95              keyData.setHashicorpVaultPublicKeyId(kp.getPublicKeyId());
96              keyData.setHashicorpVaultPrivateKeyId(kp.getPrivateKeyId());
97              keyData.setHashicorpVaultSecretEngineName(kp.getSecretEngineName());
98              keyData.setHashicorpVaultSecretName(kp.getSecretName());
99              return keyData;
100         }
101 
102         if(keyPair instanceof FilesystemKeyPair) {
103             FilesystemKeyPair kp = (FilesystemKeyPair) keyPair;
104 
105             keyData.setPublicKeyPath(kp.getPublicKeyPath());
106             keyData.setPrivateKeyPath(kp.getPrivateKeyPath());
107             return keyData;
108         }
109 
110         if(keyPair instanceof UnsupportedKeyPair) {
111             UnsupportedKeyPair kp = (UnsupportedKeyPair) keyPair;
112             return new KeyData(
113                 kp.getConfig(),
114                 kp.getPrivateKey(),
115                 kp.getPublicKey(),
116                 kp.getPrivateKeyPath(),
117                 kp.getPublicKeyPath(),
118                 kp.getAzureVaultPrivateKeyId(),
119                 kp.getAzureVaultPublicKeyId(),
120                 kp.getAzureVaultPublicKeyVersion(),
121                 kp.getAzureVaultPrivateKeyVersion(),
122                 kp.getHashicorpVaultPrivateKeyId(),
123                 kp.getHashicorpVaultPublicKeyId(),
124                 kp.getHashicorpVaultSecretEngineName(),
125                 kp.getHashicorpVaultSecretName(),
126                 kp.getHashicorpVaultSecretVersion());
127         }
128 
129         throw new UnsupportedOperationException("The keypair type " + keyPair.getClass() + " is not allowed");
130     }
131 }
