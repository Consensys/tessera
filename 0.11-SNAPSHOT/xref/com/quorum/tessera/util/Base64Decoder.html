1   package com.quorum.tessera.util;
2   
3   import com.quorum.tessera.util.exception.DecodingException;
4   
5   import java.util.Base64;
6   
7   /**
8    * An delegate interface for {@link Base64} to allow mocking in tests
9    */
10  public interface Base64Decoder {
11  
12      /**
13       * Decodes a Base64 encoded string to its constituent bytes
14       *
15       * @param data the input string that is Base64 encoded
16       * @return the decoded input
17       * @throws DecodingException if the input is not valid Base64
18       */
19      default byte[] decode(final String data) {
20          try {
21              return Base64.getDecoder().decode(data);
22          } catch (final IllegalArgumentException ex) {
23              throw new DecodingException(ex);
24          }
25      }
26  
27      /**
28       * Encodes the given input to a Base64 encoded string
29       * Does not trim the end if the input was padded
30       *
31       * @param data the input data to encode to Base64
32       * @return the encoded form of the input
33       */
34      default String encodeToString(final byte[] data) {
35          return Base64.getEncoder().encodeToString(data);
36      }
37  
38      static Base64Decoder create() {
39          return new Base64Decoder() {
40          };
41      }
42  
43  }
