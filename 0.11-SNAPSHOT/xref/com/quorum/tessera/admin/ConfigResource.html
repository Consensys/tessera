23  import static java.util.Collections.emptySet;
24  import static java.util.Collections.singleton;

26  @Path("/config")
27  @Consumes(MediaType.APPLICATION_JSON)
28  @Produces(MediaType.APPLICATION_JSON)
29  public class ConfigResource {
30  
31      private final ConfigService configService;
32  
33      private final PartyInfoService partyInfoService;
34  
35      public ConfigResource() {
36  
37          ServiceFactory serviceFactory = ServiceFactory.create();
38  
39          this.configService = serviceFactory.configService();
40  
41          this.partyInfoService = serviceFactory.partyInfoService();
42      }
43  
44      public ConfigResource(final ConfigService configService, final PartyInfoService partyInfoService) {
45          this.configService = Objects.requireNonNull(configService);
46          this.partyInfoService = Objects.requireNonNull(partyInfoService);
47      }
48  
49      @PUT
50      @Path("/peers")
51      public Response addPeer(@Valid final Peer peer) {
52  
53          final boolean existing = configService.getPeers().contains(peer);
54  
55          if (!existing) {
56              this.configService.addPeer(peer.getUrl());
57  
58              this.partyInfoService.updatePartyInfo(
59                      new PartyInfo(peer.getUrl(), emptySet(), singleton(new Party(peer.getUrl()))));
60          }
61  
62          final int index = this.configService.getPeers().indexOf(peer);
63  
64          final URI uri = UriBuilder.fromPath("config").path("peers").path(String.valueOf(index)).build();
65  
66          if (!existing) {
67              return Response.created(uri).build();
68          } else {
69              return Response.ok().location(uri).build();
70          }
71      }
72  
73      @GET
74      @Path("/peers/{index}")
75      public Response getPeer(@PathParam("index") final Integer index) {
76  
77          final List<Peer> peers = this.configService.getPeers();
78  
79          if (peers.size() <= index) {
80              throw new NotFoundException("No peer found at index " + index);
81          }
82  
83          return Response.ok(peers.get(index)).build();
84      }
85  
86      @GET
87      @Path("/peers")
88      public Response getPeers() {
89          final List<Peer> peers = this.configService.getPeers();
90  
91          return Response.ok(new GenericEntity<List<Peer>>(peers) {}).build();
92      }
93  
94      @GET
95      @Path("/keypairs/{publicKey}")
96      public Response getKeyPair(@PathParam("publicKey") String base64PublicKey) {
97  
98          Base64.Decoder base64Decoder = Base64.getDecoder();
99  
100         PublicKey publicKey = PublicKey.from(base64Decoder.decode(base64PublicKey));
101 
102         Set<PublicKey> publicKeys = configService.getPublicKeys();
103 
104         if (!publicKeys.contains(publicKey)) {
105             throw new NotFoundException("No key pair found with public key " + base64PublicKey);
106         }
107 
108         PublicKeyResponse responseData = new PublicKeyResponse(base64PublicKey);
109 
110         return Response.ok(responseData).build();
111     }
112 
113     @GET
114     @Path("/keypairs")
115     public Response getKeyPairs() {
116         Set<PublicKey> publicKeys = configService.getPublicKeys();
117 
118         List<PublicKeyResponse> responseData =
119                 publicKeys.stream()
120                         .map(PublicKey::encodeToBase64)
121                         .map(PublicKeyResponse::new)
122                         .collect(Collectors.toList());
123 
124         return Response.ok(new GenericEntity<List<PublicKeyResponse>>(responseData) {}).build();
125     }
126 }
