3   import com.quorum.tessera.io.SystemAdapter;
4   
5   import java.util.Objects;
6   
7   /**
8    * Allows a password to be read from the System console if it is available
9    * otherwise reads from the provided input stream
10   */
11  public interface PasswordReader {
12  
13      /**
14       * Read a password from either system console or the given stream
15       * @return the read password, which may be empty if no password is given
16       */
17      String readPasswordFromConsole();
18  
19      /**
20       * Requests user input for a password until two matching consecutive entries are made
21       *
22       * @return The password that the user has input
23       */
24      default String requestUserPassword() {
25  
26          for(;;) {
27  
28              sys().out().println("Enter a password if you want to lock the private key or leave blank");
29              final String password = this.readPasswordFromConsole();
30  
31              sys().out().println("Please re-enter the password (or lack of) to confirm");
32              final String passwordCheck = this.readPasswordFromConsole();
33  
34              if(Objects.equals(password, passwordCheck)) {
35                  return password;
36              } else {
37                  sys().out().println("Passwords did not match, try again...");
38              }
39  
40          }
41  
42      }
43      
44      static SystemAdapter sys() {
45          return SystemAdapter.INSTANCE;
46      }
47  
48  }
