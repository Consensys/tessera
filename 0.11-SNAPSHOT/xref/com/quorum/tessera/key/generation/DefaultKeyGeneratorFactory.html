3   import com.quorum.tessera.config.*;
4   import com.quorum.tessera.config.keys.KeyEncryptorFactory;
5   import com.quorum.tessera.config.util.EnvironmentVariableProvider;
6   import com.quorum.tessera.config.vault.data.AzureGetSecretData;
7   import com.quorum.tessera.config.vault.data.AzureSetSecretData;
8   import com.quorum.tessera.config.vault.data.HashicorpGetSecretData;
9   import com.quorum.tessera.config.vault.data.HashicorpSetSecretData;
10  import com.quorum.tessera.key.vault.KeyVaultService;
11  import com.quorum.tessera.key.vault.KeyVaultServiceFactory;
12  import com.quorum.tessera.nacl.NaclFacadeFactory;
13  import com.quorum.tessera.passwords.PasswordReaderFactory;
14  
15  public class DefaultKeyGeneratorFactory implements KeyGeneratorFactory {
16  
17      @Override
18      public KeyGenerator create(KeyVaultConfig keyVaultConfig) {
19  
20          if (keyVaultConfig != null) {
21              final KeyVaultServiceFactory keyVaultServiceFactory = KeyVaultServiceFactory.getInstance(keyVaultConfig.getKeyVaultType());
22  
23              final Config config = new Config();
24              final KeyConfiguration keyConfiguration = new KeyConfiguration();
25  
26              if (keyVaultConfig.getKeyVaultType().equals(KeyVaultType.AZURE)) {
27                  keyConfiguration.setAzureKeyVaultConfig((AzureKeyVaultConfig) keyVaultConfig);
28  
29                  config.setKeys(keyConfiguration);
30  
31                  final KeyVaultService<AzureSetSecretData, AzureGetSecretData> keyVaultService =
32                          keyVaultServiceFactory.create(config, new EnvironmentVariableProvider());
33  
34                  return new AzureVaultKeyGenerator(NaclFacadeFactory.newFactory().create(), keyVaultService);
35  
36              } else {
37                  keyConfiguration.setHashicorpKeyVaultConfig((HashicorpKeyVaultConfig) keyVaultConfig);
38  
39                  config.setKeys(keyConfiguration);
40  
41                  final KeyVaultService<HashicorpSetSecretData, HashicorpGetSecretData> keyVaultService =
42                          keyVaultServiceFactory.create(config, new EnvironmentVariableProvider());
43  
44                  return new HashicorpVaultKeyGenerator(NaclFacadeFactory.newFactory().create(), keyVaultService);
45              }
46          }
47  
48          return new FileKeyGenerator(
49              NaclFacadeFactory.newFactory().create(), KeyEncryptorFactory.create(), PasswordReaderFactory.create()
50          );
51      }
52  }
