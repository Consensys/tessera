11  import org.slf4j.LoggerFactory;
12  
13  import java.util.HashMap;
14  import java.util.Map;
15  import java.util.Objects;
16  
17  public class HashicorpVaultKeyGenerator implements KeyGenerator {
18  
19      private static final Logger LOGGER = LoggerFactory.getLogger(HashicorpVaultKeyGenerator.class);
20  
21      private final NaclFacade nacl;
22  
23      private final KeyVaultService<HashicorpSetSecretData, HashicorpGetSecretData> keyVaultService;
24  
25      public HashicorpVaultKeyGenerator(
26              final NaclFacade nacl, KeyVaultService<HashicorpSetSecretData, HashicorpGetSecretData> keyVaultService) {
27          this.nacl = nacl;
28          this.keyVaultService = keyVaultService;
29      }
30  
31      @Override
32      public HashicorpVaultKeyPair generate(String filename, ArgonOptions encryptionOptions, KeyVaultOptions keyVaultOptions) {
33          Objects.requireNonNull(filename);
34          Objects.requireNonNull(keyVaultOptions, "-keygenvaultsecretengine must be provided if using the Hashicorp vault type");
35          Objects.requireNonNull(keyVaultOptions.getSecretEngineName(), "-keygenvaultsecretengine must be provided if using the Hashicorp vault type");
36  
37          final KeyPair keys = this.nacl.generateNewKeys();
38  
39          String pubId = "publicKey";
40          String privId = "privateKey";
41          Map<String, Object> keyPairData = new HashMap<>();
42          keyPairData.put(pubId, keys.getPublicKey().encodeToBase64());
43          keyPairData.put(privId, keys.getPrivateKey().encodeToBase64());
44  
45          keyVaultService.setSecret(new HashicorpSetSecretData(keyVaultOptions.getSecretEngineName(), filename, keyPairData));
46          LOGGER.debug("Key {} saved to vault secret engine {} with name {} and id {}", keyPairData.get(pubId), keyVaultOptions.getSecretEngineName(), filename, pubId);
47          LOGGER.info("Key saved to vault secret engine {} with name {} and id {}", keyVaultOptions.getSecretEngineName(), filename, pubId);
48          LOGGER.debug("Key {} saved to vault secret engine {} with name {} and id {}", keyPairData.get(privId), keyVaultOptions.getSecretEngineName(), filename, privId);
49          LOGGER.info("Key saved to vault secret engine {} with name {} and id {}", keyVaultOptions.getSecretEngineName(), filename, privId);
50  
51          return new HashicorpVaultKeyPair(pubId, privId, keyVaultOptions.getSecretEngineName(), filename, null);
52      }
53  }
