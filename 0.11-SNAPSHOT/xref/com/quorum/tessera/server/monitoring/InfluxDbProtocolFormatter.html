1   package com.quorum.tessera.server.monitoring;

3   import com.quorum.tessera.config.AppType;
4   
5   import java.net.URI;
6   import java.util.List;
7   
8   public class InfluxDbProtocolFormatter {
9   
10      public String format(List<MBeanMetric> metrics, URI uri, AppType appType) {
11          StringBuilder formattedMetrics = new StringBuilder();
12  
13          for(MBeanMetric metric : metrics) {
14              MBeanResourceMetric resourceMetric = (MBeanResourceMetric) metric;
15  
16              formattedMetrics.append("tessera_")
17                              .append(appType)
18                              .append("_")
19                              .append(sanitize(resourceMetric.getResourceMethod()))
20                              .append(",")
21                              .append("instance=")
22                              .append(uri)
23                              .append(" ")
24                              .append(sanitize(resourceMetric.getName()))
25                              .append("=")
26                              .append(resourceMetric.getValue())
27                              .append("\n");
28          }
29  
30          return formattedMetrics.toString().trim();
31      }
32  
33      private String sanitize(String input) {
34          return input.replaceAll("(#.*)|(_total)|\\(\\)|\\)|\\[\\]|\\]|;", "")
35                      .replaceAll("->|\\(|\\[", "_");
36      }
37  }
