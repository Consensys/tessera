1   package com.quorum.tessera.server.monitoring;
2   
3   import com.quorum.tessera.config.AppType;
4   
5   import javax.management.MBeanServer;
6   import javax.ws.rs.GET;
7   import javax.ws.rs.Path;
8   import javax.ws.rs.Produces;
9   import javax.ws.rs.core.Response;
10  import java.lang.management.ManagementFactory;
11  import java.util.List;
12  
13  import static javax.ws.rs.core.MediaType.TEXT_PLAIN;
14  
15  @Path("/metrics")
16  public class MetricsResource {
17  
18      private final MBeanServer mbs;
19  
20      public MetricsResource() {
21          mbs = ManagementFactory.getPlatformMBeanServer();
22      }
23  
24      @GET
25      @Produces("text/plain")
26      public Response getMetrics() {
27          MetricsEnquirer metricsEnquirer = new MetricsEnquirer(mbs);
28          final StringBuilder formattedMetrics = new StringBuilder();
29  
30          // TODO Each app server has a /metrics endpoint but currently each endpoint returns the metrics for all servers.  Would be better to lock this down e.g. <p2puri>/metrics only returns the p2p metrics
31          for (AppType type : AppType.values()) {
32              List<MBeanMetric> metrics = metricsEnquirer.getMBeanMetrics(type);
33              PrometheusProtocolFormatter formatter = new PrometheusProtocolFormatter();
34  
35              formattedMetrics.append(formatter.format(metrics, type))
36                              .append("\n");
37          }
38  
39          return Response.status(Response.Status.OK)
40              .header("Content-Type", TEXT_PLAIN)
41              .entity(formattedMetrics.toString().trim())
42              .build();
43      }
44  }
