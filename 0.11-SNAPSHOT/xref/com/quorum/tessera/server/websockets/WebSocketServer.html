1   package com.quorum.tessera.server.websockets;
2   
3   import com.jpmorgan.quorum.server.utils.ServerUtils;
4   import com.quorum.tessera.config.ServerConfig;
5   import com.quorum.tessera.server.TesseraServer;
6   import java.util.Objects;
7   import java.util.Set;
8   import javax.websocket.server.ServerContainer;
9   import org.eclipse.jetty.server.Server;
10  import org.eclipse.jetty.servlet.ServletContextHandler;
11  import org.eclipse.jetty.websocket.jsr356.server.deploy.WebSocketServerContainerInitializer;
12  public class WebSocketServer implements TesseraServer {
13  
14      private final Server server;
15  
16      private final ServerConfig serverConfig;
17      
18      private final Set<Class<?>> services;
19      
20      public WebSocketServer(ServerConfig serverConfig,Set<Class<?>> services) {
21          this.serverConfig = Objects.requireNonNull(serverConfig);
22          this.server = ServerUtils.buildWebServer(serverConfig);
23          this.services = services;
24      }
25      
26      @Override
27      public void start() throws Exception {
28          
29          ServletContextHandler context = new ServletContextHandler(ServletContextHandler.SESSIONS);
30          context.setContextPath("/");
31          server.setHandler(context);
32          
33          ServerContainer websocketsContainer = WebSocketServerContainerInitializer.configureContext(context);
34          
35          for(Class<?> service : services) {
36              websocketsContainer.addEndpoint(service);
37          }
38          
39          
40          server.start();
41      }
42  
43      @Override
44      public void stop() throws Exception {
45          server.stop();
46      }
47  
48  }
