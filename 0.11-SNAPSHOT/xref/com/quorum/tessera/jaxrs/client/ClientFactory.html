10  import javax.net.ssl.SSLContext;
11  import javax.ws.rs.client.Client;
12  import javax.ws.rs.client.ClientBuilder;
13  import java.util.Objects;
14  import javax.ws.rs.core.Configuration;
15  
16  /**
17   * Creates HTTP clients that can make requests
18   *
19   * <p>Makes the client SSL enabled if specified
20   */
21  public class ClientFactory {
22  
23      private final SSLContextFactory sslContextFactory;
24  
25      public ClientFactory(final SSLContextFactory sslContextFactory) {
26          this.sslContextFactory = Objects.requireNonNull(sslContextFactory);
27      }
28  
29      public ClientFactory() {
30          this(com.quorum.tessera.ssl.context.ClientSSLContextFactory.create());
31      }
32  
33      private static Configuration createUnixServerSocketConfig() {
34  
35          return ReflectCallback.execute(
36                  () -> {
37                      Class configType = Class.forName("org.glassfish.jersey.client.ClientConfig");
38                      Class providerClass =
39                              Class.forName("com.quorum.tessera.jaxrs.unixsocket.JerseyUnixSocketConnectorProvider");
40                      Object config = configType.getDeclaredConstructor().newInstance();
41                      Object provider = providerClass.getDeclaredConstructor().newInstance();
42                      Method connectorProviderMethod =
43                              Arrays.stream(configType.getDeclaredMethods())
44                                      .filter(m -> m.getName().equals("connectorProvider"))
45                                      .findFirst()
46                                      .get();
47  
48                      connectorProviderMethod.invoke(config, provider);
49  
50                      return Configuration.class.cast(config);
51                  });
52      }
53  
54      /**
55       * Creates a new client, which may or may not be SSL enabled or a unix socket enabled depending on the
56       * configuration.
57       *
58       * @param config
59       * @return
60       * @see Client
61       */
62      public Client buildFrom(final ServerConfig config) {
63  
64          if (config.isUnixSocket()) {
65              Configuration clientConfig = createUnixServerSocketConfig();
66              URI unixfile = config.getServerUri();
67              return ClientBuilder.newClient(clientConfig).property("unixfile", unixfile);
68  
69          } else if (config.isSsl()) {
70              final SSLContext sslContext =
71                      sslContextFactory.from(config.getServerUri().toString(), config.getSslConfig());
72              return ClientBuilder.newBuilder().sslContext(sslContext).build();
73          } else {
74              return ClientBuilder.newClient();
75          }
76      }
77  }
