1   package com.quorum.tessera.cli.parsers;
2   
3   import org.slf4j.Logger;
4   import org.slf4j.LoggerFactory;
5   import picocli.CommandLine;
6   
7   import java.io.OutputStream;
8   import java.lang.management.ManagementFactory;
9   import java.nio.file.Files;
10  import java.nio.file.Path;
11  import java.util.concurrent.Callable;
12  
13  import static java.nio.charset.StandardCharsets.UTF_8;
14  import static java.nio.file.StandardOpenOption.CREATE;
15  import static java.nio.file.StandardOpenOption.TRUNCATE_EXISTING;
16  
17  public class PidFileMixin implements Callable<Boolean> {
18  
19      private static final Logger LOGGER = LoggerFactory.getLogger(PidFileMixin.class);
20  
21      @CommandLine.Option(names = "-pidfile", arity = "1", description = "the path to write the PID to")
22      private Path pidFilePath = null;
23  
24      @Override
25      public Boolean call() throws Exception {
26          if (pidFilePath == null) {
27              return true;
28          }
29  
30          if (Files.exists(pidFilePath)) {
31              LOGGER.info("File already exists {}", pidFilePath);
32          } else {
33              LOGGER.info("Created pid file {}", pidFilePath);
34          }
35  
36          final String pid = ManagementFactory.getRuntimeMXBean().getName().split("@")[0];
37  
38          try (OutputStream stream = Files.newOutputStream(pidFilePath, CREATE, TRUNCATE_EXISTING)) {
39              stream.write(pid.getBytes(UTF_8));
40          }
41  
42          return true;
43      }
44  
45      public void setPidFilePath(final Path pidFilePath) {
46          this.pidFilePath = pidFilePath;
47      }
48  }
