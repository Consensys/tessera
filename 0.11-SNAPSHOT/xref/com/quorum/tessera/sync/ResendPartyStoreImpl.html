3   import com.quorum.tessera.partyinfo.model.Party;
4   import com.quorum.tessera.sync.model.SyncableParty;

6   import java.util.*;
7   
8   /** An in-memory store of outstanding parties to contact for transaction resending */
9   public class ResendPartyStoreImpl implements ResendPartyStore {
10  
11      private Set<Party> allSeenParties;
12  
13      private Queue<SyncableParty> outstandingParties;
14  
15      public ResendPartyStoreImpl() {
16          this.outstandingParties = new LinkedList<>();
17          this.allSeenParties = new HashSet<>();
18      }
19  
20      @Override
21      public void addUnseenParties(final Collection<Party> partiesToRequestFrom) {
22          final Set<Party> knownParties = new HashSet<>(partiesToRequestFrom);
23          knownParties.removeAll(allSeenParties);
24          this.allSeenParties.addAll(knownParties);
25  
26          knownParties.stream().map(party -> new SyncableParty(party, 0)).forEach(outstandingParties::add);
27      }
28  
29      @Override
30      public Optional<SyncableParty> getNextParty() {
31          return Optional.ofNullable(this.outstandingParties.poll());
32      }
33  
34      @Override
35      public void incrementFailedAttempt(final SyncableParty attemptedParty) {
36  
37          if (attemptedParty.getAttempts() < MAX_ATTEMPTS) {
38              final SyncableParty updatedParty =
39                      new SyncableParty(attemptedParty.getParty(), attemptedParty.getAttempts() + 1);
40  
41              this.outstandingParties.add(updatedParty);
42          }
43      }
44  }
