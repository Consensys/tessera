20  public class PartyInfoGrpcService extends PartyInfoGrpc.PartyInfoImplBase {
21  
22      private final PartyInfoParser partyInfoParser;
23  
24      private final PartyInfoService partyInfoService;
25  
26      public PartyInfoGrpcService(PartyInfoService partyInfoService) {
27          this(partyInfoService, PartyInfoParser.create());
28      }
29  
30      public PartyInfoGrpcService(final PartyInfoService partyInfoService, final PartyInfoParser partyInfoParser) {
31          this.partyInfoService = requireNonNull(partyInfoService, "partyInfoService must not be null");
32          this.partyInfoParser = requireNonNull(partyInfoParser, "partyInfoParser must not be null");
33      }
34  
35      @Override
36      public void getPartyInfo(final PartyInfoMessage request, final StreamObserver<PartyInfoMessage> responseObserver) {
37  
38          final StreamObserverTemplate template = new StreamObserverTemplate(responseObserver);
39  
40          template.handle(
41                  () -> {
42                      final PartyInfo partyInfo = partyInfoParser.from(request.getPartyInfo().toByteArray());
43  
44                      final PartyInfo updatedPartyInfo = partyInfoService.updatePartyInfo(partyInfo);
45  
46                      return PartyInfoMessage.newBuilder()
47                              .setPartyInfo(ByteString.copyFrom(partyInfoParser.to(updatedPartyInfo)))
48                              .build();
49                  });
50      }
51  
52      @Override
53      public void getPartyInfoMessage(final Empty request, final StreamObserver<PartyInfoJson> responseObserver) {
54  
55          final StreamObserverTemplate template = new StreamObserverTemplate(responseObserver);
56  
57          template.handle(
58                  () -> {
59                      final PartyInfo partyInfo = partyInfoService.getPartyInfo();
60                      final Set<Peer> peers =
61                              partyInfo.getParties().stream()
62                                      .filter(p -> p.getUrl().endsWith("/"))
63                                      .map(
64                                              party -> {
65                                                  final Peer.Builder builder = Peer.newBuilder().setUrl(party.getUrl());
66  
67                                                  if (party.getLastContacted() != null) {
68                                                      final Timestamp timestamp =
69                                                              Timestamp.newBuilder()
70                                                                      .setSeconds(
71                                                                              party.getLastContacted().getEpochSecond())
72                                                                      .build();
73  
74                                                      builder.setUtcTimestamp(timestamp);
75                                                  }
76  
77                                                  return builder.build();
78                                              })
79                                      .collect(Collectors.toSet());
80  
81                      final Map<String, String> keys =
82                              partyInfo.getRecipients().stream()
83                                      .collect(toMap(r -> r.getKey().encodeToBase64(), Recipient::getUrl));
84  
85                      return PartyInfoJson.newBuilder()
86                              .setUrl(partyInfo.getUrl())
87                              .addAllPeers(peers)
88                              .putAllKeys(keys)
89                              .build();
90                  });
91      }
92  }
