1   package com.quorum.tessera.data.migration;
2   
3   import java.io.BufferedReader;
4   import java.io.IOException;
5   import java.io.InputStreamReader;
6   import java.nio.file.Path;
7   import java.sql.SQLException;
8   import java.util.List;
9   import java.util.stream.Collectors;
10  import java.util.stream.Stream;
11  
12  public class H2DataExporter implements DataExporter {
13  
14      private static final String INSERT_ROW = "INSERT INTO ENCRYPTED_TRANSACTION (HASH, ENCODED_PAYLOAD) VALUES (?, ?)";
15  
16      private static final String CREATE_TABLE_RESOURCE = "/ddls/h2-ddl.sql";
17  
18      @Override
19      public void export(final StoreLoader loader,
20                         final Path output,
21                         final String username,
22                         final String password) throws SQLException, IOException {
23  
24          final String connectionString = "jdbc:h2:" + output.toString();
25  
26          final List<String> createTableStatements = Stream.of(getClass().getResourceAsStream(CREATE_TABLE_RESOURCE))
27              .map(InputStreamReader::new)
28              .map(BufferedReader::new)
29              .flatMap(BufferedReader::lines)
30              .collect(Collectors.toList());
31  
32          final JdbcDataExporter jdbcDataExporter
33              = new JdbcDataExporter(connectionString, INSERT_ROW, createTableStatements);
34  
35          jdbcDataExporter.export(loader, output, username, password);
36      }
37  
38  }
