3   import org.apache.commons.io.IOUtils;
4   
5   import java.io.BufferedReader;
6   import java.io.IOException;
7   import java.io.InputStream;
8   import java.io.InputStreamReader;
9   import java.nio.file.Path;
10  import java.sql.*;
11  import java.util.List;
12  import java.util.stream.Collectors;
13  import java.util.stream.Stream;
14  
15  public class SqliteDataExporter implements DataExporter {
16  
17      private static final String INSERT_ROW = "INSERT INTO ENCRYPTED_TRANSACTION (HASH, ENCODED_PAYLOAD) VALUES (?, ?)";
18  
19      private static final String CREATE_TABLE_RESOURCE = "/ddls/sqlite-ddl.sql";
20  
21      @Override
22      public void export(final StoreLoader loader,
23                         final Path output,
24                         final String username,
25                         final String password) throws SQLException, IOException {
26  
27          final String connectionString = "jdbc:sqlite:" + output.toString();
28  
29          final List<String> createTableStatements = Stream.of(getClass().getResourceAsStream(CREATE_TABLE_RESOURCE))
30              .map(InputStreamReader::new)
31              .map(BufferedReader::new)
32              .flatMap(BufferedReader::lines)
33              .collect(Collectors.toList());
34  
35          try (Connection conn = DriverManager.getConnection(connectionString, username, password)) {
36  
37              try (Statement stmt = conn.createStatement()) {
38                  for (final String createTable : createTableStatements) {
39                      stmt.executeUpdate(createTable);
40                  }
41              }
42  
43              try (PreparedStatement insertStatement = conn.prepareStatement(INSERT_ROW)) {
44                  DataEntry next;
45                  while ((next = loader.nextEntry()) != null) {
46                      try (InputStream data = next.getValue()) {
47                          insertStatement.setBytes(1, next.getKey());
48                          insertStatement.setBytes(2, IOUtils.toByteArray(data));
49                          insertStatement.execute();
50                      }
51                  }
52              }
53  
54          }
55  
56      }
57  
58  }
