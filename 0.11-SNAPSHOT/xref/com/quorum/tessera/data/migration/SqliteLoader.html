3   import java.io.ByteArrayInputStream;
4   import java.nio.file.Path;
5   import java.sql.*;
6   
7   public class SqliteLoader implements StoreLoader {
8   
9       private static final String SELECT_QUERY = "SELECT * FROM payload";
10  
11      private Connection connection;
12  
13      private Statement statement;
14  
15      private ResultSet results;
16  
17      @Override
18      public void load(final Path input) throws SQLException {
19          final String url = "jdbc:sqlite:" + input.toString();
20  
21          this.connection = DriverManager.getConnection(url);
22          this.statement = this.connection.createStatement();
23          this.results = this.statement.executeQuery(SELECT_QUERY);
24      }
25  
26      @Override
27      public DataEntry nextEntry() throws SQLException {
28          final boolean hasNextEntry = this.results.next();
29          if (!hasNextEntry) {
30              this.results.close();
31              this.statement.close();
32              this.connection.close();
33              return null;
34          }
35  
36          return new DataEntry(results.getBytes("key"), new ByteArrayInputStream(results.getBytes("bytes")));
37      }
38  
39  }
