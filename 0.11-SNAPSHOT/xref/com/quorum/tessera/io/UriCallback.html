7   import java.io.IOException;
8   import java.io.UncheckedIOException;
9   import java.net.URISyntaxException;
10  
11  public interface UriCallback<T> {
12  
13      Logger LOGGER = LoggerFactory.getLogger(ReflectCallback.class);
14  
15      T doExecute() throws URISyntaxException;
16  
17      static <T> T execute(UriCallback<T> callback) {
18          try {
19              return callback.doExecute();
20          } catch (URISyntaxException ex) {
21              LOGGER.error(null, ex);
22              throw new UncheckedIOException(new IOException(ex));
23          }
24      }
25  
26  }
