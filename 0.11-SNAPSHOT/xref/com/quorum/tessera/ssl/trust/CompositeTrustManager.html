1   package com.quorum.tessera.ssl.trust;
2   
3   import javax.net.ssl.TrustManager;
4   import javax.net.ssl.X509TrustManager;
5   import java.security.cert.CertificateException;
6   import java.security.cert.X509Certificate;
7   
8   /**
9    * If the certificate is trusted by one of the given trust managers,
10   * the connection will be accepted.
11   */
12  public class CompositeTrustManager extends AbstractTrustManager {
13  
14      private TrustManager[] trustManagers;
15  
16      public CompositeTrustManager(TrustManager[] trustManagers) {
17          this.trustManagers = trustManagers.clone();
18      }
19  
20      @Override
21      public void checkClientTrusted(X509Certificate[] clientCertificates, String authType) throws CertificateException {
22          for (TrustManager trustManager : trustManagers) {
23              try {
24                  X509TrustManager x509TrustManager = (X509TrustManager) trustManager;
25                  x509TrustManager.checkClientTrusted(clientCertificates, authType);
26                  return;
27              }
28              catch(CertificateException ex) {
29                  //Ignore and move on to the next trust manager
30              }
31          }
32          throw new CertificateException("Certificate is not trusted by any of the trust managers");
33      }
34  
35      @Override
36      public void checkServerTrusted(X509Certificate[] serverCertificates, String authType) throws CertificateException {
37          for (TrustManager trustManager : trustManagers) {
38              try {
39                  final X509TrustManager x509TrustManager = (X509TrustManager) trustManager;
40                  x509TrustManager.checkServerTrusted(serverCertificates, authType);
41                  return;
42              }
43              catch (CertificateException ex) {
44                  //Ignore and move on to the next trust manager
45              }
46          }
47          throw new CertificateException("Certificate is not trusted by any of the trust managers");
48      }
49  
50      @Override
51      public X509Certificate[] getAcceptedIssuers() {
52          return new X509Certificate[0];
53      }
54  
55  }
