27  public interface TlsUtils {
28  
29      String ENCRYPTION = "RSA";
30      String COMMON_NAME_STRING = "CN=";
31      String SIGNATURE_ALGORITHM = "SHA512WithRSAEncryption";
32      String KEYSTORE_TYPE = "JKS";
33  
34      // These localhost values are added in SAN names for dev. They can be removed for production
35      // if necessary
36      String LOCALHOST = "localhost";
37      String LOCALHOST_IP = "127.0.0.1";
38      String LOCALHOST_IP_2 = "0.0.0.0";
39  
40      Provider provider = new BouncyCastleProvider();
41  
42      default void generateKeyStoreWithSelfSignedCertificate(String address, Path privateKeyFile, String password)
43          throws NoSuchAlgorithmException, IOException, OperatorCreationException,
44          CertificateException, InvalidKeyException, NoSuchProviderException, SignatureException, KeyStoreException {
45  
46          final SecureRandom secureRandom = new SecureRandom();
47  
48          KeyPairGenerator keyGen = KeyPairGenerator.getInstance(ENCRYPTION);
49          keyGen.initialize(2048, secureRandom);
50          KeyPair keypair = keyGen.generateKeyPair();
51          final PublicKey publicKey = keypair.getPublic();
52          final PrivateKey privateKey = keypair.getPrivate();
53          final String cnString = address.replaceFirst("^(http[s]?://www\\.|http[s]?://|www\\.)","");
54          final X500Name commonName = new X500Name(COMMON_NAME_STRING + cnString);
55          Date startDate = new Date(System.currentTimeMillis());
56          Calendar calendar = Calendar.getInstance();
57          calendar.setTime(startDate);
58          calendar.add(Calendar.YEAR, 1);
59          Date endDate = calendar.getTime();
60  
61          X509v3CertificateBuilder builder = new JcaX509v3CertificateBuilder(
62              commonName,
63              new BigInteger(64, secureRandom),
64              startDate,
65              endDate,
66              commonName,
67              publicKey);
68  
69          GeneralName[] subjectAlternativeNames = new GeneralName[] {
70              new GeneralName(GeneralName.dNSName, LOCALHOST),
71              new GeneralName(GeneralName.dNSName, HostnameUtil.create().getHostName()),
72              new GeneralName(GeneralName.iPAddress, LOCALHOST_IP),
73              new GeneralName(GeneralName.iPAddress, LOCALHOST_IP_2),
74              new GeneralName(GeneralName.iPAddress, HostnameUtil.create().getHostIpAddress())
75          };
76  
77          builder.addExtension(
78              Extension.subjectAlternativeName,
79              false,
80              new GeneralNames(subjectAlternativeNames));
81  
82          ContentSigner contentSigner = new JcaContentSignerBuilder(SIGNATURE_ALGORITHM).build(privateKey);
83          X509CertificateHolder certHolder = builder.build(contentSigner);
84          X509Certificate certificate =
85              new JcaX509CertificateConverter().setProvider(provider).getCertificate(certHolder);
86  
87          certificate.verify(publicKey);
88  
89          KeyStore keyStore = KeyStore.getInstance(KEYSTORE_TYPE);
90          keyStore.load(null, null);
91          keyStore.setKeyEntry("tessera",privateKey, password.toCharArray(), new X509Certificate[]{certificate});
92  
93          try(OutputStream keyStoreFile = Files.newOutputStream(privateKeyFile)) {
94              keyStore.store(keyStoreFile, password.toCharArray());
95          }
96      }
97  
98      static TlsUtils create(){
99          Security.addProvider(new BouncyCastleProvider());
100         return new TlsUtils() {
101         };
102     }
103 
104 }
