13  import java.net.URL;
14  import java.util.Set;

16  import org.slf4j.Logger;
17  import org.slf4j.LoggerFactory;

19  /**
20   * Applies a filter to all endpoints that only allows certain IP address and ghost names to get access to the HTTP
21   * endpoints
22   *
23   * <p>If an error occurs whilst checking the whitelist, the filter is disabled. This is done since not all webservers
24   * have support for the {@link HttpServletRequest} context class, which is required.
25   */
26  @GlobalFilter
27  public class IPWhitelistFilter implements ContainerRequestFilter {
28  
29      private static final Logger LOGGER = LoggerFactory.getLogger(IPWhitelistFilter.class);
30  
31      private final ConfigService configService;
32  
33      private boolean disabled;
34  
35      private HttpServletRequest httpServletRequest;
36  
37      public IPWhitelistFilter() {
38          this(ServiceFactory.create().configService());
39      }
40  
41      protected IPWhitelistFilter(ConfigService configService) {
42          this.configService = configService;
43          this.disabled = !configService.isUseWhiteList();
44      }
45  
46      /**
47       * If the filter is disabled, return immediately Otherwise, extract the callers hostname and address, and check it
48       * against the whitelist
49       *
50       * <p>If a problem occurs, then disable the filter
51       *
52       * <p>If the host is not whitelisted, finish the filter chain here and return an Unauthorized response
53       *
54       * @param requestContext the context of the current request
55       */
56      @Override
57      public void filter(final ContainerRequestContext requestContext) {
58  
59          if (disabled) {
60              return;
61          }
62  
63          // this is the unix socket request, so let it through the filter
64          if ("unixsocket".equals(requestContext.getUriInfo().getBaseUri().toString())) {
65              return;
66          }
67  
68          try {
69  
70              final Set<String> whitelisted =
71                      configService.getPeers().stream()
72                              .map(Peer::getUrl)
73                              .map(s -> IOCallback.execute(() -> new URL(s)))
74                              .map(URL::getHost)
75                              .collect(Collectors.toSet());
76  
77              final String remoteAddress = httpServletRequest.getRemoteAddr();
78              final String remoteHost = httpServletRequest.getRemoteHost();
79  
80              final boolean allowed = whitelisted.contains(remoteAddress) || whitelisted.contains(remoteHost);
81  
82              if (!allowed) {
83                  requestContext.abortWith(Response.status(Response.Status.UNAUTHORIZED).build());
84              }
85  
86          } catch (final Exception ex) {
87              LOGGER.error("Unexpected error while processing request.", ex);
88              this.disabled = true;
89          }
90      }
91  
92      /**
93       * Apply the current HTTP context to the filter, to check the remote host
94       *
95       * @param request the request to be filtered
96       */
97      @Context
98      public void setHttpServletRequest(final HttpServletRequest request) {
99          this.httpServletRequest = request;
100     }
101 }
