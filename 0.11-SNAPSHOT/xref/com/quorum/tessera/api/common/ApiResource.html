1   package com.quorum.tessera.api.common;
2   
3   import io.swagger.annotations.Api;
4   import io.swagger.annotations.ApiResponse;
5   import io.swagger.annotations.ApiResponses;
6   
7   import javax.ws.rs.GET;
8   import javax.ws.rs.Path;
9   import javax.ws.rs.Produces;
10  import javax.ws.rs.core.Context;
11  import javax.ws.rs.core.Request;
12  import javax.ws.rs.core.Response;
13  import javax.ws.rs.core.Response.Status;
14  import javax.ws.rs.core.Variant;
15  import java.io.IOException;
16  import java.net.URL;
17  import java.util.List;
18  
19  import static javax.ws.rs.core.MediaType.*;
20  
21  /** Provides HTTP endpoints for accessing the OpenAPI schema */
22  @Path("/api")
23  @Api("Provides access to OpenAPI schema documentation.")
24  public class ApiResource {
25  
26      private static final List<Variant> VARIANTS = Variant.mediaTypes(APPLICATION_JSON_TYPE, TEXT_HTML_TYPE).build();
27  
28      @GET
29      @Produces({APPLICATION_JSON, TEXT_HTML})
30      @ApiResponses({@ApiResponse(code = 200, message = "Returns JSON or HTML OpenAPI document")})
31      public Response api(@Context final Request request) throws IOException {
32  
33          final Variant variant = request.selectVariant(VARIANTS);
34  
35          final URL url;
36          if (variant.getMediaType() == APPLICATION_JSON_TYPE) {
37              url = getClass().getResource("/swagger.json");
38  
39          } else if (variant.getMediaType() == TEXT_HTML_TYPE) {
40              url = getClass().getResource("/swagger.html");
41  
42          } else {
43  
44              return Response.status(Status.BAD_REQUEST).build();
45          }
46  
47          return Response.ok(url.openStream(), variant.getMediaType()).build();
48      }
49  }
