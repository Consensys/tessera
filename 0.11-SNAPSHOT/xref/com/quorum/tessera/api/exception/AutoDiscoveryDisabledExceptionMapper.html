3   import com.quorum.tessera.partyinfo.AutoDiscoveryDisabledException;
4   import org.slf4j.Logger;
5   import org.slf4j.LoggerFactory;

7   import javax.ws.rs.core.MediaType;
8   import javax.ws.rs.core.Response;
9   import javax.ws.rs.ext.ExceptionMapper;
10  import javax.ws.rs.ext.Provider;
11  
12  @Provider
13  public class AutoDiscoveryDisabledExceptionMapper implements ExceptionMapper<AutoDiscoveryDisabledException> {
14  
15      private static final Logger LOGGER = LoggerFactory.getLogger(AutoDiscoveryDisabledExceptionMapper.class);
16  
17      @Override
18      public Response toResponse(final AutoDiscoveryDisabledException exception) {
19          LOGGER.debug(null, exception);
20  
21          return Response.status(Response.Status.FORBIDDEN)
22                  .entity(exception.getMessage())
23                  .type(MediaType.TEXT_PLAIN)
24                  .build();
25      }
26  }
