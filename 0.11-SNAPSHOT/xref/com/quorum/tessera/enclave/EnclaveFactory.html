16  /**
17   * Creates {@link Enclave} instances, which may point to remote services or
18   * local, in-app instances.
19   */
20  public interface EnclaveFactory {
21  
22      default Enclave createLocal(Config config) {
23          return createServer(config);
24      }
25  
26      static Enclave createServer(Config config) {
27  
28          final KeyPairConverter keyPairConverter = new KeyPairConverter(config, new EnvironmentVariableProvider());
29          final Collection<KeyPair> keys = keyPairConverter.convert(config.getKeys().getKeyData());
30  
31          final Collection<PublicKey> forwardKeys = keyPairConverter.convert(config.getAlwaysSendTo());
32  
33          return new EnclaveImpl(NaclFacadeFactory.newFactory().create(), new KeyManagerImpl(keys, forwardKeys));
34      }
35  
36      /**
37       * Determines from the provided configuration whether to construct a client
38       * to a remote service, or to create a local instance.
39       * <p>
40       * If a remote instance is requested, it is constructed from a
41       * {@link EnclaveClientFactory}.
42       *
43       * @param config the global configuration to use to create a remote enclave connection
44       * @return the {@link Enclave}, which may be either local or remote
45       */
46      default Enclave create(Config config) {
47          final Optional<ServerConfig> enclaveServerConfig = config
48              .getServerConfigs()
49              .stream()
50              .filter(sc -> sc.getApp() == AppType.ENCLAVE)
51              .findAny();
52  
53          if (enclaveServerConfig.isPresent()) {
54              return EnclaveClientFactory.create().create(config);
55          }
56  
57          return createServer(config);
58  
59      }
60  
61      static EnclaveFactory create() {
62          return ServiceLoaderUtil.load(EnclaveFactory.class).orElse(new EnclaveFactory() {
63          });
64      }
65  
66  }
