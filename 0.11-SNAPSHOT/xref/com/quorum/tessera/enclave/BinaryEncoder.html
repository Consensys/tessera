1   package com.quorum.tessera.enclave;
2   
3   import java.nio.ByteBuffer;
4   import java.util.List;
5   
6   /**
7    * Encodes bytes to the binary form expected by other nodes on the network
8    */
9   public interface BinaryEncoder {
10  
11      /**
12       * Converts a byte array to its binary form, which takes the following:
13       * 8 bytes to describe the number of bytes long the data is
14       * the bytes of the data
15       *
16       * @param data the input array to encode
17       * @return the encoded byte array
18       */
19      default byte[] encodeField(final byte[] data) {
20          return ByteBuffer
21              .allocate(Long.BYTES + data.length)
22              .putLong(data.length)
23              .put(data)
24              .array();
25      }
26  
27      /**
28       * Converts a list of byte arrays to its binary form, which takes the following:
29       *
30       * 8 bytes to describe the number of elements
31       * and for each element:
32       * - 8 bytes to describe the number of bytes long the element is
33       * - the bytes of the element
34       *
35       * @param data the input array to encode
36       * @return the encoded byte array
37       */
38      default byte[] encodeArray(final List<byte[]> data) {
39          final byte[] elements = data
40              .stream()
41              .map(this::encodeField)
42              .reduce(new byte[0], (a, b) -> {
43                  final byte[] concat = new byte[a.length + b.length];
44                  System.arraycopy(a, 0, concat, 0, a.length);
45                  System.arraycopy(b, 0, concat, a.length, b.length);
46                  return concat;
47              });
48  
49          return ByteBuffer
50              .allocate(Long.BYTES + elements.length)
51              .putLong(data.size())
52              .put(elements)
53              .array();
54      }
55  
56  }
