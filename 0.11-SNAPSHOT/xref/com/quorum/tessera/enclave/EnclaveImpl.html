20      public EnclaveImpl(NaclFacade nacl, KeyManager keyManager) {

25      @Override
26      public EncodedPayload encryptPayload(final byte[] message,
27                                           final PublicKey senderPublicKey,
28                                           final List<PublicKey> recipientPublicKeys) {
29          final MasterKey masterKey = nacl.createMasterKey();
30          final Nonce nonce = nacl.randomNonce();
31          final Nonce recipientNonce = nacl.randomNonce();
32  
33          final byte[] cipherText = nacl.sealAfterPrecomputation(message, nonce, masterKey);
34  
35          final List<byte[]> encryptedMasterKeys = buildRecipientMasterKeys(senderPublicKey, recipientPublicKeys, recipientNonce, masterKey);
36  
37          return new EncodedPayload(
38              senderPublicKey, cipherText, nonce, encryptedMasterKeys, recipientNonce, recipientPublicKeys
39          );
40      }
41  
42      @Override
43      public byte[] createNewRecipientBox(final EncodedPayload payload, final PublicKey publicKey) {
44  
45          if (payload.getRecipientKeys().isEmpty() || payload.getRecipientBoxes().isEmpty()) {
46              throw new RuntimeException("No key or recipient-box to use");
47          }
48  
49          final MasterKey master = this.getMasterKey(
50              payload.getRecipientKeys().get(0), payload.getSenderKey(),
51              payload.getRecipientNonce(), payload.getRecipientBoxes().get(0)
52          );
53  
54          final List<byte[]> sealedMasterKeyList = this.buildRecipientMasterKeys(
55              payload.getSenderKey(), singletonList(publicKey), payload.getRecipientNonce(), master
56          );
57  
58          return sealedMasterKeyList.get(0);
59      }
60  
61      @Override
62      public EncodedPayload encryptPayload(final RawTransaction rawTransaction,
63                                           final List<PublicKey> recipientPublicKeys) {
64          final MasterKey masterKey = this.getMasterKey(
65              rawTransaction.getFrom(), rawTransaction.getFrom(),
66              rawTransaction.getNonce(), rawTransaction.getEncryptedKey()
67          );
68  
69          final Nonce recipientNonce = nacl.randomNonce();
70          final List<byte[]> encryptedMasterKeys
71              = buildRecipientMasterKeys(rawTransaction.getFrom(), recipientPublicKeys, recipientNonce, masterKey);
72  
73          return new EncodedPayload(
74              rawTransaction.getFrom(), rawTransaction.getEncryptedPayload(),
75              rawTransaction.getNonce(), encryptedMasterKeys, recipientNonce, recipientPublicKeys
76          );
77      }
78  
79      private List<byte[]> buildRecipientMasterKeys(final PublicKey senderPublicKey,
80                                                    final List<PublicKey> recipientPublicKeys,
81                                                    final Nonce recipientNonce,
82                                                    final MasterKey masterKey){
83          final PrivateKey privateKey = keyManager.getPrivateKeyForPublicKey(senderPublicKey);
84  
85          return recipientPublicKeys
86              .stream()
87              .map(publicKey -> nacl.computeSharedKey(publicKey, privateKey))
88              .map(sharedKey -> nacl.sealAfterPrecomputation(masterKey.getKeyBytes(), recipientNonce, sharedKey))
89              .collect(Collectors.toList());
90      }
91  
92      @Override
93      public RawTransaction encryptRawPayload(byte[] message, PublicKey sender) {
94          final MasterKey masterKey = nacl.createMasterKey();
95          final Nonce nonce = nacl.randomNonce();
96  
97          final byte[] cipherText = nacl.sealAfterPrecomputation(message, nonce, masterKey);
98  
99          final PrivateKey privateKey = keyManager.getPrivateKeyForPublicKey(sender);
100 
101         // TODO NL - check if it makes sense to compute a shared key from the public and private parts of the same key
102         SharedKey sharedKey = nacl.computeSharedKey(sender, privateKey);
103         final byte[] encryptedMasterKey = nacl.sealAfterPrecomputation(masterKey.getKeyBytes(), nonce, sharedKey);
104 
105         return new RawTransaction(cipherText, encryptedMasterKey, nonce, sender);
106     }

108     @Override
109     public byte[] unencryptTransaction(EncodedPayload payload, final PublicKey providedSenderKey) {
110 
111         final PublicKey senderPubKey;
112 
113         final PublicKey recipientPubKey;
114 
115         if (!this.getPublicKeys().contains(payload.getSenderKey())) {
116             // This is a payload originally sent to us by another node
117             senderPubKey = providedSenderKey;
118             recipientPubKey = payload.getSenderKey();
119         } else {
120             // This is a payload that originated from us
121             senderPubKey = payload.getSenderKey();
122             recipientPubKey = payload.getRecipientKeys().get(0);
123         }
124 
125         final PrivateKey senderPrivKey = keyManager.getPrivateKeyForPublicKey(senderPubKey);
126 
127         final SharedKey sharedKey = nacl.computeSharedKey(recipientPubKey, senderPrivKey);
128 
129         final byte[] recipientBox = payload.getRecipientBoxes().iterator().next();
130 
131         final Nonce recipientNonce = payload.getRecipientNonce();
132 
133         final byte[] masterKeyBytes = nacl.openAfterPrecomputation(recipientBox, recipientNonce, sharedKey);
134 
135         final MasterKey masterKey = MasterKey.from(masterKeyBytes);
136 
137         final byte[] cipherText = payload.getCipherText();
138         final Nonce cipherTextNonce = payload.getCipherTextNonce();
139 
140         return nacl.openAfterPrecomputation(cipherText, cipherTextNonce, masterKey);
141 
142     }
143 
144     private MasterKey getMasterKey(PublicKey recipient, PublicKey sender, Nonce nonce, byte[] encryptedKey) {
145 
146         final SharedKey sharedKey = nacl.computeSharedKey(recipient, keyManager.getPrivateKeyForPublicKey(sender));
147 
148         final byte[] masterKeyBytes = nacl.openAfterPrecomputation(encryptedKey, nonce, sharedKey);
149 
150         return MasterKey.from(masterKeyBytes);
151     }
