001package com.quorum.tessera.api.exception;
002
003import org.slf4j.Logger;
004import org.slf4j.LoggerFactory;
005
006import javax.ws.rs.WebApplicationException;
007import javax.ws.rs.core.MediaType;
008import javax.ws.rs.core.Response;
009import javax.ws.rs.ext.ExceptionMapper;
010import javax.xml.bind.UnmarshalException;
011
012public class WebApplicationExceptionMapper implements ExceptionMapper<WebApplicationException> {
013
014    private static final Logger LOGGER = LoggerFactory.getLogger(DefaultExceptionMapper.class);
015
016    @Override
017    public Response toResponse(final WebApplicationException exception) {
018
019        LOGGER.debug("{}", exception.getClass());
020        LOGGER.debug("{}", exception.getCause() == null ? "No cause" : exception.getCause().getClass());
021
022        LOGGER.error("{}", exception.getMessage());
023
024        try {
025            throw exception.getCause();
026
027        } catch (final UnmarshalException ex) {
028            LOGGER.warn("Unable to unmarshal payload");
029            return Response.status(Response.Status.BAD_REQUEST)
030                    .entity(exception.getMessage())
031                    .type(MediaType.TEXT_PLAIN)
032                    .build();
033
034        } catch (final Throwable ex) {
035            return Response.status(Response.Status.INTERNAL_SERVER_ERROR)
036                    .entity(exception.getMessage())
037                    .type(MediaType.TEXT_PLAIN)
038                    .build();
039
040        }
041
042    }
043
044}
