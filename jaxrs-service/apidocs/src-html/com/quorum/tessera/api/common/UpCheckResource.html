001package com.quorum.tessera.api.common;
002
003import io.swagger.annotations.ApiResponse;
004import io.swagger.annotations.ApiResponses;
005import io.swagger.annotations.ApiOperation;
006
007import javax.ws.rs.GET;
008import javax.ws.rs.Path;
009import javax.ws.rs.Produces;
010import javax.ws.rs.core.MediaType;
011
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014
015/**
016 * Provides endpoints about the health status of this node
017 */
018@Path("/upcheck")
019public class UpCheckResource {
020
021    private static final Logger LOGGER = LoggerFactory.getLogger(UpCheckResource.class);
022
023    private static final String UPCHECK_RESPONSE = "I'm up!";
024
025    /**
026     * Called to check if the application is running and responsive
027     * Gives no details about the health of the application other than it is up
028     *
029     * @return a string stating the application is running
030     */
031    @GET
032    @Produces(MediaType.TEXT_PLAIN)
033    @ApiResponses({@ApiResponse(code = 200, message = UPCHECK_RESPONSE)})
034    @ApiOperation(value = "Check if local P2PRestApp Node is up", produces = "I'm up")
035    public String upCheck() {
036
037        LOGGER.info("GET upcheck");
038
039        return UPCHECK_RESPONSE;
040    }
041}
