001package com.quorum.tessera.client;
002
003import com.quorum.tessera.config.CommunicationType;
004import com.quorum.tessera.config.Config;
005import com.quorum.tessera.jaxrs.client.ClientFactory;
006import com.quorum.tessera.ssl.context.ClientSSLContextFactory;
007import com.quorum.tessera.ssl.context.SSLContextFactory;
008import javax.ws.rs.client.Client;
009
010public class RestP2pClientFactory implements P2pClientFactory {
011
012    @Override
013    public P2pClient create(Config config) {
014
015        SSLContextFactory clientSSLContextFactory = ClientSSLContextFactory.create();
016        
017        ClientFactory clientFactory = new ClientFactory(clientSSLContextFactory);
018
019        Client client = clientFactory.buildFrom(config.getP2PServerConfig());
020
021        PostDelegate postDelegate = new PostDelegate(client);
022
023        return new RestP2pClient(postDelegate);
024        
025    }
026
027    @Override
028    public CommunicationType communicationType() {
029        return CommunicationType.REST;
030    }
031
032}
