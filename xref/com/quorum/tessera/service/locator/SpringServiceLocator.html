1   package com.quorum.tessera.service.locator;
2   
3   import org.springframework.context.ApplicationContext;
4   import org.springframework.context.support.ClassPathXmlApplicationContext;
5   
6   import java.util.Set;
7   import java.util.stream.Collectors;
8   import java.util.stream.Stream;
9   
10  /**
11   * A Spring implementation of the Service Locator
12   * that accepts xml bean definition files
13   */
14  public class SpringServiceLocator implements ServiceLocator {
15  
16      private static ApplicationContext context;
17  
18      /**
19       * If the Spring context is already established, then returns the
20       * previously generated set of beans
21       *
22       * {@inheritDoc}
23       */
24      @Override
25      public Set<Object> getServices(final String filename) {
26          if(context == null) {
27              context = new ClassPathXmlApplicationContext(filename);
28          }
29  
30          return Stream
31              .of(context.getBeanDefinitionNames())
32              .map(context::getBean)
33              .collect(Collectors.toSet());
34      }
35  
36  }
