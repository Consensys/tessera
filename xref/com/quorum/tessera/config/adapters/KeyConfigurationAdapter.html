1   package com.quorum.tessera.config.adapters;

3   import com.quorum.tessera.config.KeyConfiguration;
4   import com.quorum.tessera.config.keypairs.ConfigKeyPair;
5   import com.quorum.tessera.io.SystemAdapter;
6   
7   import javax.xml.bind.annotation.adapters.XmlAdapter;
8   import java.io.IOException;
9   import java.nio.charset.StandardCharsets;
10  import java.nio.file.Files;
11  import java.util.ArrayList;
12  import java.util.List;
13  import java.util.stream.Collectors;
14  import java.util.stream.IntStream;
15  
16  public class KeyConfigurationAdapter extends XmlAdapter<KeyConfiguration, KeyConfiguration> {
17  
18      @Override
19      public KeyConfiguration unmarshal(final KeyConfiguration input) {
20  
21          final List<String> allPasswords = new ArrayList<>();
22          if (input.getPasswords() != null) {
23              allPasswords.addAll(input.getPasswords());
24          } else if (input.getPasswordFile() != null) {
25              try {
26                  allPasswords.addAll(Files.readAllLines(input.getPasswordFile(), StandardCharsets.UTF_8));
27              } catch (final IOException ex) {
28                  //dont do anything, if any keys are locked validation will complain that
29                  //locked keys were provided without passwords
30                  SystemAdapter.INSTANCE.err().println("Could not read the password file");
31              }
32          }
33  
34          final List<ConfigKeyPair> keyDataWithPasswords;
35          if (allPasswords.isEmpty()) {
36              keyDataWithPasswords = input.getKeyData();
37          } else {
38              keyDataWithPasswords = IntStream
39                  .range(0, input.getKeyData().size())
40                  .mapToObj(i -> {
41  
42                      final ConfigKeyPair kd = input.getKeyData().get(i);
43                      if(i < allPasswords.size()) {
44                          kd.withPassword(allPasswords.get(i));
45                      }
46  
47                      return kd;
48                  }).collect(Collectors.toList());
49          }
50  
51          return new KeyConfiguration(input.getPasswordFile(), input.getPasswords(), keyDataWithPasswords, input.getAzureKeyVaultConfig(), input.getHashicorpKeyVaultConfig());
52      }
53  
54      @Override
55      public KeyConfiguration marshal(final KeyConfiguration input) {
56          return input;
57      }
58  
59  }
